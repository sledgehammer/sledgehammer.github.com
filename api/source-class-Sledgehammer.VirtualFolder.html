 66     function __construct() {
 67         $methods = get_public_methods($this);
 68         foreach ($methods as $index => $method) {
 69             if (substr($method, 0, 1) == '_') {
 70                 unset($methods[$index]); // Functies die beginnen met een "_" uit de publicMethods halen
 71             }
 72         }
 73         $this->publicMethods = array_diff($methods, array('execute', 'getPath', 'dynamicFilename', 'dynamicFoldername', 'onFileNotFound', 'onFolderNotFound', 'getDocument')); // Een aantal funties minder public maken
 74     }

 76     /**
 77      * Aan de hand van de url de betreffende action functie aanroepen.
 78      * Valt terug op dynamicFilename() en dynamicFoldername() functies, als de geen action functie gevonden wordt.
 79      *
 80      * @return View
 81      */
 82     function generateContent() {
 83         $this->initDepth();
 84         $url = URL::getCurrentURL();
 85         $folders = $url->getFolders();
 86         $filename = $url->getFilename();
 87         $folder_count = count($folders);
 88         if ($folder_count == $this->depth) {
 89             $extension = file_extension($filename, $file);
 90             if ($extension === NULL && $this->handle_filenames_without_extension == false) { // Ongeldige bestandsnaam? (geen  extentie)
 91                 error_log('filename without extension, redirecting to "'.$filename.'/"', E_NOTICE);
 92                 return new Redirect($filename.'/'); // Redirect naar dezelfde url, maar dan als mapnaam
 93             }
 94             if ($this->publicMethods === null) {
 95                 notice('Check if the parent::__construct() is called in '.get_class($this)."__construct()");
 96             }
 97             $function = str_replace('-', '_', $file);
 98             if (substr($function, -7) != '_folder' && in_array($function, $this->publicMethods)) {
 99                 return $this->$function($extension); // Roept bijvoorbeeld de $this->index('html') functie aan.
100             }
101             return $this->dynamicFilename($filename);
102         }
103         if ($folder_count > $this->depth) {
104             if ($folder_count != ($this->depth + 1)) {
105                 $filename = false;; // Deze submap heeft nog 1 of meer submappen.
106             }
107             $folder = $folders[$this->depth];
108             $function = str_replace('-', '_', $folder).'_folder';
109             if (in_array($function, $this->publicMethods)) {
110                 return $this->$function($filename); // Roept bijvoorbeeld de $this->fotos_folder('index.html') functie aan.
111             }
112             return $this->dynamicFoldername($folder, $filename);
113         }
114         warning('Not enough (virtual) subfolders in URI', 'VirtualFolder depth('.$this->depth.') exceeds maximum('.count($folders).')');
115         return $this->onFolderNotFound(); // @todo eigen event?
116     }

118     /**
119      * Het pad opvragen van deze VirtualFolder
120      *
121      * @param bool $includeSubfolders  De actieve submap(pen) aan het path toevoegen (mappen die door deze VirtualFolder worden afgehandeld)
122      */
123     function getPath($includeSubfolders = false) {
124         $this->initDepth();
125         $folders = URL::getCurrentURL()->getFolders();
126         $path = '/';
127         for($i = 0; $i < $this->depth; $i++) {
128             $path .= $folders[$i].'/';
129         }
130         if ($includeSubfolders) {
131             for($i = $this->depth; $i < ($this->depth + $this->depthIncrement); $i++) {
132                 if (empty($folders[$i])) {
133                     break;
134                 }
135                 $path .= $folders[$i].'/';
136             }
137         }
138         return $path;
139     }

192     /**
193      * Event/Action voor het afhandelen van niet bestaande (virtuele) mappen.
194      * Geeft deze of een parent van deze virtualfolder de mogenlijkheid om een custom actie uit te voeren.
195      *
196      * @return HttpError
197      */
198     protected function onFolderNotFound() {
199         if ($this->parent !== null) {
200             return $this->parent->onFolderNotFound();
201         }
202         $url = URL::getCurrentURL();
203         $relativePath = substr(rawurldecode($url->path), strlen(WEBPATH) - 1); // Relative path vanaf de WEBROOT
204         $isFolder = (substr($relativePath, -1) == '/'); // Gaat de request om een folder?
205         if ($isFolder) {
206             $folder = $relativePath;
207         } else {
208             $folder = dirname($relativePath).'/';
209         }
210         $publicFolder = array(PATH.'application/public'.$folder);
211         $folders = explode('/', substr($folder, 1, -1));
212         if (count($folders) != 0) {
213             $module = $folders[0];
214             $modules = Framework::getModules();
215             if (isset($modules[$module])) {
216                 $publicFolder[] = $modules[$module]['path'].'public'.substr($folder, strlen($module) + 1);
217             }
218         }
219         // Zoek door de public mappen en kijk of de map/bestand bestaat.
220         $foundPublicFolder = false;
221         foreach ($publicFolder as $folder) {
222             if (is_dir($folder)) {
223                 $foundPublicFolder = $folder;
224                 if ($isFolder) {
225                     error_log('HTTP[403] Directory listing for "'.$url.'" not allowed');
226                     return new HttpError(403);
227                 }
228             }
229             $path = $isFolder ? $folder : $folder.basename($relativePath);
230             if (file_exists($path)) {
231                 if (is_readable($path)) {
232                     return new HttpError(500, array('warning' => 'render_public_folder.php should have renderd the file: "'.$path.'"'));
233                 } else {
234                     return new HttpError(403, array('warning' => 'Permission denied for file "'.basename($path).'" in "'.dirname($path).'/"'));
235                 }
236             }
237         }
238         if ($foundPublicFolder) {
239             return new HttpError(404, array('notice' => array(
240                 'HTTP[404] File "'.basename($relativePath).'" not found in "'.dirname($relativePath).'/"',
241                 'VirtualFolder "'.get_class(VirtualFolder::$current).'" doesn\'t handle the "'.basename(VirtualFolder::$current->getPath(true)).'" folder'
242             )));
243         }
244         // Gaat om een bestand in een virtualfolder
245         return new HttpError(404, array('notice' => 'HTTP[404] VirtualFolder "'.get_class(VirtualFolder::$current).'" has no "'.basename(VirtualFolder::$current->getPath(true)).'" folder'));
246     }

253     function &getParentByClass($class) {
254         if (strtolower(get_class($this)) == strtolower($class)) { // Is dit de gespecifeerde virtualfolder?
255             return $this;
256         } elseif ($this->parent === NULL) { // Is de virtualfolder niet gevonden in de hierarchie?
257             $message = ($class === null) ? 'VirtualFolder "'.get_class($this).'" has no parent' : 'VirtualFolder \''.$class.'\' is niet actief';
258             throw new \Exception($message);
259         }
260         return $this->parent->getParentByClass($class);
261     }

263     /**
264      * Mits de $this->depth niet is ingesteld zal de waarde van $this->depth berekent worden.
265      * Hoe diep de handler genest is wordt aan de hand van de Parent->depth berekend.
266      *
267      * @return int
268      */
269     private function initDepth() {
270         if ($this->depth !== NULL) { // Is de diepte reeds ingesteld?
271             return;
272         }
273         if (isset(VirtualFolder::$current) == false) { // Gaat het om de eerste VirtualFolder (Website)
274             if (($this instanceof Website) == false) {
275                 notice('VirtualFolder outside a Website object?');
276             }
277             VirtualFolder::$current = &$this; // De globale pointer laten verwijzen naar deze 'virtuele map'
278             if (defined('Sledgehammer\WEBPATH')) {
279                 $this->depth = preg_match_all('/[^\/]+\//', WEBPATH, $match);
280             } else {
281                 $this->depth = 0;
282             }
283             return;
284         }
285         $this->parent = &VirtualFolder::$current;
286         VirtualFolder::$current = &$this; // De globale pointer laten verwijzen naar deze 'virtuele map'
287         $this->depth = $this->parent->depth + $this->parent->depthIncrement;
288     }
