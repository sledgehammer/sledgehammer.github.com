 88     /**
 89      * Replace the placeholder and return the array.
 90      *
 91      * @param bool $ignoreDuplicateReplacement A foreach($hasManyPlaceholder as $item) will call the iterator methods in this class instead of the Collection
 92      * @return void
 93      */
 94     private function replacePlaceholder($ignoreDuplicateReplacement = false) {
 95         if (is_string($this->__placeholder) === false) { // Is the __reference already replaced?
 96             if ($ignoreDuplicateReplacement === false) {
 97                 notice('This placeholder is already replaced', 'Did you clone the object?');
 98             }
 99             return;
100         }
101         $parts = explode('/', $this->__placeholder);
102         $repositoryId = array_shift($parts);
103         $model = array_shift($parts);
104         $property = implode('/', $parts);
105         $data = PropertyPath::get($this->__container, $property);
106         if ($data !== $this) {
107             notice('This placeholder belongs to an other (cloned?) container');
108             $this->__placeholder = $data;
109             return;
110         }
111         $repo = getRepository($repositoryId);
112         $repo->loadAssociation($model, $this->__container, $property);
113         $this->__placeholder = PropertyPath::get($this->__container, $property);
114     }
