 50     /**
 51      * Return a new collection where each element is a subselection of the original element.
 52      * (Known as "collect" in Ruby or "pluck" in underscore.js)
 53      *
 54      * @param string|array $selector  Path to the variable to select. Examples: "->id", "[message]", "customer.name", array('id' => 'message_id', 'message' => 'message_text')
 55      * @param string|null|false $selectKey  (optional) The path that will be used as key. false: Keep the current key, null:  create linear keys.
 56      * @return Collection
 57      */
 58     function select($selector, $selectKey = false) {
 59         if ($this->data !== null || is_string($this->sql) || (is_object($selector) && is_callable($selector))) {
 60             return parent::select($selector, $selectKey);
 61         }
 62         if (is_int($selector)) {
 63             $selector = (string) $selector;
 64         }
 65         $selectorPaths = is_string($selector) ? array($selector => $selector) : $selector;
 66         $hasKeySelector = ($selectKey !== false && $selectKey !== null);
 67         if ($hasKeySelector) {
 68             array_key_unshift($selectorPaths, $selectKey, $selectKey);
 69         }
 70         if (count($selectorPaths) === 0) { // empty selector?
 71             return parent::select($selector, $selectKey);
 72         }
 73         $isWildcardSelector = ($this->sql->columns === '*' || $this->sql->columns == array('*' => '*'));
 74         if ($isWildcardSelector === false && (is_string($this->sql->columns) || count($selectorPaths) >= count($this->sql->columns))) {
 75             // The selector can't be a subsection of current columns.
 76             return parent::select($selector, $selectKey);
 77         }
 78 
 79         $columns = array();
 80         foreach ($selectorPaths as $to => $from) {
 81             $column = $this->convertPathToColumn($from);
 82             $alias = $this->convertPathToColumn($to);
 83             if ($column === false || $alias === false) { // Path can't be mapped to column
 84                 return parent::select($selector, $selectKey);
 85             }
 86             if ($isWildcardSelector === false) {
 87                 if (isset($this->sql->columns[$column])) {
 88                     $column = $this->sql->columns[$column]; // Use the original column/calculation (just change the alias)
 89                 } else {
 90                     // @todo Use array_search() for support of indexed array of columns.
 91                     return parent::select($selector, $selectKey);
 92                 }
 93             }
 94             if ($hasKeySelector) {
 95                 $alias = $column; // Don't alias in SQL, but (re)use the $selector.
 96             }
 97             $columns[$alias] = $column;
 98         }
 99         $collection = new DatabaseCollection($this->sql->select($columns), $this->dbLink);
100         if ($hasKeySelector || is_string($selector)) { // Does the $collection requires additional selecting?
101             return $collection->select($selector, $selectKey);
102         }
103         return $collection;
104     }

106     /**
107      * Return a subsection of the collection based on the conditions.
108      *
109      * Convert the $conditions to SQL object when appropriate.
110      *
111      * auto converts
112      *   ['x_id' => null]  to "x_id IS NULL"
113      *   ['x_id !=' => null]  to "x_id IS NOT NULL"
114      *  'hits' => 0]  to "hits = '0'"  (Because in mysql '' = 0 evaluates to true, '' = '0' to false)
115      *
116      * @param array $conditions
117      * @return Collection
118      */
119     function where($conditions) {
120         if ($this->data !== null || is_string($this->sql) || (is_object($conditions) && is_callable($conditions)) || $this->sql->limit !== false || $this->sql->offset != 0) {
121             return parent::where($conditions);
122         }
123         $db = getDatabase($this->dbLink);
124         $sql = $this->sql;
125         $logicalOperator = extract_logical_operator($conditions);
126         if ($logicalOperator === false) {
127             if (count($conditions) > 1) {
128                 notice('Conditions with multiple conditions require a logical operator.', "Example: array('AND', 'x' => 1, 'y' => 5)");
129             }
130             $logicalOperator = 'AND';
131         } else {
132             unset($conditions[0]);
133         }
134         if ($logicalOperator === 'AND') {
135             $method = 'andWhere';
136         } elseif ($logicalOperator === 'OR') {
137             $method = 'orWhere';
138         } else {
139             throw new \Exception('Unsupported logical operator "'.$logicalOperator.'", expecting "AND" or "OR"');
140         }
141 
142         // The result are rows(fetch_assoc arrays), all conditions must be columnnames (or invalid)
143         foreach ($conditions as $path => $value) {
144             if (preg_match('/^(.*) ('.COMPARE_OPERATORS.')$/', $path, $matches)) {
145                 $column = $this->convertPathToColumn($matches[1]);
146                 $operator = $matches[2];
147             } else {
148                 $column = $this->convertPathToColumn($path);
149                 $operator = '==';
150             }
151             if ($column === false) { // Converting to path failed?
152                 array_key_unshift($conditions, 0, $logicalOperator);
153                 return parent::where($conditions);
154             }
155             if ($value === null) {
156                 switch ($operator) {
157                     case '==':
158                         $operator = 'IS';
159                         $expectation = 'NULL';
160                         break;
161 
162                     case '!=':
163                         $operator = 'IS NOT ';
164                         $expectation = 'NULL';
165                         break;
166 
167                     case '>':
168                     case '<':
169                     case '>=':
170                     case '<=':
171                         $expectation = "''";
172                         break;
173 
174                     default:
175                         warning('Unknown behavior for NULL values with operator "'.$operator.'"');
176                         $expectation = $db->quote($expectation);
177                         break;
178                 }
179                 $sql = $sql->$method($column.' '.$operator.' '.$expectation);
180             } else {
181                 if ($operator === '!=') {
182                     $sql = $sql->$method('('.$column.' != '.$db->quote($value, \PDO::PARAM_STR).' OR '.$column.' IS NULL)');
183                 } elseif ($operator === 'IN') {
184                     if ((is_array($value) || $value instanceof \Traversable) === false) {
185                         notice('Operator IN expects an array or Traversable', $value);
186                         $value = explode(',', $value);
187                     }
188                     $quoted = array();
189                     foreach ($value as $val) {
190                         $quoted[] = $this->quote($db, $column, $val);
191                     }
192                     $sql = $sql->$method($column.' '.$operator.' ('.implode(', ', $quoted).')');
193                 } else {
194                     if ($operator === '==') {
195                         $operator = '=';
196                     }
197                     $sql = $sql->$method($column.' '.$operator.' '.$this->quote($db, $column, $value));
198                 }
199             }
200         }
201         return new DatabaseCollection($sql, $this->dbLink);
202     }

237      * Return a new Collection without the first x items.
238      *
239      * @param int $offset
240      * @return Collection
241      */
242     function skip($offset) {
243         if ($this->data === null && is_string($this->sql) === false) {
244             $sql = clone $this->sql;
245             $sql->offset += $offset; // Add to the current offset.
246             if ($sql->limit === false) {
247                 return new DatabaseCollection($sql, $this->dbLink);
248             } else {
249                 $sql->limit -= $offset;
250                 if ($sql->limit <= 0) { // Will all entries be skipped?
251                     return new Collection(array()); // return an empty collection
252                 }
253                 return new DatabaseCollection($sql, $this->dbLink);
254             }
255         }
256         return parent::skip($offset);
257     }

301     function count() {
302         if ($this->data === null && $this->sql instanceof Sql && is_array($this->sql->groupBy) && count($this->sql->groupBy) === 0) {
303             $sql = $this->sql->select('COUNT(*)');
304             return intval(getDatabase($this->dbLink)->fetchValue($sql));

310     /**
311      * Converts $this->data into an array.
312      *
313      * @return void
314      */
315     protected function dataToArray() {
316         if (is_array($this->data)) {
317             return;
318         }
319         if ($this->data === null) {
320             $db = getDatabase($this->dbLink);
321             $this->data = $db->fetchAll($this->sql);
322             return;
323         }
324         if ($this->data instanceof \PDOStatement) {
325             $this->data = $this->data->fetchAll();
326             return;
327         }
328         if ($this->data === false) {
329             throw new InfoException('Unable to "fetchAll()", the query failed', (string) $this->sql);
330         }
331         return parent::dataToArray();
332     }

334     /**
335      * Don't put quotes around number for columns that are assumend to be integers ('id' or ending in '_id')
336      *
337      * @param Database $db
338      * @param string $column
339      * @param mixed $value
340      */
341     private function quote($db, $column, $value) {
342         if (($column == 'id' || substr($column, -3) == '_id') && (is_int($value) || preg_match('/^[123456789]{1}[0-9]*$/', $value))) {
343             return $value;
344         }
345         return $db->quote($value);
346     }

348     /**
349      * Convert a path to a (escaped) columnname.
350      *
351      * @param string $path
352      */
353     private function convertPathToColumn($path) {
354         $compiled = PropertyPath::parse($path);
355         if (count($compiled) > 1) {
356             return false;
357         }
358         if (in_array($compiled[0][0], array(PropertyPath::TYPE_ANY, PropertyPath::TYPE_ELEMENT))) {
359             $db = getDatabase($this->dbLink);
360             return $db->quoteIdentifier($compiled[0][1]);
361         }
362         return false;
363     }
