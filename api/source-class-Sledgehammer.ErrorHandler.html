163      * Functie die wordt aangeroepen door de functies notice() / warning() / error() en de error handler functie
164      *
165      * @param int|\Exception $type
166      * @param string $message
167      * @param mixed $information
168      * @param bool $check_for_alternate_error_handler Controleert of de error door een andere error_handler moet worden afgehandeld (Bv: SimpleTestErrorHandler)
169      * @return void
170      */
171     function report($type, $message = '__EMPTY_ERROR_MESSAGE__', $information = null, $check_for_alternate_error_handler = false) {
172         if ($type instanceof \Exception) {
173             // @todo check the exception handler is this instance.
174         } elseif ($check_for_alternate_error_handler) {
175             $callback = set_error_handler(array($this, 'errorCallback'));
176             restore_error_handler();
177             if (is_array($callback) == false || $callback[0] !== $this) {
178                 $conversion_table = array(
179                     E_ERROR => E_USER_ERROR,
180                     E_WARNING => E_USER_WARNING,
181                     E_NOTICE => E_USER_NOTICE,
182                     E_DEPRECATED => E_USER_DEPRECATED,
183                 );
184                 if (array_key_exists($type, $conversion_table)) {
185                     $type = $conversion_table[$type]; // Voorkom "Invalid error type specified" bij trigger_error
186                 }
187                 trigger_error($message, $type); // De fout doorgeven aan de andere error handler
188                 return;
189             }
190         }
191         $this->process($type, $message, $information);
192     }

364     /**
365      * De fout afhandelen volgens de afhandel opties (email, log, html).
366      *
367      * @param int $type E_USER_NOTICE, E_USER_ERROR, enz
368      * @param string $message De foutmelding
369      * @param mixed $information Extra informatie voor de fout. bv array('Tip' => 'Controleer ...')
370      */
371     private function process($type, $message, $information = null) {
372         if ($this->isProcessing) {
373             echo '<span style="color:red">[ErrorHandler failure]';
374             if ($this->html && is_string($message)) { // show error?
375                 echo ' ', htmlentities($message);
376             }
377             echo ' </span>';
378             error_log('ErrorHandler failure: '.$message);
379             return;
380         }
381         $this->isProcessing = true;
382         if ($this->log || $this->cli) {
383             if ($type instanceof \Exception) {
384                 $error_message = get_class($type).': '.$type->getMessage();
385             } else {
386                 $error_message = $this->error_types[$type].': '.$message;
387             }
388             $location = $this->resolveLocation();
389             if ($location) {
390                 $error_message .= ' in '.$location['file'].' on line '.$location['line'];
391             }
392             if ($this->log) {
393                 error_log($error_message);
394             }
395             if ($this->cli) {
396                 echo '[', date('Y-m-d H:i:s'), '] ', $error_message, "\n";
397             }
398         }
399         // Limiet contoleren
400         if ($this->email) { // Een foutrapport via email versturen?
401             $limit = $this->import_email_limit();
402             if ($limit < 1) { // Is de limiet bereikt?
403                 $this->email = false; // Niet emailen
404             }
405         }
406         if (!$this->email && !$this->html) { // Geen email en geen html uitvoer
407             $this->isProcessing = false;
408             return; // Stoppen met het bouwen van een foutrapport
409         }
410         if ($this->email) { // De limiet is niet niet bereikt.
411             ob_start(); // bouw de html van de foutmelding eerst op in een buffer.
412         }
413         self::render($type, $message, $information);
414         if ($this->email) { // email versturen?
415             // Headers voor de email met HTML indeling
416             $headers = "MIME-Version: 1.0\n";
417             $headers .= "Content-type: text/html; charset=iso-8859-1\n";
418             $headers .= 'From: '.$this->fromEmail()."\n";
419 
420             if ($type instanceof \Exception) {
421                 $subject = get_class($type).': '.$type->getMessage();
422                 if ($message === '__UNCAUGHT_EXCEPTION__') {
423                     $subject = ' Uncaught '.$subject;
424                 }
425             } else {
426                 $subject = $this->error_types[$type].': '.$message;
427             }
428 
429             if (function_exists('mail') && !mail($this->email, $subject, '<html><body style="background-color: #fcf8e3">'.ob_get_contents()."</body></html>\n", $headers)) {
430                 error_log('The Sledgehammer\ErrorHandler was unable to email the report.');
431             }
432             if ($this->html) {
433                 ob_end_flush(); // buffer weergeven
434             } else {
435                 ob_end_clean(); // buffer niet weergeven
436             }
437         } elseif (ob_get_level() === 0) {
438             flush();
439         }
440         $this->isProcessing = false;
441     }

443     /**
444      * Returns the filename and linenumber where the error was triggered.
445      *
446      * @return array|false  array('file' => $filename, 'line' => $linenumber)
447      */
448     private function resolveLocation() {
449         $backtrace = debug_backtrace();
450         foreach ($backtrace as $call) {
451             if (isset($call['file']) && $call['file'] !== __FILE__ && $call['function'] !== 'handle') {
452                 return array(
453                     'file' => ((strpos($call['file'], PATH) === 0) ? substr($call['file'], strlen(PATH)) : $call['file']),
454                     'line' => $call['line']
455                 );
456             }
457         }
458         return false;
459     }

461     /**
462      * De bestanden, regelnummers, functie en objectnamen waar de fout optrad weergeven.
463      */
464     private function renderBacktrace() {
465         $backtrace = debug_backtrace();
466         echo "<b>Backtrace</b><div>\n";
467         // Pass 1:
468         //   Backtrace binnen de errorhandler niet tonen
469         //   Bij een Exception de $Exception->getTrace() tonen.
470         //   De plaats van de errror extra benadrukken (door een extra witregel)
471         while ($call = next($backtrace)) {
472             if (isset($call['object']) && $call['object'] === $this) {
473                 if ($call['function'] == 'errorCallback') { // Is de fout getriggerd door php
474                     if (isset($call['file'])) { // Zit de fout niet in een functie?
475                         $this->renderBacktraceCall($call, true); // Dan is de fout afkomsting van deze $call, maar verberg de ErrorHandler->errorCallback parameters
476                         next($backtrace);
477                         break;
478                     } else { // De fout komt uit een functie (bijvoorbeeld een Permission denied uit een fopen())
479                         $call = next($backtrace); // Sla deze $call over deze bevat alleen de ErrorHandler->errorCallback() aanroep
480                         $this->renderBacktraceCall($call);
481                         next($backtrace);
482                         break;
483                     }
484                 } elseif ($call['function'] == 'shutdown_callback') {
485                     $error = error_get_last();
486                     $this->renderBacktraceCall($error);
487                     next($backtrace);
488                     break;
489                 } elseif ($call['function'] == 'report' && $call['args'][0] instanceof \Exception) { // Gaat het om een Exception
490                     $exception = $call['args'][0];
491                     $reported = false;
492                     $viaExceptionCallback = next($backtrace);
493                     if (isset($viaExceptionCallback['function']) && $viaExceptionCallback['function'] === 'exceptionCallback') {
494                         $caught = next($backtrace);
495                         if ($caught !== false) { // backtrace after the callback?
496                             $reported = true; // someone called exceptionCallback() directly
497                         }
498                     } else {
499                         $reported = true;
500                         if (isset($viaExceptionCallback['function']) && $viaExceptionCallback['function'] === 'report_exception' && empty($viaExceptionCallback['class'])) {
501                             $call = $viaExceptionCallback; // skip the report_exception() call
502                         }
503                     }
504                     echo 'Exception thrown ';
505                     $trace = array(
506                         'file' => $exception->getFile(),
507                         'line' => $exception->getLine()
508                     );
509                     $this->renderBacktraceCall($trace, true);
510 
511                     $previous = $exception->getPrevious();
512                     if ($reported || $previous !== null) {
513                         echo '<span style="color:#aaa;margin-left:20px"'; //d5ac68
514                         if ($previous !== null) {
515                             echo "title=", htmlentities($previous->getTraceAsString(), ENT_COMPAT, 'UTF-8').'">Previous ';
516                             echo get_class($previous), '("'.$previous->getMessage(), '") was thrown';
517                             $trace = array(
518                                 'file' => $previous->getFile(),
519                                 'line' => $previous->getLine()
520                             );
521                             $this->renderBacktraceCall($trace, true);
522                         } else {
523                             echo '>';
524                         }
525                         if ($reported) {
526                             echo 'Reported ';
527                             $this->renderBacktraceCall($call, true);
528                         }
529                         echo '</span>';
530                     }
531                     $backtrace = $exception->getTrace(); // Show the trace of the exception
532                     break;
533                 }
534             } else {
535                 $this->renderBacktraceCall($call);
536                 next($backtrace);
537                 break;
538             }
539         }
540         echo '<br />';
541         // Pass 2: Alle elementen weergeven uit de $backtrace die nog niet behandeld zijn
542         while ($call = current($backtrace)) {
543             $this->renderBacktraceCall($call);
544             next($backtrace);
545         }
546         echo '</div>';
547     }

549     /**
550      * Render a call from the backtrace with syntax highlighting.
551      *
552      * @param array $call
553      * @param bool $location_only
554      */
555     private function renderBacktraceCall($call, $location_only = false) {
556         if (!$location_only) {
557             if (isset($call['object'])) {
558                 echo syntax_highlight($call['object'], null, 512);
559                 echo syntax_highlight($call['type'], 'operator');
560             } elseif (isset($call['class'])) {
561                 echo syntax_highlight($call['class'], 'class');
562                 echo syntax_highlight($call['type'], 'operator');
563             }

696     /**
697      * Opvragen hoeveel emails er nog verstuurd mogen worden.
698      *
699      * @return false|int Retourneert the limiet
700      */
701     private function import_email_limit() {
702         // Waardes uit de configuratie corrigeren
703         $this->emails_per_day = $this->mergeLimit($this->emails_per_day);
704         $this->emails_per_minute = $this->mergeLimit($this->emails_per_minute);
705         $this->emails_per_request = $this->mergeLimit($this->emails_per_request);
706         //
707         if ($this->emails_per_request !== 'NO_LIMIT') {
708             $limit = $this->emails_per_request;
709             $this->emails_per_request--;
710             if ($limit < 1) {
711                 $this->emails_per_request = 0;
712                 return 0;
713             }
714         }
715         $filename = TMP_DIR.'error_handler_email_limit.txt';
716         if (!@file_exists($filename)) {
717             error_log('File "'.$filename.'" doesn\'t exist (yet)');
718             // nieuwe voorraad.
719             $day_limit = $this->emails_per_day;
720             $minute_limit = $this->emails_per_minute;
721         } else { // het bestand bestaat.
722             // Bestand inlezen.
723             $fp = @fopen($filename, 'r');
724             if ($fp === false) {
725                 error_log('Reading file "'.$filename.'" failed');
726                 return false; // Kon het bestand niet openen.
727             }
728             $datum = @rtrim(fgets($fp)); // datum inlezen.
729             $day_limit = $this->mergeLimit(@rtrim(fgets($fp)), $this->emails_per_day);
730             $tijdstip = @rtrim(fgets($fp)); // minuut inlezen
731             $minute_limit = $this->mergeLimit(@rtrim(fgets($fp)), $this->emails_per_minute);
732             @fclose($fp);
733 
734             if ($datum != @date('j-n')) { // is het een andere datum?
735                 $day_limit = $this->emails_per_day; // nieuwe dag voorraad.
736                 $minute_limit = $this->emails_per_minute;
737                 error_log('Resetting errorlimit');
738             } elseif ($tijdstip != @date('H:i')) { // Is het een andere minuut
739                 $minute_limit = $this->emails_per_minute; // nieuwe minuut voorraad.
740             }
741         }
742         $limit = 0; // Standaard instellen dat de limiet is bereikt.
743         if ($day_limit === 'NO_LIMIT' && $minute_limit === 'NO_LIMIT') { // Is er helemaal geen limiet?
744             $limit = 999;
745         } elseif ($day_limit === 'NO_LIMIT') { // Er is geen limiet per dag?
746             $limit = $minute_limit; // Gebruik de limit van de minuut
747             $minute_limit--; // voorraad verminderen
748         } elseif ($minute_limit === 'NO_LIMIT') { // Geen limit per minuut?
749             $limit = $day_limit; // Gebruik de limit van de minuut
750             $day_limit--; // voorraad verminderen
751         } else { // Er is zowel een limiet per dag als per minuut
752             $limit = ($day_limit < $minute_limit) ? $day_limit : $minute_limit; // limit wordt de laagste waarde
753             // Voorraad verminderen
754             $day_limit--;
755             $minute_limit--;
756         }
757         // Is de limiet van 0 naar -1 veranderd, zet deze weer op 0
758         $day_limit = $this->mergeLimit($day_limit);
759         $minute_limit = $this->mergeLimit($minute_limit);
760 
761         // Wijzigingen opslaan
762         $fp = @fopen($filename, 'w');
763         if ($fp !== false) {
764             $write = @fwrite($fp, @date('j-n')."\n".$day_limit."\n".@date('H:i')."\n".$minute_limit);
765             if ($write == false) {
766                 error_log('Writing to file "'.$filename.'"  failed');
767                 return false; // Kon niet schrijven.
768             }
769             @fclose($fp);
770         } else {
771             error_log('Unable to open file "'.$filename.'" in write mode');
772             return false; // Kon het bestand niet openen om te schrijven
773         }
774         return $limit;
775     }

777     /**
778      * Zet een variable om naar een limiet (0 of groter of  "NO_LIMIT")
779      * En corrigeert de waarde als een van de waardes NO_LIMIT
780      *
781      * @param mixed $limit wordt omgezet naar een int of "NO_LIMIT"
782      * @param int|string $configuredLimit ingestelde limiet
783      */
784     private function mergeLimit($limit, $configuredLimit = null) {
785         $limit = ($limit === 'NO_LIMIT') ? 'NO_LIMIT' : (int) $limit;
786         if ($configuredLimit !== null) {
787             if ($limit === 'NO_LIMIT' && $configuredLimit !== 'NO_LIMIT') { // Als er in het error_handler_email_limit.txt "NO_LIMIT" staat, maar in de config 100, return dan 100
788                 return $configuredLimit;
789             }
790             if ($configuredLimit === 'NO_LIMIT') { // Als er NO_LIMIT in de config staat return dan NO_LIMIT ongeacht waarde uit error_handler_email_limit.txt
791                 return 'NO_LIMIT';
792             }
793             if ($limit > $configuredLimit) { // Als er in het error_handler_email_limit een groter getal staan dan in de config?
794                 return $configuredLimit; // return de config
795             }
796         }
797         // Rond negative getallen af naar 0
798         if (is_int($limit) && $limit < 0) {
799             return 0;
800         }
801         return $limit;
802     }

807      * return string
808      */
809     private function fromEmail() {
810         $hostname = isset($_SERVER['HTTP_HOST']) ? $_SERVER['HTTP_HOST'] : php_uname('n');
811         $regexDomain = '/[a-z0-9-]+(.[a-z]{2}){0,1}\.[a-z]{2,4}$/i';
812         if (preg_match($regexDomain, $hostname, $match)) { // Zit er een domeinnaam in de hostname?
813             $domain = $match[0];
814         } elseif (preg_match($regexDomain, $this->email, $match)) { // de hostname was een ip, localhost, ofzo. Pak dan de domeinnaam van het "To:" adres
815             $domain = $match[0];
816         } else { // Er zit zelfs geen domeinnaam in het email-adres?
817             $domain = 'localhost';
818         }
819         return '"ErrorHandler ('.$hostname.')" <errorhandler@'.$domain.'>';
820     }
