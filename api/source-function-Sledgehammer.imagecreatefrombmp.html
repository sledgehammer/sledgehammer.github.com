 20 /**
 21  * Create a GD resource from a BMP file.
 22  *
 23  * @link http://php.net/gd
 24  * @param string $filename
 25  * @return resource gd
 26  */
 27 function imagecreatefrombmp($filename) {
 28     // Load the image into a string
 29     $fp = fopen($filename, "rb");
 30     $read = fread($fp, 10);
 31     while (!feof($fp) && ($read <> "")) {
 32         $read .= fread($fp, 1024);
 33     }
 34 
 35     $temp = unpack("H*", $read);
 36     $hex = $temp[1];
 37     $header = substr($hex, 0, 108);
 38 
 39     // Process the header
 40     // Structure: http://www.fastgraph.com/help/bmp_header_format.html
 41     if (substr($header, 0, 4) == "424d") {
 42 
 43         $header_parts = str_split($header, 2); // Cut it in parts of 2 bytes
 44         $width = hexdec($header_parts[19].$header_parts[18]); // Get the width (4 bytes)
 45         $height = hexdec($header_parts[23].$header_parts[22]); // Get the height (4 bytes)
 46         unset($header_parts);
 47     }
 48     $x = 0;
 49     $y = 1;
 50 
 51     //    Create newimage
 52     $image = imagecreatetruecolor($width, $height);
 53 
 54     // Grab the body from the image
 55     $body = substr($hex, 108);
 56 
 57     // Calculate if padding at the end-line is needed
 58     // Divided by two to keep overview.
 59     // 1 byte = 2 HEX-chars
 60     $body_size = (strlen($body) / 2);
 61     $header_size = ($width * $height);
 62 
 63     // Use end-line padding? Only when needed
 64     $usePadding = ($body_size > ($header_size * 3) + 4);
 65 
 66     // Using a for-loop with index-calculation instaid of str_split to avoid large memory consumption
 67     // Calculate the next DWORD-position in the body
 68     for ($i = 0; $i < $body_size; $i += 3) {
 69         //    Calculate line-ending and padding
 70         if ($x >= $width) {
 71             // If padding needed, ignore image-padding
 72             // Shift i to the ending of the current 32-bit-block
 73             if ($usePadding) {
 74                 $i += $width % 4;
 75             }
 76             $x = 0; // Reset horizontal position
 77             $y++; // Raise the height-position (bottom-up)
 78 
 79 
 80             if ($y > $height) { // Reached the image-height?
 81                 break;
 82             }
 83         }
 84 
 85         // Calculation of the RGB-pixel (defined as BGR in image-data)
 86         // Define $i_pos as absolute position in the body
 87         $i_pos = $i * 2;
 88         $r = hexdec($body[$i_pos + 4].$body[$i_pos + 5]);
 89         $g = hexdec($body[$i_pos + 2].$body[$i_pos + 3]);
 90         $b = hexdec($body[$i_pos].$body[$i_pos + 1]);
 91 
 92         // Calculate and draw the pixel
 93         $color = imagecolorallocate($image, $r, $g, $b);
 94         imagesetpixel($image, $x, ($height - $y), $color);
 95 
 96         $x++; // Raise the horizontal position
 97     }
 98     unset($body);
 99     return $image;
100 }
