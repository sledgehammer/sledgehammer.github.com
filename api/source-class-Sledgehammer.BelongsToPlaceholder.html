58     private function __replacePlaceholder() {
59         if (is_string($this->__placeholder) === false) { // Is the placeholder already replaced
60             notice('This placeholder is already replaced', 'Did you clone the object?');

62         }
63         $parts = explode('/', $this->__placeholder);
64         $repositoryId = array_shift($parts);
65         $model = array_shift($parts);
66         $property = implode('/', $parts);

70             $this->__placeholder = $self;
71             return;
72         }

74         $repo->loadAssociation($model, $this->__container, $property);
75         $this->__placeholder = PropertyPath::get($this->__container, $property);
