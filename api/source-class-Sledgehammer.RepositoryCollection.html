 63     function select($selector, $selectKey = false) {
 64         if ($this->isConverted || ($this->data instanceof Collection) === false) {
 65             return parent::select($selector, $selectKey);
 66         }
 67         if ($selectKey !== false && $selectKey !== null) {
 68             if (empty($this->mapping[$selectKey])) { // Key not in the mapping array?
 69                 return parent::select($selector, $selectKey);
 70             }
 71             $selectKeyMapped = $this->mapping[$selectKey];
 72         } else {
 73             $selectKeyMapped = $selectKey;
 74         }
 75         if (is_string($selector) && isset($this->mapping[$selector])) {
 76             // Bypass Repository and return the resultset directly from the backend-collection.
 77             return $this->data->select($this->mapping[$selector], $selectKeyMapped);
 78         } elseif (is_array($selector)) {
 79             $selectorMapped = array();
 80             foreach ($selector as $to => $from) {
 81                 if (empty($this->mapping[$from])) { // Field not in the mapping array?
 82                     return parent::select($selector, $selectKey);
 83                 }
 84                 $selectorMapped[$to] = $this->mapping[$from];

 87             return $this->data->select($selectorMapped, $selectKeyMapped);
 88         }

 90         return parent::select($selector, $selectKey);

 93     function where($conditions) {
 94         if ($this->isConverted) {
 95             return parent::where($conditions);
 96         }
 97         if ($this->data instanceof Collection && is_array($conditions)) {
 98             $convertedConditions = array();
 99             foreach ($conditions as $path => $value) {
100                 if (isset($this->mapping[$path])) {
101                     $convertedConditions[$this->mapping[$path]] = $value;
102                     unset($conditions[$path]);
103                 }
104             }
105             if (count($convertedConditions) !== 0) { // There are conditions the low-level collection can handle?
106                 $collection = new RepositoryCollection($this->data->where($convertedConditions), $this->model, $this->repository, $this->mapping);
107                 if (count($conditions) == 0) {
108                     return $collection;
109                 }
110                 return $collection->where($conditions); // Apply the remaining conditions
111             } elseif (count($conditions) === 0) { // An empty array was given as $conditions?
112                 return new RepositoryCollection(clone $this->data, $this->model, $this->repository, $this->mapping);
113             }
114         }
115         return parent::where($conditions);
116     }
