 67     /**
 68      * Constructor
 69      * @param null|string $url  De url om te parsen, bij NULL wordt de huidige url gebruikt
 70      */
 71     function __construct($url) {
 72         $info = parse_url($url);
 73         if ($info === false) {
 74             throw new \Exception('Invalid url: "'.$url.'"');
 75         }
 76         if (isset($info['query'])) {
 77             parse_str($info['query'], $info['query']); // Zet de query om naar een array
 78         }
 79         if (isset($info['path'])) {
 80             $info['path'] = rawurldecode($info['path']); // "%20" omzetten naar " " e.d.
 81         }
 82         set_object_vars($this, $info);
 83     }

 85     /**
 86      * Generate the url as a string.
 87      * Allows URL object to be used as php strings.
 88      *
 89      * @return string
 90      */
 91     function __toString() {
 92         $url = '';
 93         if ($this->scheme !== null && $this->host !== null) {
 94             $url .= $this->scheme.'://';
 95             if (empty($this->user) == false) {
 96                 $url .= rawurlencode($this->user);
 97                 if (empty($this->pass) == false) {
 98                     $url .= ':'.$this->pass;
 99                 }
100                 $url .= '@';
101             }
102             $url .= $this->host;
103             if ($this->port) {
104                 $standardPorts = array(
105                     'http' => 80,
106                     'https' => 443,
107                 );
108                 if ($this->scheme === null || empty($standardPorts[$this->scheme]) || $standardPorts[$this->scheme] != $this->port) { // Is the port non-standard?
109                     $url .= ':'.$this->port;
110                 }
111             }
112         }
113         if ($this->path !== null) {
114             $url .= str_replace('%2F', '/', rawurlencode($this->path));
115         }
116         if (is_string($this->query) && trim($this->query) !== '') {
117             $url .= '?'.$this->query;
118         } elseif (is_array($this->query) && count($this->query) != 0) {
119             $url .= '?'.http_build_query($this->query);
120         }
121         if ($this->fragment !== null) {
122             $url .= '#'.$this->fragment;
123         }
124         return $url;
125     }

159     static function getCurrentURL() {

162                 $scheme = 'https';
163                 $port = ($_SERVER['SERVER_PORT'] == '443') ? '' : ':'.$_SERVER['SERVER_PORT'];
164             } else {
165                 $scheme = 'http';
166                 $port = ($_SERVER['SERVER_PORT'] == '80') ? '' : ':'.$_SERVER['SERVER_PORT'];
167             }
168             self::$current = new URL($scheme.'://'.$_SERVER['SERVER_NAME'].$port.$_SERVER['REQUEST_URI']);
169         }
170         return clone self::$current;

185     /**
186      * Multi-functionele functie om parameters op te vragen en toe te voegen
187      *
188      * URL:parameters(); vraagt de huidige parameters op. ($_GET)
189      * URL:parameters("naam['test']=1234"); of URL::parameters(array('naam'=>array('test'=>1234))); voegt deze parameter toe aan de huidige parameter array.
190      * URL:parameter("bla=true", 'x=y'); voegt 2 parameter 'arrays' samen
191      *
192      * @param array $append  De parameter die toegevoegd moet worden
193      * @param mixed $stack   De url of array waarde parameters waaraan toegevoegd moet worden, bij NULL worden de huidige $_GET parameters gebruikt
194      * @return array
195      */
196     static function parameters($append = array(), $stack = NULL) {
197         deprecated('Maar nog geen alternatief beschikbaar');
198         if ($stack === NULL) { // Huidige parameters opvragen
199             $stack = $_GET;
200         } elseif (is_string($stack)) { // Is er geen array, maar een query string meegegeven?
201             parse_str($stack, $stack);
202         }
203         if (is_string($append)) {
204             parse_str($append, $append);
205         }
206         return array_merge(array_diff_key($stack, $append), $append); // De array kan gebruikt worden in een http_build_query()
207     }

209     /**
210      * Een sub-domein opvragen van een domein
211      *
212      * @param int $index Bepaald welke subdomein van de subdomeinen er wordt opgevraagd. 0 = eerste subdomein van links, -1 =  eerste subdomein van rechts
213      * @param NULL|string $uri de uri waarvan het subdomein opgevraagd moet worden
214      * @return string
215      */
216     static function subdomain($index = -1, $uri = NULL) {
217         deprecated('Maar nog geen alternatief beschikbaar');
218 
219         if ($uri === NULL) {
220             $uri = URL::info('host');
221         }
222         $parts = explode('.', $uri);
223         $count = count($parts);
224         if ($index < 0) { // is $index negatief?
225             $index = $count - 2 + $index; // van links naar rechts
226         } elseif ($index + 2 >= $count) { // is $index groter dan aantal subdomeinen?
227             return '';
228         }
229         $subdomain = @$parts[$index];
230         return ($subdomain === NULL) ? '' : $subdomain;
231     }
