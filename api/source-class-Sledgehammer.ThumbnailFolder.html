14     protected $imagesFolder;

17     function __construct($imagesFolder, $targetFolder = null) {
18         parent::__construct();
19         $this->imagesFolder = $imagesFolder;
20         if ($targetFolder == null) {
21             $targetFolder = TMP_DIR.'ThumbnailFolder/'.basename($imagesFolder).'_'.substr(md5($imagesFolder), 8, 16).'/';
22         }
23         $this->targetFolder = $targetFolder;
24     }

26     function dynamicFilename($filename) {
27         if ($this->isImage($filename)) {
28             return new FileDocument($this->imagesFolder.$filename);
29         }
30         return $this->onFileNotFound();
31     }

33     function dynamicFoldername($folder, $filename = null) {
34         if (!preg_match('/^[0-9]+x[0-9]+$/', $folder)) { // Zijn er geen afmetingen meegegeven?
35             return $this->onFolderNotFound();
36         }
37         if (!$filename) { // Komt de afbeelding uit een subfolder($recursive)?
38             $url = Url::getCurrentURL();
39             $path = array(
40                 'folders' => $url->getFolders(),
41                 'filename' =>  $url->getFilename(),
42             );
43             $subfolders = array_slice($path['folders'], $this->depth + 1);
44             $filename = implode('/', $subfolders).'/'.$path['filename'];
45         }
46         $source = $this->imagesFolder.$filename;
47         if (!file_exists($source)) {
48             return new HttpError(404, array('warning' => 'Image "'.$filename.'" not found in "'.$this->imagesFolder.'"'));
49         }
50         $target = $this->targetFolder.$folder.'/'.$filename;
51         if (!file_exists($target) || filemtime($source) > filemtime($target)) {
52             $dimensions = explode('x', $folder);
53             mkdirs(dirname($target));
54             $image = new Image($source);
55             $image->saveThumbnail($target, $dimensions[0], $dimensions[1]);
56         }
57         return new FileDocument($target);
58     }
59 
60     protected function isImage($filename) {
61         return substr(mimetype($filename, true), 0, 6) == 'image/';
62     }
