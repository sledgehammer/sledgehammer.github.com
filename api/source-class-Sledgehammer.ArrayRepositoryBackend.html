 28         );

 31         if ($row !== null) { // The $items array is NOT empty
 32             if (is_object($row)) {
 33                 $row = get_object_vars($row); // Convert to array
 34             }
 35             if (count($config->properties) == 0) { // No "column => property" mapping defined?
 36                 // Generate a direct 1 on 1 mapping based on the first item.
 37                 $columns = array_keys($row);
 38                 $config->properties = array_combine($columns, $columns);
 39             }
 40             if (count($config->id) === 0 && array_key_exists('id', $row) === false) { // 'id' field detected in the first row?
 41                 $config->id = array('id');
 42                 $config->backendConfig['indexed'] = true;
 43                 foreach (array_keys($items) as $index) {
 44                     $items[$index]['id'] = $index;
 45                 }
 46                 if (array_key_exists('id', $config->properties) == false) {
 47                     array_key_unshift($config->properties, 'id', 'id');
 48                 }
 49                 $config->backendConfig['key'] = 'id';
 50             } elseif (count($config->id) == 1) { // Only 1 field as id?
 51                 // Copy the data using the id as index
