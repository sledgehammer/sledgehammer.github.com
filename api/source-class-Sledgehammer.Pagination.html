 39     /**
 40      *
 41      * @param int $count     Number of pages
 42      * @param int $current    Current page
 43      * @param array $options [optional]
 44      */
 45     function __construct($count, $current = 1, $options = array()) {
 46         $this->count = $count;
 47         $this->current = $current;
 48         foreach ($options as $key => $value) {
 49             $this->$key = $value;
 50         }
 51         if ($this->href === null) {
 52             $url = Url::getCurrentURL();

 54             $url->query[$this->parameter] = '';
 55             $this->href = (string) $url;
 56         }

 59     function render() {
 60         if ($this->count == 0) { // No pages, no pagination
 61             return;
 62         }
 63         $start = 1;
 64         $end = $this->count;
 65         $pages = array();
 66         $class = 'pagination';
 67         if ($this->align != 'left') {
 68             $class .= ' pagination-'.$this->align;
 69         }
 70         echo "<div class=\"".$class."\"><ul>\n";
 71         // previous
 72         if ($this->current != 1) {
 73             echo "\t<li>", Html::element('a', array('href' => $this->href.($this->current - 1)), '&laquo;'), "</li>\n";
 74         }
 75         if ($this->count > $this->max) {
 76             $offset = floor($this->max / 2);
 77             if ($this->current < $offset + 2) {
 78                 $end = $this->max - 1;
 79             } else {
 80                 if ($this->current + $offset > $this->count) {
 81                     $start = $this->count - $this->max;
 82                 } else {
 83                     $start = $this->current - $offset;
 84                 }
 85                 $end = $this->current + $offset;
 86                 if ($end > $this->count) {
 87                     $end = $this->count;
 88                 }
 89             }
 90         }
 91         // numbers
 92         for ($i = $start; $i <= $end; $i++) {
 93             $attributes = ($i == $this->current) ? array('class' => 'active') : array();
 94             echo "\t", Html::element('li', $attributes, Html::element('a', array('href' => $this->href.$i), $i)), "\n";
 95         }
 96         // total pages indication
 97         if ($end != $this->count) {
 98             if ($end != ($this->count - 1)) {
 99                 $nextDecimal = (floor($this->current / 10) * 10) + 10;
100                 if ($nextDecimal > $this->count) {
101                     $nextDecimal = $this->count;
102                 }
103                 echo "\t<li>", Html::element('a', array('href' => $this->href.$nextDecimal), '...'), "</li>\n"; // @todo jump to page
104             }
105             echo "\t<li>", Html::element('a', array('href' => $this->href.($this->count)), $this->count), "</li>\n";
106         }
107         // next
108         if ($this->current != $this->count) {
109             echo "\t<li>", Html::element('a', array('href' => $this->href.($this->current + 1)), '&raquo;'), "</li>\n";
110         }
