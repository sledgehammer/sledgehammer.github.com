 49     /**
 50      * Create model configs based on the tables in the database schema
 51      *
 52      * @param string $dbLink
 53      */
 54     function inspectDatabase($dbLink = 'default', $tablePrefix = '') {
 55         // Pass 1: Retrieve and parse schema information
 56         $db = getDatabase($dbLink);
 57         if (self::$cacheTimeout === false) {
 58             $schema = $this->getSchema($db, $tablePrefix);
 59         } else {
 60             $cacheIdentifier = $dbLink.' '.$tablePrefix.' ';
 61             if (count($db->logger->entries) > 0) {
 62                 $cacheIdentifier .= $db->logger->entries[0][0]; // Add the connect statement to the identifier.
 63             }
 64             $backend = $this;
 65             $schema = cache('DatabaseRepositoryBackend['.md5($cacheIdentifier).']', self::$cacheTimeout, function () use ($backend, $db, $tablePrefix) {
 66                 return $backend->getSchema($db, $tablePrefix);
 67             });
 68         }
 69         $models = array();
 70         $junctions = array();
 71         foreach ($schema as $tableName => $table) {
 72             $name = Inflector::modelize($tableName, array(
 73                 'stripPrefix' => $tablePrefix,
 74                 'singularizeLast' => true
 75             ));
 76             $plural = Inflector::modelize($tableName, array(
 77                 'stripPrefix' => $tablePrefix,
 78                 'singularizeLast' => false
 79             ));
 80             $config = new ModelConfig($name, array(
 81                     'plural' => $plural,
 82                     'backendConfig' => $table,
 83                 ));
 84             $config->backendConfig['dbLink'] = $dbLink;
 85             $config->backendConfig['collection'] = array('columns' => array()); // database collection config.
 86             $config->id = $table['primaryKeys'];
 87             foreach ($table['columns'] as $column => $info) {
 88                 $default = @$info['default'];
 89                 $config->defaults[$column] = $default;
 90                 if (empty($info['foreignKeys'])) {
 91                     $property = Inflector::variablize($column);
 92                     $config->properties[$column] = $property;
 93                     if (substr($info['type'], -10) === 'tinyint(1)' && $info['null'] === false) {
 94                         $config->readFilters[$column] = __CLASS__.'::valueToBool';
 95                         $config->writeFilters[$column] = __CLASS__.'::boolToValue';
 96                         $config->defaults[$column] = self::valueToBool($config->defaults[$column]);
 97                     }
 98                 } else {
 99                     if (count($info['foreignKeys']) > 1) {
100                         notice('Multiple foreign-keys per column not supported');
101                     }
102                     $foreignKey = $info['foreignKeys'][0];
103                     $property = $this->stripIdSuffix($column);
104                     if ($column !== $property && array_key_exists($property, $table['columns'])) {
105                         $property = $column; // restore prefix, to prevent a naming collision.
106                     }
107                     if (array_key_exists($property, $table['columns']) && $property != $column) {
108                         notice('Unable to use belongsTo["'.$property.'"], an column with the same name exists', array('belongsTo' => $info, 'Exising column' => $table['columns'][$property]));
109                     }
110                     $foreignModel = Inflector::modelize($foreignKey['table'], array(
111                         'stripPrefix' => $tablePrefix,
112                         'singularizeLast' => true
113                     ));
114                     $config->belongsTo[$property] = array(
115                         'reference' => $column, // foreignKey
116                         'model' => $foreignModel,
117                         'id' => $foreignKey['column'], // primary key
118                         'default' => $config->defaults[$column]
119                     );
120                     unset($config->defaults[$column]);
121                 }
122             }
123 
124             // Detect junction table
125             if (count($config->id) === 2) {
126                 $primaryKeyAreForeignKeys = true;
127                 foreach ($config->belongsTo as $belongsTo) {
128                     if (in_array($belongsTo['reference'], $config->id) === false) {
129                         $primaryKeyAreForeignKeys = false;
130                         break;
131                     }
132                 }
133                 if ($primaryKeyAreForeignKeys) {
134                     $junctions[$tableName] = $config;
135                     $this->junctions[$config->name] = $config;
136                     continue;
137                 }
138             }
139             $models[$tableName] = $config;
140             if (isset($this->configs[$config->name])) {
141                 notice('Model "'.$config->name.'" (for the "'.$tableName.'" table) is defined in both "'.$this->configs[$config->name]->backendConfig['dbLink'].'" and "'.$dbLink.'" ');
142                 $suffix = 2;
143                 while(isset($this->configs[$config->name.'_'.$suffix])) {
144                     $suffix++;
145                 }
146                 $this->configs[$config->name.'_'.$suffix] = $config;
147             } else {
148                 $this->configs[$config->name] = $config;
149             }
150         }
151         // Pass 2: hasMany
152         foreach ($this->configs as $config) {
153             $table = $schema[$config->backendConfig['table']];
154             foreach ($table['referencedBy'] as $i => $reference) {
155                 $property = $reference['table'];
156                 if ($tablePrefix != '' && substr($property, 0, strlen($tablePrefix)) == $tablePrefix) {
157                     $property = substr($property, strlen($tablePrefix)); // Strip prefix
158                 }
159                 foreach ($table['referencedBy'] as $j => $otherRef) {
160                     if ($i != $j && $otherRef['table']  === $reference['table']) { // The model has relations to the same table.
161                         $property = $this->stripIdSuffix($reference['column']).'_'.$property;
162                     }
163                 }
164                 $property = Inflector::variablize($property);
165                 if (in_array($property, $config->getPropertyNames())) {
166                     notice('Unable to use '.$config->name.'->hasMany['.$property.'] a property with the same name exists');
167                     break;
168                 }
169                 if (isset($models[$reference['table']])) {
170                     // One-to-many relation
171                     $belongsToModel = $models[$reference['table']];
172                     foreach ($belongsToModel->belongsTo as $belongsToProperty => $belongsTo) {
173                         if ($belongsTo['model'] == $config->name && $belongsTo['reference'] == $reference['column']) {
174                             $config->hasMany[$property] = array(
175                                 'model' => $belongsToModel->name,
176                                 'reference' => $reference['column'],
177                                 'belongsTo' => $belongsToProperty
178                             );
179                             $config->defaults[$property] = array();
180                             break;
181                         }
182                     }
183                 } elseif (empty($junctions[$reference['table']])) {
184                     notice('Missing a model or relation for "'.$reference['table'].'" in the database schema');
185                 }
186             }
187         }
188         // Pass 3: hasMany junctions
189         foreach ($this->configs as $config) {
190             $table = $schema[$config->backendConfig['table']];
191             foreach ($table['referencedBy'] as $reference) {
192                 if (isset($junctions[$reference['table']])) {
193                     // Many-to-many realtion
194                     $junction = $junctions[$reference['table']];
195                     $hasMany = array(
196                         'through' => $junction->name,
197                         'fields' => $junction->properties,
198                     );
199                     foreach ($junction->belongsTo as $belongsToProperty => $belongsTo) {
200                         if ($belongsTo['model'] === $config->name) {
201                             $hasMany['reference'] = $belongsTo['reference'];
202                         } else {
203                             $property = Inflector::pluralize($belongsToProperty);
204                             $hasMany['model'] = $belongsTo['model'];
205                             $hasMany['id'] = $belongsTo['reference'];
206                         }
207                     }
208                     if (in_array($property, $config->getPropertyNames())) {
209                         $property = Inflector::variablize($reference['table']);
210                     }
211                     if (in_array($property, $config->getPropertyNames())) {
212                         notice('Unable to use '.$property.'->hasMany['.$property.'] a property with the same name exists');
213                         break;
214                     }
215                     $config->hasMany[$property] = $hasMany;
216                     $config->defaults[$property] = array();
217                 }
218             }
219         }
220     }

222     /**
223      * Load the record from the db
224      *
225      * @param mixed $id
226      * @param array $config
227      * @return array
228      */
229     function get($id, $config) {
230         $db = getDatabase($config['dbLink']);
231         if (is_array($id)) {
232             if (count($config['primaryKeys']) != count($id)) {
233                 throw new \Exception('Incomplete id, table: "'.$config['table'].'" requires: "'.human_implode('", "', $config['primaryKeys']).'"');
234             }
235         } elseif (count($config['primaryKeys']) == 1) {
236             $id = array($config['primaryKeys'][0] => $id); // convert $id to array notation
237         } else {
238             throw new \Exception('Incomplete id, table: "'.$config['table'].'" requires: "'.human_implode('", "', $config['primaryKeys']).'"');
239         }
240         $params = array();
241         foreach ($config['primaryKeys'] as $column) {
242             if (isset($id[$column]) == false) {
243                 throw new \Exception('Missing key: "'.$column.'"'); // @todo better error
244             }
245             $params[] = $id[$column];
246         }
247         if (isset($this->preparedStatements[$config['dbLink']][$config['table']])) {
248             $statement = $this->preparedStatements[$config['dbLink']][$config['table']];
249         } else {
250             $statement = $db->prepare('SELECT * FROM '.$db->quoteIdentifier($config['table']).' WHERE '.$this->generateWhere($id, $config, true));
251             $this->preparedStatements[$config['dbLink']][$config['table']] = $statement;
252         }
253         $success = $statement->execute($params);
254         if ($success === false) {
255             throw new \Exception('Failed to retrieve "'.$this->generateWhere($id, $config).'" from "'.$config['table'].'"');
256         }
257         $rows = $statement->fetchAll();
258         if (count($rows) !== 1) {
259             if (count($rows) === 0) {
260                 throw new \Exception('Record "'.$this->generateWhere($id, $config).'" doesn\'t exist in "'.$config['table'].'"');
261             }
262             throw new \Exception('Multiple record found for "'.$this->generateWhere($id, $config).'" in "'.$config['table'].'"');
263         }
264         return $rows[0];
265     }

277     /**
278      *
279      * @param array $new
280      * @param array $old
281      * @param array $config
282      * @return array
283      */
284     function update($new, $old, $config) {
285         $db = getDatabase($config['dbLink']);
286         $changes = array();
287         foreach ($new as $column => $value) {
288             if ($value !== $old[$column]) { // is the value changed?
289                 $changes[] = $db->quoteIdentifier($column).' = '.$this->quote($db, $column, $value);
290             }
291         }
292         if (count($changes) == 0) {
293             return $new; // No changes, no query
294         }
295         $where = $this->generateWhere($old, $config);
296         $result = $db->exec('UPDATE '.$db->quoteIdentifier($config['table']).' SET '.implode(', ', $changes).' WHERE '.$where);
297         if ($result === false) {
298             throw new \Exception('Updating record "'.$where.'"  in "'.$config['table'].'" failed');
299         }
300         return $new;
301     }

303     /**
304      * INSERT record INTO the database.
305      *
306      * @param array $data
307      * @param array $config
308      * @return array
309      */
310     function add($data, $config) {
311         $db = getDatabase($config['dbLink']);
312         $columns = array();
313         $values = array();
314         foreach ($data as $column => $value) {
315             if ($value === value($config['columns'][$column]['default'])) {
316                 continue;
317             }
318             $columns[] = $db->quoteIdentifier($column);
319             $values[] = $this->quote($db, $column, $value);
320         }
321         $result = $db->exec('INSERT INTO '.$db->quoteIdentifier($config['table']).' ('.implode(', ', $columns).') VALUES ('.implode(', ', $values).')');
322         if ($result === false) {
323             throw new \Exception('Adding record into "'.$config['table'].'" failed');
324         }
325         if (count($config['primaryKeys']) == 1) {
326             $idColumn = $config['primaryKeys'][0];
327             if ($data[$idColumn] === null) {
328                 if ($db instanceof \PDO) {
329                     $data[$idColumn] = $db->lastInsertId();
330                 } elseif ($db instanceof \mysqli) {
331                     $data[$idColumn] = $db->insert_id;
332                 } else {
333                     notice('Implement insert_id for '.get_class($db));
334                 }
335             }
336         }
337         return $data;
338     }

342      * @param array $row  The data
343      * @param array $config

345     function delete($row, $config) {
346         $db = getDatabase($config['dbLink']);
347         $where = $this->generateWhere($row, $config);
348         $result = $db->exec('DELETE FROM '.$db->quoteIdentifier($config['table']).' WHERE '.$where);
349         if ($result === false) {
350             throw new \Exception('Deleting record "'.$where.'"  from "'.$config['table'].'" failed');
351         }
352         if ($db instanceof \PDO) {
353             if ($result !== 1) {
354                 throw new \Exception('Removing "'.$where.'" from "'.$config['table'].'" failed, '.$result.' rows were affected');
355             }
356         } elseif ($db instanceof \mysqli) {
357             if ($db->affected_rows != 1) {
358                 throw new \Exception('Removing "'.$where.'" from "'.$config['table'].'" failed, '.$db->affected_rows.' rows were affected');
359             }
360         } else {
361             notice('Implement affected_rows for '.get_class($db));
362         }
363     }

365     /**
366      *
367      * @param array $keys  Array containing the primay keys
368      * @param array $config
369      * @return string
370      */
371     private function generateWhere($keys, $config, $prepare = false) {
372         $db = getDatabase($config['dbLink']);
373         $where = array();
374         foreach ($config['primaryKeys'] as $column) {
375             if (isset($keys[$column]) == false) {
376                 throw new \Exception('Missing key: "'.$column.'"'); // @todo better error
377             }
378             if ($prepare) {
379                 $where[] = $db->quoteIdentifier($column).' = ?';
380             } else {
381                 $where[] = $db->quoteIdentifier($column).' = '.$this->quote($db, $column, $keys[$column]);
382             }
383         }
384         if (count($where) == 0) {
385             throw new \Exception('Invalid config, no "primaryKeys" defined');
386         }
387         return implode(' AND ', $where);
388     }

392      *
393      * @param Database $db
394      * @param string $column
395      * @param mixed $value

397     private function quote($db, $column, $value) {
398         if ((is_int($value) || preg_match('/^[123456789]{1}[0-9]*$/', $value)) && ($column == 'id' || substr($column, -3) == '_id')) {
399             return $value;
400         }
401         return $db->quote($value);

424     /**
425      * Extract the Database schema from a MySQL database
426      * @param Database $db
427      * @return array  schema definition
428      */
429     private function getSchemaMySql($db, $prefix = '') {
430         $schema = array();
431         if ($prefix != '') {
432             $tables = $db->query('SHOW TABLES LIKE '.$db->quote($prefix.'%'));
433         } else {
434             $tables = $db->query('SHOW TABLES');
435         }
436         foreach ($tables as $row) {
437             $table = current($row);
438             $referencedBy = isset($schema[$table]) ? $schema[$table]['referencedBy'] : array();
439             $schema[$table] = array(
440                 'table' => $table,
441                 'columns' => array(),
442                 'primaryKeys' => array(),
443                 'referencedBy' => $referencedBy,
444             );
445             $config = &$schema[$table];
446             $showCreate = $db->fetchRow('SHOW CREATE TABLE '.$db->quoteIdentifier($table));
447             $createSyntax = $showCreate['Create Table'];
448             $lines = explode("\n", $createSyntax);
449 
450             unset($lines[0]); // CREATE TABLE * (
451             unset($lines[count($lines)]); // ) ENGINE=*
452             foreach ($lines as $line) {
453                 $line = preg_replace('/^  |,$/', '', $line); // "  " & "," weghalen
454                 $line = str_replace('NOT ', 'NOT_', $line);
455                 $line = str_replace(' KEY', '_KEY', $line);
456                 $line = str_ireplace('CHARACTER SET', 'CHARACTER_SET', $line);
457                 $line = str_ireplace('ON UPDATE', 'ON_UPDATE', $line);
458                 $line = str_replace('  ', ' ', $line);
459                 $parts = explode(' ', $line);
460                 if (substr($parts[0], 0, 1) == '`') { // Column description
461                     $column = substr($parts[0], 1, -1);
462                     $config['columns'][$column] = array(
463                         'type' => $parts[1],
464                         'null' => true,
465                     );
466                     $columnConfig = &$config['columns'][$column];
467                     for ($i = 2; $i < count($parts); $i++) {
468                         $part = $parts[$i];
469 
470                         switch (strtoupper($part)) {
471                             case 'NOT_NULL';
472                                 $columnConfig['null'] = false;
473                                 break;
474                             case 'NULL';
475                                 $columnConfig['null'] = true;
476                                 break;
477 
478                             case 'AUTO_INCREMENT': break;
479 
480                             case 'DEFAULT':
481                                 $default = '';
482                                 while ($part = $parts[$i + 1]) {
483                                     $i++;
484                                     $default .= $part;
485                                     if (substr($default, 0, 1) != "'") { // Not a quoted string value?
486                                         break; // end for loop
487                                     }
488                                     if (substr($default, -1) == "'") { // End of quoted string?
489                                         break; // end for loop
490                                     }
491                                     $default .= ' ';
492                                 }
493                                 if (substr($default, 0, 1) == "'") {
494                                     $config['columns'][$column]['default'] = substr($default, 1, -1); // remove quotes
495                                 } else {
496                                     switch ($default) {
497                                         case 'NULL';
498                                             $default = null;
499                                             break;
500                                         case 'CURRENT_TIMESTAMP':
501                                             $default = null;
502                                             break;
503                                         default:
504                                             notice('Unknown default "'.$default.'" in "'.$line.'"');
505                                             break;
506                                     }
507                                     $config['columns'][$column]['default'] = $default;
508                                 }
509                                 break;
510 
511                             case 'UNSIGNED':
512                                 $config['columns'][$column]['type'] = 'unsigned '.$config['columns'][$column]['type'];
513                                 break;
514 
515                             case 'COMMENT':
516                                 $i += $this->stripQuotedValue($parts, $comment, $i);
517                                 $config['columns'][$column]['comment'] = $comment;
518                                 break;
519 
520                             case 'ON_UPDATE':
521                             case 'CHARACTER_SET':
522                                 $i++; // ignore value
523                                 break;
524 
525                             default:
526                                 notice('Unknown part "'.$part.'" in "'.$line.'"');
527                                 dump($parts);
528                                 break;
529                         }
530                     }
531                 } else { // Key description
532                     $exploded = explode(' ', $line);
533                     $parts = array();
534                     for ($i = 0; $i < count($exploded); $i++) {
535                         $part = $exploded[$i];
536                         if (substr($part, 0, 1) === '`') {
537                             $i--;
538                             $i += $this->stripQuotedValue($exploded, $value, $i, '`');
539                             $parts[] = $value;
540                         } else {
541                             $parts[] = str_replace('`', '', $part); // @todo Parse the "(`id`)" parts. Spaces in columnnames are unlikely but possible.
542                         }
543                     }
544                     switch ($parts[0]) {
545                         case 'PRIMARY_KEY':
546                             $config['primaryKeys'] = explode(',', substr($parts[1], 1, -1));
547                             break;
548 
549                         case 'KEY':
550                         case 'UNIQUE_KEY';
551                             break; // Skip
552 
553                         case 'CONSTRAINT':
554                             if ($parts[2] != 'FOREIGN_KEY') {
555                                 notice('Unknown constraint: "'.$line.'"');
556                                 break;
557                             }
558                             if ($parts[4] != 'REFERENCES') {
559                                 notice('Unknown foreign key: "'.$line.'"');
560                                 break;
561                             }
562                             $column = substr($parts[3], 1, -1);
563                             $foreignTable = $parts[5];
564                             $foreignColumn = substr($parts[6], 1, -1);
565                             $config['columns'][$column]['foreignKeys'][] = array(
566                                 'table' => $foreignTable,
567                                 'column' => $foreignColumn
568                             );
569                             $schema[$foreignTable]['referencedBy'][] = array('table' => $table, 'column' => $column);
570                             break;
571 
572                         default:
573                             notice('Unknown metadata "'.$parts[0].'" in "'.$line.'"');
574                             dump($parts);
575                             break;
576                     }
577                 }
578             }
579             unset($config);
580         }
581         return $schema;
582     }

584     /**
585      * Extract the Database schema from a Sqlite database
586      * @param Database $dbLink
587      * @return array  schema definition
588      */
589     private function getSchemaSqlite($db, $prefix = false) {
590         $schema = array();
591         $sql = 'SELECT tbl_name FROM sqlite_master WHERE type = "table" AND name != "sqlite_sequence"';
592         if ($prefix) {
593             $sql .= ' tbl_name LIKE '.$db->quote($prefix.'%');
594         }
595         $tables = $db->query($sql)->fetchAll();
596         // Pass 1: columns
597         foreach ($tables as $row) {
598             $table = $row['tbl_name'];
599 
600             $schema[$table] = array(
601                 'table' => $table,
602                 'columns' => array(),
603                 'primaryKeys' => array(),
604             );
605             $config = &$schema[$table];
606             if (array_key_exists('referencedBy', $config) == false) {
607                 $config['referencedBy'] = array();
608             }
609 
610             $columns = $db->query('PRAGMA table_info('.$table.')');
611             foreach ($columns as $column) {
612                 $name = $column['name'];
613                 $config['columns'][$name] = array(
614                     'type' => $column['type'],
615                     'null' => ($column['notnull'] == '0'),
616                 );
617                 if ($column['dflt_value'] !== null) {
618                     $config['columns'][$name]['default'] = $column['dflt_value'];
619                 }
620 
621                 if ($column['pk']) {
622                     $config['primaryKeys'][] = $name;
623                 }
624             }
625         }

654         if (preg_match('/^(.+)(_id|_code)$/i', $column, $match)) { // A "_code", "_id", "_Id" or "_ID" suffix?

663     /**
664      *
665      * @param type $parts
666      * @param type $value
667      * @param type $offset
668      * @param type $open
669      * @param type $close
670      * @return int
671      */
672     private function stripQuotedValue($parts, &$value, $offset = 0, $open = "'", $close = null) {
673         $value = '';
674         $i = $offset;
675         if ($close === null) {
676             $close = $open;
677         }
678         while (array_key_exists($i + 1, $parts)) {
679             $value .= $parts[$i + 1];
680             if ($i === $offset && substr($value, 0, 1) != $open) { // Not quoted?
681                 return 1;
682             }
683             $i++;
684             if (substr($value, -1) == $close) { // Last part of quoted string?
685                 break; // end for loop
686             }
687             $value .= ' '; // re-add the exploded space
688         }
689         $value = substr($value, 1, -1); // strip quotes
690         return $i - $offset;
691     }
