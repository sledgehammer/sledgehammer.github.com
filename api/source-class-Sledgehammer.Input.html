 50     function import(&$error, $request = null) {
 51         if ($request === null) {
 52             $request = $_REQUEST;
 53         }
 54         switch ($this->type) {
 55 
 56             case 'file':
 57                 // Import a file upload

 64                 }
 65                 if (array_key_exists($this->name, $_FILES) == false) {
 66                     $error = 'Invalid name';
 67                     return null;
 68                 }

111     function render() {
112         if ($this->label === null) {
113             $this->renderElement();
114         } else {
115             if (in_array($this->type, array('checkbox', 'radio'))) {
116                 echo '<label>';
117                 $this->renderElement();
118                 echo '&nbsp;', Html::escape($this->label), '</label>';
119             } else {
120                 echo '<label>', Html::escape($this->label), '</label>';
121                 $this->renderElement();
122             }
123             return;
124         }

127     protected function renderElement() {
128         $attributes = $this->attributes;
129         if ($this->name !== null) {
130             array_key_unshift($attributes, 'name', $this->name);
131         }
132         switch ($this->type) {
133             case 'select':
134                 $options = $attributes['options'];
135                 unset($attributes['options']);
136                 echo Html::element('select', $attributes, true);
137                 $isIndexed = is_indexed($options);
138                 foreach ($options as $value => $label) {
139                     $option = array();
140                     if ($isIndexed) {
141                         $value = $label;
142                     } else {
143                         $option['value'] = $value;
144                     }
145                     if (equals($value, $this->value)) {
146                         $option['selected'] = 'selected';
147                     }
148                     echo Html::element('option', $option, Html::escape($label));
149                 }
150                 echo '</select>';
151                 break;
152 
153             case 'textarea':
154                 echo Html::element('textarea', $attributes, Html::escape($this->value));
155                 break;
156 
157             default:
158                 array_key_unshift($attributes, 'type', $this->type);
159                 if ($this->value !== null) {
160                     $attributes['value'] = $this->value;
161                 }
162                 echo Html::element('input', $attributes);
163                 break;
164         }
165     }
