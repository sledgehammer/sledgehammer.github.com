 24     /**
 25      * @param array $options  array(
 26      *  'etag'=> bool,
 27      *  'file_get_contents' => bool,
 28      * )
 29      */
 30     function __construct($filename, $options = array('etag' => false)) {
 31         $this->filename = $filename;
 32         $this->etag = array_value($options, 'etag');
 33         if (!file_exists($filename)) {
 34             if (basename($filename) == 'index.html') {
 35                 $this->error = new HttpError(403);
 36             } else {
 37                 $this->error = new HttpError(404);
 38             }
 39             return;
 40         }
 41         $last_modified = filemtime($filename);
 42         if ($last_modified === false) {
 43             $this->error = new HttpError(500);
 44             return;
 45         }
 46         if (array_key_exists('HTTP_IF_MODIFIED_SINCE', $_SERVER)) {
 47             $if_modified_since = strtotime(preg_replace('/;.*$/', '', $_SERVER['HTTP_IF_MODIFIED_SINCE']));
 48             if ($if_modified_since >= $last_modified) { // Is the Cached version the most recent?
 49                 $this->notModified = true;
 50                 return;
 51             }
 52         }
 53         if ($this->etag) {
 54             $etag = md5_file($filename);
 55             if (array_value($_SERVER, 'HTTP_IF_NONE_MATCH') === $etag) {
 56                 $this->notModified = true;
 57                 return;
 58             }
 59             $this->headers[] = 'ETag: '.md5_file($filename);
 60         }
 61         $this->notModified = false;
 62         if (is_dir($filename)) {
 63             $this->error = new HttpError(403);
 64             return;
 65         }
 66         $this->headers['Content-Type'] = mimetype($filename);
 67         $this->headers['Last-Modified'] = gmdate('r', $last_modified);
 68         $filesize = filesize($filename);
 69         if ($filesize === false) {
 70             $this->error = new HttpError(500);
 71             return;
 72         }
 73         $this->headers['Content-Length'] = $filesize; // @todo Detecteer bestanden groter dan 2GiB, deze geven fouten.
 74         if (array_value($options, 'file_get_contents')) {
 75             $this->fileContents = file_get_contents($filename);
 76         }
 77     }

101             echo $this->fileContents;
102         } else {
103             readfile($this->filename);
104         }
