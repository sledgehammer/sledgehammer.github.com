 20     /**
 21      * Constructor
 22      *
 23      * @param mixed  $data      The data to be sent as json.
 24      * @param string $charset   The encoding used in $data, use null for autodetection. Assume UTF-8 by default.
 25      */
 26     function __construct($data, $charset = 'UTF-8') {
 27         if (strtoupper($charset) !== 'UTF-8') {
 28             $this->data = $this->convertToUTF8($data, $charset);
 29         } else {
 30             $this->data = $data;
 31         }

 35      * Change Content-Type to "application/json"
 36      */
 37     function getHeaders() {
 38         if (count($_FILES) == 0) {
 39             return array(
 40                 'http' => array(
 41                     'Content-Type' => 'application/json',
 42                 )
 43             );
 44         } else {
 45             // Als er bestanden ge-upload zijn, gaat het *niet* om een XMLHttpRequest, maar waarschijnlijk om een upload naar een hidden iframe via javascript.
 46             // Een "application/json" header zal dan een ongewenste download veroorzaken.
 47             // (Of als de JSONView extensie is geinstalleerd, wordt de json versmurft als html)

 89     /**
 90      * Returns a string containing the JSON representation of value.
 91      *
 92      * @param mixed $data
 93      * @param array|int $options
 94      * @throws \Exception
 95      * @return string JSON formatted string
 96      */
 97     static function encode($data, $options = array()) {
 98         if (is_int($options)) {
 99             $optionMask = $options;
100         } else {
101             $optionMask = 0;
102             foreach ($options as $option) {
103                 $optionMask += $option;
104             }
105         }
106         $json = json_encode($data, $optionMask);
107         $error = json_last_error();
108         if ($error !== JSON_ERROR_NONE) {
109             throw new \Exception(self::errorMessage($error), $error);
110         }
111         return $json;
112     }
113 
114     /**
115      * Takes a JSON encoded string and converts it into a PHP variable.
116      *
117      * @param string $json JSON formatted string
118      * @param bool $assoc
119      * @param int $depth
120      * @param array|int $options
121      * @throws \Exception
122      * @return mixed data
123      */
124     static function decode($json, $assoc = false, $depth = 512, $options = array()) {
125         if (is_int($options)) {
126             $optionMask = $options;
127         } else {
128             $optionMask = 0;
129             foreach ($options as $option) {
130                 $optionMask += $option;
131             }
132         }
133         if ($optionMask === 0) {
134             $data = json_decode($json, $assoc, $depth);
135         } else {
136             $data = json_decode($json, $assoc, $depth, $optionMask); //Opstions available since php 5.4.0
137         }
138         $error = json_last_error();
139         if ($error !== JSON_ERROR_NONE) {
140             throw new \Exception(self::errorMessage($error), $error);
141         }
142         return $data;
143     }

145     /**
146      * Reports the error/exception to the ErrorHandler and returns the error as Json object.
147      * The javascript client should detect and report the error to the user:
148      *   if (result.succes !== true) { alert(result.error); }
149      *
150      * @param string|Exception $error  The error message or Exception
151      * @return Json
152      */
153     static function error($error) {
154         if ($error instanceof \Exception) {
155             report_exception($error);
156             $error = $error->getMessage();
157         } else {
158             warning($error);
159         }
160         return new Json(array(
161             'success' => false,
162             'error' => $error
163         ));
164     }

184     /**
185      * Return a data-structure where all string are UTF-8 encoded.
186      *
187      * @param mixed       $data     The non UTF-8 encoded data
188      * @param string|null $charset  The from_encoding, Use null for autodetection
189      * @return mixed  UTF8 encoded data
190      */
191     private function convertToUTF8($data, $charset) {
192         if (is_string($data)) {
193             return mb_convert_encoding($data, 'UTF-8', $charset);
194         }
195         if (is_object($data)) {
196             $data = get_object_vars($data);
197         }
198         if (is_array($data)) {
199             foreach ($data as $key => $value) {
200                 $data[$key] = $this->convertToUTF8($value, $charset);
201             }
202             return $data;
203         }
204         // Is $data a integer, float, etc
205         return $data;
206     }

208     /**
209      * Translates a json error into a human readable error.
210      *
211      * @param int $errno JSON_ERROR_*
212      * @return string
213      */
214     private static function errorMessage($errno) {
215         static $lookup = null;
216         if ($lookup === null) {
217             $lookup = array();
218             $constants = get_defined_constants();
219             foreach ($constants as $constant => $constant_value) {
220                 if (substr($constant, 0, 11) === 'JSON_ERROR_') {
221                     $lookup[$constant_value] = $constant;
222                 }
223             }
224         }
225         if (array_key_exists($errno, $lookup)) {
226             $message = '['.$lookup[$errno].'] ';
227         } else {
228             $message = '['.$errno.'] ';
229         }
230         switch ($errno) {
231             case JSON_ERROR_NONE:
232                 $message .= 'No errors';
233                 break;
234             case JSON_ERROR_DEPTH:
235                 $message .= 'The maximum stack depth has been exceeded';
236                 break;
237             case JSON_ERROR_STATE_MISMATCH:
238                 $message .= 'Invalid or malformed JSON';
239                 break;
240             case JSON_ERROR_CTRL_CHAR:
241                 $message .= 'Control character error, possibly incorrectly encoded';
242                 break;
243             case JSON_ERROR_SYNTAX:
244                 $message .= 'Syntax error';
245                 break;
246             case JSON_ERROR_UTF8:
247                 $message .= 'Malformed UTF-8 characters, possibly incorrectly encoded';
248                 break;
249             default: $message = 'Unknown error';
250         }
251         return $message;
252     }
