14 class Nav extends Object implements View {

22      * Attributes for the ul element.
23      * @var array
24      */
25     protected $attributes = array(
26         'class' => 'nav',
27     );
28 
29     /**
30      * Constructor
31      * @param array $items format: array(url => label, ...) of array(url => array('icon' => icon_url, 'label' => label))
32      * @param $options array
33      */
34     function __construct($items, $options = array()) {
35         $this->items = $items;
36         // Set attributes and properties
37         foreach ($options as $option => $value) {
38             if (property_exists($this, $option)) {
39                 $this->$option = $value;
40             } else {
41                 $this->attributes[$option] = $value;
42             }
43         }
44     }

46     /**
47      * Render the navigation
48      */
49     function render() {
50         echo HTML::element('ul', $this->attributes, true), "\n";
51         $this->renderContents();
52         echo '</ul>';
53     }
54 
55     /**
56      * Render the items.
57      */
58     function renderContents() {
59         foreach ($this->items as $url => $action) {
60             if (is_int($url) && is_string($action)) {
61                 echo "\t<li class=\"nav-header\">".HTML::escape($action)."</li>\n";
62             } else {
63                 echo "\t<li><a href=\"".$url.'">';
64                 if (is_array($action)) { //  has an icon?
65                     echo HTML::icon($action['icon']), ' ', HTML::escape($action['label']);
66                 } else {
67                     echo HTML::escape($action);
68                 }
69                 echo "</a></li>\n";
70             }
71         }
72     }

76      * @param type $items
