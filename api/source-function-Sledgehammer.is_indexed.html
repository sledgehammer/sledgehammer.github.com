 285      * implode(), maar dan zodat deze leesbaar is voor mensen.
 286      * Bv: echo human_implode(' of ', array('appel', 'peer', 'banaan')); wordt 'appel, peer of banaan'
 287      *
 288      * @param string $glueLast deze wordt gebruikt tussen de laaste en het eennalaatste element. bv: ' en ' of ' of '
 289      * @param array $array
 290      * @param string $glue
 291      * @return string
 292      */
 293     function human_implode($glueLast, $array, $glue = ', ') {
 294         if (!$array || !count($array)) { // sanity check
 295             return '';
 296         }
 297         $last = array_pop($array); // get last element
 298         if (!count($array)) { // if it was the only element - return it
 299             return $last;
 300         }
 301         return implode($glue, $array).$glueLast.$last;
 302     }

 399     /**
 400      * Compare two values with the given operator.
 401      * For '==' the values are compared with strcasecmp().
 402      *
 403      * compare("1",'==', 1) == true
 404      * compare("abc", '==', "ABC") == true
 405      * compare("0", '==', 0) == true
 406      * compare("", '==', 0) == false
 407      *
 408      * @see CollectionTests for more examples
 409      *
 410      * @param mixed $value
 411      * @param string $operator
 412      * @param mixed $expectation
 413      * @return bool
 414      */
 415     function compare($value, $operator, $expectation) {
 416         switch ($operator) {
 417             case '==': return equals($value, $expectation);
 418             case '!=': return !equals($value, $expectation);
 419             case '<': return $value < $expectation;
 420             case '>': return $value > $expectation;
 421             case '<=': return $value <= $expectation;
 422             case '>=': return $value >= $expectation;
 423             case 'IN':
 424                 foreach ($expectation as $val) {
 425                     if (equals($value, $val)) {
 426                         return true;
 427                     }
 428                 }
 429                 return false;
 430         }
 431         throw new \Exception('Invalid operator: "'.$operator.'" use '.quoted_human_implode(' or ', explode('|', COMPARE_OPERATORS)));
 432     }

 681      * Get the timestamp for the latest change in the directory.
 682      *
 683      * @param string $path
 684      * @return int
 685      */
 686     function mtime_folders($path) {
 687         $max_ts = filemtime($path); // Vraag de mtime op van de map
 688         if ($max_ts === false) { // Bestaat het $path niet?
 689             return false;
 690         }
 691         if (substr($path, -1) != '/') {
 692             $path .= '/';
 693         }
 694         // Controleer of een van de bestanden of submappen een nieuwere mtime heeft.
 695         $dir = opendir($path);
 696         if ($dir) {
 697             while (($filename = readdir($dir)) !== false) {
 698                 if ($filename === '.' || $filename === '..') {
 699                     continue;
 700                 }
 701                 $filepath = $path.$filename;
 702                 if (is_dir($filepath)) {
 703                     $ts = mtime_folders($filepath.'/');
 704                 } else {
 705                     $ts = filemtime($filepath);
 706                 }
 707                 if ($ts > $max_ts) { // Heeft de submap een nieuwere timestamp?
 708                     $max_ts = $ts;
 709                 }
 710             }
 711             closedir($dir);
 712         }
 713         return $max_ts;
 714     }

 899                         $charactersLeft = 8;

 912                 $title .= get_class($value).",\n";
 913             } else {
 914                 $title .= $value.','."\n";
 915             }
 916         }

1306             $if_modified_since = strtotime(preg_replace('/;.*$/', '', $_SERVER['HTTP_IF_MODIFIED_SINCE']));

1308                 header($_SERVER['SERVER_PROTOCOL'].' 304 Not Modified');
1309                 exit();
1310             }
1311         }
1312         $headers = array();
1313         /*
1314           $resume_support = false; // @todo Bestanden in tmp/ geen resume_support geven.
1315           if ($resume_support) {
1316           $headers[] = 'Accept-Ranges: bytes';
1317           } */
1318         if (is_dir($filename)) {
1319             throw new \Exception('Unable to render_file(). "'.$filename.'" is a folder');
1320         }
1321         $headers['Content-Type'] = mimetype($filename);
1322         $headers['Last-Modified'] = gmdate('r', $last_modified);
1323         $filesize = filesize($filename);
1324         if ($filesize === false) {
1325             throw new \Exception('Filesize unknown');

1327         if (empty($_SERVER['HTTP_RANGE'])) {
1328             $headers['Content-Length'] = $filesize; // @todo Detectie inbouwen voor bestanden groter dan 2GiB, deze geven fouten.
1329             send_headers($headers);
1330             // Output buffers uitschakelen, anders zal readfile heb bestand in het geheugen inladen. (en mogelijk de memory_limit overschrijden)
1331             while (ob_get_level() > 0) {
1332                 ob_end_flush();

1334             $success = readfile($filename); // Het gehele bestand sturen
1335             if ($success) {

1338             throw new \Exception('readfile() failed');
1339         } else {

1341             $success = readfile($filename);

1344             }

1359         }

1362         // also check for invalid ranges.

1364         $seek_start = (empty($seek_start) || $seek_end < abs(intval($seek_start))) ? 0 : max(abs(intval($seek_start)), 0);

1378         set_time_limit(0);
