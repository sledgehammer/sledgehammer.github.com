  6 /**
  7  * HTTP error page
  8  * Sends the correct HTTP header and displays an page with the error.
  9  *
 10  * @todo Add support for all known HTTP errors http://www.w3.org/Protocols/rfc2616/rfc2616-sec10.html
 11  * @package MVC
 12  */
 13 class HttpError extends Object implements View {
 14 
 15     /**
 16      * The HTTP ErrorCode (404, 500, etc)
 17      * @var int
 18      */
 19     private $errorCode;
 20     /**
 21      * @var array
 22      */
 23     private $options;
 24 
 25     /**
 26      * @param int $statusCode  HTTP Foutcode van de fout 404,403 enz
 27      * @param array $options  [optional] Array with additional settings
 28      *   notice: Report a notice after render()
 29      *   warning: Report a warning after render()
 30      *   exception: Report an exception after render()
 31      */
 32     function __construct($errorCode, $options = array()) {
 33         $this->errorCode = $errorCode;
 34         $this->options = $options;
 35     }
 36 
 37     function getHeaders() {
 38         $error = $this->getError();
 39         return array(
 40             'title' => $this->errorCode.' - '.$error['title'],
 41             'http' => array('Status' => $this->errorCode.' '.$error['header']),
 42         );
 43     }
 44 
 45     /**
 46      * Render an error page.
 47      *
 48      * @return void
 49      */
 50     function render() {
 51         $error = $this->getError();
 52         $messageBox = new Template('HttpError.php', $error);
 53         $messageBox->render();
 54         foreach ($this->options as $option => $value) {
 55             switch ((string) $option) {
 56 
 57                 case 'notice':
 58                 case 'warning':
 59                     $function = $option;
 60                     if (is_array($value)) {
 61                         call_user_func_array($function, $value);
 62                     } else {
 63                         call_user_func($function, $value);
 64                     }
 65                     break;
 66 
 67                 case 'exception':
 68                     report_exception($value);
 69                     break;
 70 
 71                 default:
 72                     notice('Unknown option: "'.$option.'"', array('value' => $value));
 73                     break;
 74             }
 75         }
 76     }
 77 
 78     private function getError() {
 79 
 80         switch ($this->errorCode) {
 81 
 82             case 400:
 83                 return array(
 84                     'header' => 'Bad Request',
 85                     'icon' => 'error',
 86                     'title' => 'Bad Request',
 87                     'message' => 'Server begreep de aanvraag niet'
 88                 );
 89 
 90             case 401:
 91                 return array(
 92                     'header' => 'Unauthorized',
 93                     'icon'=> 'warning',
 94                     'title' => 'Niet geauthoriseerd',
 95                     'message' => 'U heeft onvoldoende rechten om deze pagina te bekijken.',
 96                 );
 97 
 98             case 403:
 99                 return array(
100                     'header' => 'Forbidden',
101                     'icon'=> 'warning',
102                     'title' => 'Verboden toegang',
103                     'message' => (substr(Url::getCurrentURL()->path, -1) == '/') ? 'U mag de inhoud van deze map niet bekijken' : 'U mag deze pagina niet bekijken',
104                 );
105 
106             case 404:
107                 return array(
108                     'header' => 'Not Found',
109                     'icon'=> 'warning',
110                     'title' => 'Bestand niet gevonden',
111                     'message' => 'De opgegeven URL "'.Url::getCurrentURL().'" kon niet worden gevonden.',
112                 );
113 
114             case 500:
115                 return array(
116                     'header' => 'Internal Server Error',
117                     'icon'=> 'error',
118                     'title' => 'Interne serverfout',
119                     'message' => 'Er is een interne fout opgetreden, excuses voor het ongemak.',
120                 );
121 
122             case 501:
123                 return array(
124                     'header' => 'Not Implemented',
125                     'icon' => 'error',
126                     'title' => 'Not Implemented',
127                     'message' => 'Dit wordt niet door de server ondersteund'
128                 );
129 
130             default:
131                 throw new \Exception('HTTP errorCode '.$this->errorCode.' is not (yet) supported.');
132 
133         }
134     }
