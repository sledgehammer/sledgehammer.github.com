 87     function open($filename) {
 88         $tokens = new PHPTokenizer(file_get_contents($filename));
 89 
 90         $namespace = '';
 91         $uses = array();
 92         $definitions = array();
 93         $definition = array(
 94             'level' => -1
 95         );
 96         $globalFunctions = array();
 97         $functions = &$globalFunctions;
 98         $level = 0;
 99         $parameterLevel = -1; // skip parameters in anonymous functions
100         foreach ($tokens as $token) {
101             $type = $token[0];
102             $value = $token[1];
103             if ($value == '' && $type != 'T_NAMESPACE') {
104                 notice('Empty token', $token);
105             }
106             if ($type == 'T_PHP' || $type == 'T_HTML') {
107                 continue;
108             }
109             switch ($type) {
110 
111                 case 'T_NAMESPACE':
112                     $namespace = $value;
113                     break;
114 
115                 case 'T_USE':
116                     $pos = strrpos($value, '\\');
117                     $namespaceAlias = substr($value, $pos + 1);
118                     $uses[$namespaceAlias] = $value;
119                     break;
120 
121                 case 'T_USE_ALIAS':
122                     $uses[$value] = $uses[$namespaceAlias];
123                     unset($uses[$namespaceAlias]);
124                     break;
125 
126 
127                 case 'T_INTERFACE':
128                     $definitions[] = array(
129                         'type' => 'INTERFACE',
130                         'namespace' => $namespace,
131                         'interface' => $value,
132                         'identifier' => $this->prefixNamespace($namespace, $value, $uses),
133                         'extends' => array(),
134                         'methods' => array(),
135                         'level' => $level
136                     );
137                     $definition = &$definitions[count($definitions) - 1];
138                     break;
139 
140                 case 'T_CLASS':
141                     $definitions[] = array(
142                         'type' => 'CLASS',
143                         'namespace' => $namespace,
144                         'class' => $value,
145                         'identifier' => $this->prefixNamespace($namespace, $value, $uses),
146                         'extends' => array(),
147                         'implements' => array(),
148                         'methods' => array(),
149                         'level' => $level
150                     );
151                     $definition = &$definitions[count($definitions) - 1];
152                     break;
153 
154                 case 'T_EXTENDS':
155                     $extends = $this->prefixNamespace($namespace, $value, $uses);
156                     $definition['extends'][] = $extends;
157                     $this->addUsedIn($extends, $filename, $token[2]);
158                     break;
159 
160                 case 'T_IMPLEMENTS':
161                     $interface = $this->prefixNamespace($namespace, $value, $uses);
162                     $definition['implements'][] = $interface;
163                     $this->addUsedIn($interface, $filename, $token[2]);
164                     break;
165 
166                 case 'T_FUNCTION':
167                     $function = $value;
168                     $parameter = null;
169                     if ($level == ($definition['level'] + 1)) {
170                         $definition['methods'][$function] = array();
171                         $functions = &$definition['methods'];
172                     } else {
173                         $functions = &$globalFunctions;
174                     }
175                     $parameterLevel = $level;
176                     break;
177 
178                 case 'T_TYPE_HINT':
179                     if (strtolower($value) !== 'array') {
180                         $this->addUsedIn($this->prefixNamespace($namespace, $value, $uses), $filename, $token[2]);
181                     }
182                     break;
183 
184                 case 'T_PARAMETER':
185                     if ($parameterLevel != $level) { // Doesn't this parameter belong to the function?
186                         break; // Propably a catch () parameter
187                     }
188                     $parameter = substr($value, 1); // strip '$'
189                     $functions[$function][$parameter] = null;
190                     break;
191 
192                 case 'T_PARAMETER_VALUE':
193                     $functions[$function][$parameter] = $value;
194                     $parameter = null;
195                     break;
196 
197                 case 'T_OPEN_BRACKET':
198                     $level++;
199                     break;
200 
201                 case 'T_CLOSE_BRACKET':
202                     $level--;
203                     break;
204 
205                 case 'T_CALL':
206                     $this->addCalledIn($value, $filename, $token[2]);
207                     break;
208 
209                 case 'T_METHOD_CALL':
210                     break;
211 
212                 case 'T_OBJECT':
213                     $this->addUsedIn($this->prefixNamespace($namespace, $value, $uses), $filename, $token[2]);
214                     break;
215 
216                 default:
217                     notice('Unexpected tokenType: "'.$type.'"');
218                     break;
219             }
220         }
221         if ($level != 0) {
222             notice('Level: '.$level.' Number of "{" doesn\'t match the number of "}"');
223         }
224         unset($definition);
225         // Add definitions to de loader
226         foreach ($definitions as $index => $definition) {
227             $identifier = $definition['identifier'];
228             unset($definition['identifier'], $definition['level']);
229             $definition['filename'] = $filename;
230 //          $duplicate = false;
231 //          if (isset($this->classes[$identifier])) {
232 //              $duplicate = $this->classes[$identifier];
233 //          } elseif (isset($this->interfaces[$identifier])) {
234 //              $duplicate = $this->interfaces[$identifier];
235 //          }
236 //          if ($duplicate) {
237 //              $this->parserNotice('"'.$identifier.'" is ambiguous, it\'s found in multiple files: "'.$duplicate['filename'].'" and "'.$definition['filename'].'"');
238 //          }
239             switch ($definition['type']) {
240 
241                 case 'CLASS':
242                     unset($definition['type']);
243                     if (count($definition['extends']) > 1) {
244                         notice('Class: "'.$definition['class'].'" Multiple inheritance is not allowed for classes');
245                         $definition['extends'] = $definition['extends'][0];
246                     } elseif (count($definition['extends']) == 1) {
247                         $definition['extends'] = $definition['extends'][0];
248                     } else {
249                         unset($definition['extends']);
250                     }
251                     if (count($definition['implements']) == 0) {
252                         unset($definition['implements']);
253                     }
254                     $this->classes[$identifier] = $definition;
255                     break;
256 
257                 case 'INTERFACE':
258                     unset($definition['type']);
259                     $this->interfaces[$identifier] = $definition;
260                     break;
261 
262                 default:
263                     throw new \Exception('Unsupported type: "'.$definition['type'].'"');
264             }
265         }

268     /**
269      * Extract definition information.
270      *
271      * @param string $definition
272      * @return array
273      */
274     function getInfo($definition) {
275         // Check analyzed definitions
276         if (isset($this->classes[$definition])) {
277             return $this->classes[$definition];
278         }
279         if (isset($this->interfaces[$definition])) {
280             return $this->interfaces[$definition];
281         }
282         $filename = $this->getAutoLoader()->getFilename($definition);
283         if ($filename !== null) {
284             $this->open($filename);
285         } elseif (class_exists($definition, false) || interface_exists($definition, false)) {
286             $this->getInfoWithReflection($definition);
287         }
288         if (isset($this->classes[$definition])) {
289             return $this->classes[$definition];
290         }
291         if (isset($this->interfaces[$definition])) {
292             return $this->interfaces[$definition];
293         }
294         throw new \Exception('Definition "'.$definition.'" is not found');
295     }

297     /**
298      * Use PHP's Reflection classes to extract definition information.
299      *
300      * @param string $definition Class or Interace name
301      * @return array
302      */
303     function getInfoWithReflection($definition) {
304         if (class_exists($definition) == false && interface_exists($definition) == false) {
305             //throw new \Exception('Definition "'.$definition.'" is unknown');
306         }
307         $reflectionClass = new \ReflectionClass($definition);
308         $info = array(
309             'namespace' => $reflectionClass->getNamespaceName()
310         );
311         $class = $reflectionClass->name;
312         if ($reflectionClass->isInterface()) {
313             $info['interface'] = $class;
314             $info['extends'] = $reflectionClass->getInterfaceNames();
315         } else {
316             $info['class'] = $class;
317             $info['implements'] = $reflectionClass->getInterfaceNames();
318             $info['extends'] = $reflectionClass->getParentClass();
319             if ($info['extends'] == false || $info['extends']->name == 'stdClass') {
320                 unset($info['extends']);
321             } else {
322                 $info['extends'] = $info['extends']->name;
323             }
324         }
325         $info['methods'] = array();
326         foreach ($reflectionClass->getMethods() as $reflectionMethod) {
327             if ($reflectionMethod->class != $class) {
328                 continue; // De methoden uit de parentclass negeren
329             }
330             $method = $reflectionMethod->name;
331             $info['methods'][$method] = array();
332             foreach ($reflectionMethod->getParameters() as $reflectionParameter) {
333                 $parameter = $reflectionParameter->name;
334                 $info['methods'][$method][$parameter] = null;
335                 if ($reflectionParameter->isDefaultValueAvailable()) {
336                     $value = $reflectionParameter->getDefaultValue();
337                     $info['methods'][$method][$parameter] = $value;
338                 }
339             }
340         }
341         if ($reflectionClass->isInterface()) {
342             $this->interfaces[$definition] = $info;
343         } else {
344             $this->classes[$definition] = $info;
345         }
346         return $info;
347     }

370     /**
371      * Resolve the full classname.
372      *
373      * @param string $namespace  Active namespace
374      * @param string $identifier  The class or interface name
375      * @param array $uses  Active USE statements.
376      * @return string
377      */
378     private function prefixNamespace($namespace, $identifier, $uses = array()) {
379         $pos = strpos($identifier, '\\');
380         if ($pos !== false) {
381             if ($pos === 0) {
382                 // Fully qualified name (\Foo\Bar)
383                 return substr($identifier, 1);
384             }
385             // Qualified name (Foo\Bar)
386             foreach ($uses as $alias => $namespace) {
387                 $alias .= '\\';
388                 if (substr($identifier, 0, strlen($alias)) === $alias) {
389                     return $namespace.substr($identifier, strlen($alias) - 1);
390                 }
391             }
392         } else {
393             // Unqualified name (Foo)
394             if (isset($uses[$identifier])) { // Is an alias?
395                 return $uses[$identifier];
396             }
397         }
398         if ($namespace == '') {
399             return $identifier;
400         }
401         return $namespace.'\\'.$identifier;
402     }
