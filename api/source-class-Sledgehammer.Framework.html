 37         if (defined('Sledgehammer\ENVIRONMENT') === false) {
 38             if (defined('ENVIRONMENT') === false) {
 39                 /**
 40                  * The configured environment. Uses $_SERVER['APPLICATION_ENV'] or uses 'production' as fallback.
 41                  */
 42                 define('Sledgehammer\ENVIRONMENT', isset($_SERVER['APPLICATION_ENV']) ? $_SERVER['APPLICATION_ENV'] : 'production');
 43             } else {
 44                 define('Sledgehammer\ENVIRONMENT', ENVIRONMENT);
 45             }
 46         }
 47 
 48         /**
 49          * Directory of the installed Sledgehammer modules. Usually the "sledgehammer/" folder
 50          */
 51         define('Sledgehammer\MODULES_DIR', dirname(CORE_DIR).DIRECTORY_SEPARATOR);
 52         /**
 53          * Directory of the project.
 54          */
 55         define('Sledgehammer\PATH', dirname(dirname(MODULES_DIR)).DIRECTORY_SEPARATOR);
 56         if (!defined('Sledgehammer\APP_DIR')) {
 57             /**
 58              * Directory for the app.
 59              */
 60             define('Sledgehammer\APP_DIR', PATH.'app'.DIRECTORY_SEPARATOR);
 61         }
 62         if (!defined('Sledgehammer\VENDOR_DIR')) {
 63             /**
 64              * Directory for the app.
 65              */
 66             define('Sledgehammer\VENDOR_DIR', PATH.'vendor'.DIRECTORY_SEPARATOR);
 67         }
 68 
 69         if (version_compare(PHP_VERSION, '5.4.0') >= 0) {
 70             /**
 71              * Errorlevel for all errors messages.
 72              */
 73             define('Sledgehammer\E_MAX', E_ALL);
 74         } else {
 75             // Declare constans missing in PHP 5.3
 76             define('Sledgehammer\SORT_NATURAL', 6); // "natural order" sorting method for Collection->orderBy()
 77             define('Sledgehammer\E_MAX', (E_ALL | E_STRICT)); // E_MAX an error_reporing level that includes all message types (E_ALL doesn't include E_STRICT)
 78             define('T_TRAIT', -1); // Used in the AutoLoader
 79         }
 80 
 81         /**
 82          * Case insensitive "natural order" sorting method for Collection->orderBy()
 83          * Uses natcasesort()
 84         */
 85         define('Sledgehammer\SORT_NATURAL_CI', -2);
 86 
 87         // Detect & create a writable tmp folder
 88         if (defined('Sledgehammer\TMP_DIR') === false) {
 89             $tmpDir = PATH.'tmp'.DIRECTORY_SEPARATOR;
 90             if (is_dir($tmpDir) && is_writable($tmpDir)) { // A writable local tmp folder exist?
 91                 if (function_exists('posix_getpwuid')) {
 92                     $tmpDir .= array_value(posix_getpwuid(posix_geteuid()), 'name').'/';
 93                 }
 94             } else {
 95                 $tmpDir = '/tmp/sledgehammer-'.md5(PATH);
 96                 if (function_exists('posix_getpwuid')) {
 97                     $tmpDir .= '-'.array_value(posix_getpwuid(posix_geteuid()), 'name').'/';
 98                 }
 99             }
100             /**
101              * Directory for temporary files.
102              */
103             define('Sledgehammer\TMP_DIR', $tmpDir);
104         }
105         if (!defined('DEBUG_VAR') && !defined('Sledgehammer\DEBUG_VAR')) {
106             /**
107              * Configure the "?debug=1" to use another $_GET variable.
108              */
109             define('Sledgehammer\DEBUG_VAR', 'debug'); // Use de default DEBUG_VAR "debug"
110         }

174             foreach ($modules as $module) {
175                 $path = $module['path'];
176                 if (file_exists($module['path'].'classes')) { // A sledgehammer folder layout?
177                     $path = $path.'classes'; // Only import the classes folder
178                     $settings = array(); // Use the strict default settings
179                 } else {
180                     // Disable validations
181                     $settings = array(
182                         'matching_filename' => false,
183                         'mandatory_definition' => false,
184                         'mandatory_superclass' => false,
185                         'one_definition_per_file' => false,
186                         'revalidate_cache_delay' => 20,
187                         'detect_accidental_output' => false,
188                     );
189                 }
190                 self::$autoloader->importFolder($path, $settings);
191             }
192         }
193 
194         // Add Autoloader support for the other vendor packages.
195         // Fixes cASe issues, repair namespaces, etc.
196         if (file_exists(VENDOR_DIR)) { // Does the app have vendor packages?
197             self::$autoloader->importFolder(VENDOR_DIR, array(
198                 'matching_filename' => false,
199                 'mandatory_definition' => false,
200                 'mandatory_superclass' => false,
201                 'one_definition_per_file' => false,
202                 'revalidate_cache_delay' => 30,
203                 'detect_accidental_output' => false,
204                 'ignore_folders' => array(VENDOR_DIR.'sledgehammer'),
205                 'cache_level' => 3,

210     /**
211      * List required sledgehammer modules and sort on depedency.
212      * (The module without depedency comes first.)
213      *
214      * @param string $modulesPath
215      * @return array
216      */
217     static function getModules($modulesPath = null) {
218         if ($modulesPath === null) {
219             $modulesPath = MODULES_DIR;
220             $appPath = APP_DIR;
221         } else {
222             $appPath = dirname($modulesPath).DIRECTORY_SEPARATOR.'app'.DIRECTORY_SEPARATOR;
223         }
224         static $cache = array();
225         if (isset($cache[$modulesPath])) {
226             return $cache[$modulesPath];
227         }
228         $required_modules = array();
229         $module_info = array(
230             'app' => array(
231                 'name' => 'app',
232                 'path' => $appPath,
233                 'required_modules' => self::detectModules($modulesPath),
234                 'optional_modules' => array(),
235                 'app' => true
236             )
237         );
238         // Fetch all required_modules
239         self::appendModules($modulesPath, $required_modules, $module_info, 'app', 'detectModules()');
240         if (!file_exists($appPath)) {
241             unset($required_modules[array_search('app', $required_modules)]); // De app is zelf niet required
242         }
243         // Sort by dependancy
244         $sorted_modules = array();
245         $cyclic_dependency_check = count($required_modules) + 1;
246         while (count($required_modules) > 0) { // Loop until all required_modules are sorted
247             if ($cyclic_dependency_check == count($required_modules)) { // No modules are sorted in the previous while loop?
248                 throw new \Exception('Cyclic depedency detected for modules "'.implode('" and "', $required_modules).'"');
249             }
250             $cyclic_dependency_check = count($required_modules);
251             foreach ($required_modules as $index => $module) {
252                 $all_dependancy_are_met = true;
253                 foreach ($module_info[$module]['required_modules'] as $required_module) {
254                     if (!in_array($required_module, $sorted_modules)) { // is the required depencancy not met
255                         $all_dependancy_are_met = false;
256                         break;
257                     }
258                 }
259                 if ($all_dependancy_are_met) {
260                     $sorted_modules[] = $module;
261                     unset($required_modules[$index]);
262                 }
263             }
264         }
265         // Merge the sorted modules with the module_info array
266         $modules = array();
267         foreach ($sorted_modules as $module) {
268             $modules[$module] = $module_info[$module];
269         }
270         $cache[$modulesPath] = $modules;
271         return $modules;
272     }

286         }
287         if (file_exists($path.'functions.php')) {

295     /**
296      * Retreiving module info and find all dependend (required) modules
297      *
298      * @param string $modulesPath  De map waar de "modules" in staan
299      * @param array $required_modules  Dit is een array met reeds toegevoegde modules. Zodat er elke modules maximaal 1x wordt ingeladen.
300      * @param array $module_info  In dit array worden gegevens uit de ini bestanden gezet, zodat deze maar 1x ingeladen worden.
301      * @param string $module  Dit is de naam van de module die toegevoegd zal worden aan de $required_modules (inclusief modules waar deze van afhandelijk is)
302      * @param string $required_by  De {$module} is een afhankelijkheid van de {$required_by} module.
303      */
304     private static function appendModules($modulesPath, &$required_modules, &$module_info, $module, $required_by) {
305         if (in_array($module, $required_modules)) { // Is this module already included
306             return;
307         }
308         $required_modules[] = $module;
309         if (!isset($module_info[$module])) {
310             if ($module === 'app') {
311                 throw new \Exception('Info for the app "module" must be configured');
312             } else {
313                 $module_path = $modulesPath.$module.DIRECTORY_SEPARATOR;
314                 if (file_exists($module_path) == false) {
315                     warning('Module: "'.$module.'" is missing, but is required by "'.$required_by.'"');
316                 } elseif (file_exists($module_path.'composer.json') == false) {
317                     notice('Missing "composer.json" for module: "'.$module.'"', 'Module is required by "'.$required_by.'"');
318                     $module_info[$module] = array('name' => $module);
319                 } else {
320                     $module_info[$module] = json_decode(file_get_contents($module_path.'composer.json'), true);
321                     if ($module_info[$module] === null) {
322                         $constants = get_defined_constants();
323                         $jsonError = json_last_error();
324                         foreach ($constants as $constant => $value) {
325                             if ($value === $jsonError && substr($constant, 0, 10) === 'JSON_ERROR') {
326                                 $jsonError = $constant;
327                                 break;
328                             }
329                         }
330                         warning('"'.$module.'/composer.json" is corrupted', $jsonError);
331                         $module_info[$module] = array('name' => $module);
332                     }
333                 }
334                 $module_info[$module]['path'] = $module_path;
335             }
336             $module_info[$module]['required_modules'] = array();
337             if ($module !== 'core') {
338                 if (empty($module_info[$module]['require'])) {
339                     warning('No "require" found in "'.$module.'" composer.json','Add `"require": { "sledgehammer/core": "*" }` to the composer.json');
340                 } else {
341                     $module_info[$module]['required_modules'] = array();
342                     foreach ($module_info[$module]['require'] as $required_module => $version) {
343                         if (dirname($required_module) == 'sledgehammer') {
344                             $module_info[$module]['required_modules'][] = basename($required_module);
345                         }
346                         // @todo Handle unofficial sledgehammer modules (detect composer.json?)
347                     }
348                 }
349             }
350             $module_info[$module]['optional_modules'] = array();
351             if (isset($module_info[$module]['suggest'])) {
352                 foreach (array_keys($module_info[$module]['suggest']) as $optional_module) {
353                     if (dirname($required_module) == 'sledgehammer') {
354                         $module_info[$module]['optional_modules'][] = basename($optional_module);
355                     }
356                 }
357                 // @todo Handle unofficial sledgehammer modules (detect composer.json?)
358             }
359         }
360         foreach ($module_info[$module]['required_modules'] as $required_dependancy) {
361             self::appendModules($modulesPath, $required_modules, $module_info, $required_dependancy, $module);
362         }
363         foreach ($module_info[$module]['optional_modules'] as $recommended_dependancy) {
364             if (file_exists($modulesPath.$recommended_dependancy.'/composer.json')) {
365                 self::appendModules($modulesPath, $required_modules, $module_info, $recommended_dependancy, $module);
366             }
367         }
368     }

399         if (headers_sent($file, $line)) {

402             if (isset($_SESSION)) { // Is de sessie al gestart?

404             }

406             if (empty($_SERVER['HTTP_USER_AGENT'])) { // Is de browser meegegeven?
407                 // Er is geen user_agent/browser opgegeven, waarschijnlijk dan geen browser
408                 ini_set('url_rewriter.tags', ''); // Geen PHPSESSID in de html code stoppen
409             }
410             session_start();

414     /**
415      * Detecteer alle modules in de modules map.
416      *
417      * @param string $modulesPath
418      * @return array
419      */
420     private static function detectModules($modulesPath) {
421         $modules = array();
422         $Directory = new \DirectoryIterator($modulesPath);
423         foreach ($Directory as $entry) {
424             if ($entry->isDir() && substr($entry->getFilename(), 0, 1) != '.') { // Is het een niet verborgen map
425                 $modules[] = $entry->getFilename();
426             }
427         }
428         return $modules;
