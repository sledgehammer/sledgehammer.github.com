19 
20     private $url;
21     private $permanently;

25         $this->permanently = $permanently;
26     }
27 
28     function getHeaders() {
29         return array(
30             'http' => array(
31                 'Status' => ($this->permanently ? '301 Moved Permanently' : '302 Found'),
32                 'Location' => $this->url
33             )
34         );
35     }
36 
37     function isDocument() {
38         return true;
39     }
40 
41     function render() {
42         // javascript fallback (headers already sent)
43         // Javascript fallback
44         echo '<script type="text/javascript">window.location='.json_encode((string) $this->url).';</script>';
45         echo '<noscript>';
46         // Meta refresh fallback
47         echo '<meta http-equiv="refresh" content="0; url='.htmlentities($this->url, ENT_QUOTES).'">';
48         // Show a link
