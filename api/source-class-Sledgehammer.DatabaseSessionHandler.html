 26:      * Contructor
 27:      * @param array $options  Hiermee kun je de "table" en "dbLink" overschrijven
 28:      */
 29:     function __construct($options = array()) {
 30:         $availableOptions = array('table', 'dbLink');
 31:         foreach ($options as $property => $value) {
 32:             if (in_array($property, $availableOptions)) {
 33:                 $this->$property = $value;
 34:             } else {
 35:                 notice('Invalid option "'.$property.'"', 'Use: '.human_implode(' or ', $availableOptions));
 36:             }

 39:             $this->table = session_name();
 40:         }
 41:     }

 44:      * Set the session handler to this instance.
 45:      */
 46:     function init() {
 47:         session_set_save_handler(array($this, 'noop'), array($this, 'noop'), array($this, 'read'), array($this, 'write'), array($this, 'destroy'), array($this, 'cleanup'));
 48:         register_shutdown_function('session_write_close');
 49:     }

 52:      * Callback for $open & $close.
 53:      * @return boolean

 59:     /**
 60:      * Load session-data based for the given $id.
 61:      * @param string $id session_id
 62:      * @return string|boolean
 63:      */
 64:     function read($id) {
 65:         $db = getDatabase($this->dbLink);
 66:         try {
 67:             $gegevens = $db->fetch_value('SELECT session_data FROM '.$db->quoteIdentifier($this->table).' WHERE id = '.$db->quote($id), true);
 68:         } catch (\Exception $e) {
 69:             report_exception($e);
 70:             $gegevens = false;
 71:         }
 72:         if ($gegevens == false) {
 73:             if ($db->errno == 1146) { // Table doesnt exist?
 74:                 $db->query('CREATE TABLE '.$db->quoteIdentifier($this->table).' (id varchar(32) NOT NULL, last_used INT(10) UNSIGNED, session_data TEXT, PRIMARY KEY (id));'); // Sessie tabel maken
 75:             }
 76:             return '';
 77:         }
 78:         return $gegevens;
 79:     }

 87:      */
