 51     /**
 52      * Connect to a database.
 53      *
 54      * @param string $dsn  The pdo-dsn "mysql:host=localhost" or url: "mysql://root@localhost/my_database?charset=utf-8"
 55      * @param string $username
 56      * @param string $passwd
 57      * @param array $options
 58      */
 59     function __construct($dsn, $username = null, $passwd = null, $options = array()) {
 60         $start = microtime(true);
 61         $isUrlStyle = preg_match('/^[a-z]+:\/\//i', $dsn, $match);
 62         if ($isUrlStyle) { // url syntax?
 63             $url = new Url($dsn);
 64             $config = $url->query;
 65             $config['host'] = $url->host;
 66             if ($url->port !== null) {
 67                 $config['port'] = $url->port;
 68             }
 69             if ($url->path !== null && $url->path != '/') {
 70                 $config['dbname'] = substr($url->path, 1); // strip "/"
 71             }
 72             $driver = strtolower($url->scheme);
 73             $username = $url->user;
 74             $passwd = $url->pass;
 75         } else {
 76             preg_match('/^([a-z]+):/i', $dsn, $match);
 77             $driver = strtolower($match[1]);
 78         }
 79 
 80         if ($driver == 'mysql') {
 81             $this->reportWarnings = true;
 82             if (isset($config['charset'])) {
 83                 $charset = $config['charset'];
 84             } elseif (strpos($dsn, ';charset=')) {
 85                 $parts = explode(';', $dsn);
 86                 foreach ($parts as $i => $part) {
 87                     if (strpos($part, 'charset=') === 0) {
 88                         $charset = substr($part, 8);
 89                         unset($parts[$i]);
 90                     }
 91                 }
 92                 $dsn = implode(';', $parts);
 93             } else {
 94                 // Use Sledgehammer setting (default utf-8)
 95                 switch (strtolower(Framework::$charset)) {
 96 
 97                     case 'utf-8':
 98                         $charset = 'UTF8';
 99                         break;
100 
101                     case 'iso-8859-1':
102                     case 'iso-8859-15':
103                         $charset = 'latin1';
104                         break;
105 
106                     default:
107                         $charset = Framework::$charset;
108                         break;
109                 }
110             }
111             if (version_compare(PHP_VERSION, '5.3.6') == -1) {
112                 if (empty($options[\PDO::MYSQL_ATTR_INIT_COMMAND])) {
113                     $options[\PDO::MYSQL_ATTR_INIT_COMMAND] = 'SET NAMES "'.mysql_escape_string($charset).'"';
114                 }
115             } elseif ($isUrlStyle) {
116                 $config['charset'] = $charset;
117             } else {
118                 $dsn .= ";charset=".$charset;
119             }
120         } else {
121             unset($this->reportWarnings); // Unset the reportWarning property for database drivers that don't support it
122         }
123         if ($isUrlStyle) {
124             $dsn = $driver.':';
125             foreach ($config as $name => $value) {
126                 $dsn .= $name.'='.$value.';';
127             }
128         }
129         // Parse $options
130         $loggerOptions = array(
131             'identifier' => 'Database['.$dsn.']',
132             'limit' => 1000,
133             'renderer' => array($this, 'renderLog'),
134             'start' => 0,
135             'plural' => 'queries',
136             'singular' => 'query',
137             'columns' => array('SQL', 'Duration'),
138         );
139         foreach ($options as $property => $value) {
140             if (substr($property, 0, 3) === 'log') {
141                 $loggerOptions[lcfirst(substr($property, 3))]= $value;
142                 unset($options[$property]);
143             } elseif ($property === 'reportWarnings') {
144                 $this->$property = $value;
145             }
146         }
147         $this->logger = new Logger($loggerOptions);
148         if (empty($options[\PDO::ATTR_DEFAULT_FETCH_MODE])) {
149             $options[\PDO::ATTR_DEFAULT_FETCH_MODE] = \PDO::FETCH_ASSOC;
150         }
151         if (empty($options[\PDO::ATTR_STATEMENT_CLASS])) {
152             $options[\PDO::ATTR_STATEMENT_CLASS] = array('Sledgehammer\Statement');
153         }
154         parent::__construct($dsn, $username, $passwd, $options);
155         $this->logger->append('Database connection to '.$dsn, array('duration' => (microtime(true) - $start)));
156         $this->logger->count--;
157 
158         $this->driver = $this->getAttribute(\PDO::ATTR_DRIVER_NAME);
159 
160         if (isset($this->reportWarnings) && $this->reportWarnings === true) {
161             parent::exec('SET sql_warnings = ON');
162         }
163     }

260     /**
261      * Quotes a string for use in a query.
262      * @link http://php.net/manual/en/pdo.quote.php
263      *
264      * @param string $value  The string to be quoted.
265      * @param int $parameterType [optional] Provides a data type hint for drivers that have alternate quoting styles.
266      * @return string  A quoted string that is safe to pass into an SQL statement.
267      */
268     function quote($value, $parameterType = null) {
269         if ($parameterType === null && $value === null) {
270             return 'NULL';
271         }
272         // No quotes around INT values.
273         if (is_int($value) && ($parameterType === null || $parameterType === \PDO::PARAM_INT)) {
274             return $value;
275         }
276         if ($parameterType === \PDO::PARAM_INT && preg_match('/^[1-9]{1}[0-9]*$/', $value)) {
277             return $value;
278         }
279         // Force notation with a dot. "0.5"
280         if (is_float($value)) {
281             $previousLocale = setlocale(LC_NUMERIC, 'C');
282             $value = (string) $value;
283             setlocale(LC_NUMERIC, $previousLocale); // restore locale
284         }
285         return parent::quote($value, $parameterType);
286     }

370     /**
371      * Fetch a single row
372      *
373      * @param string $statement  The SQL query
374      * @param bool $allow_empty_results  true: Suppress the notice when no record is found.
375      * @return array|false
376      */
377     function fetchRow($statement, $allow_empty_results = false) {
378         $result = $this->query($statement);
379         if ($result == false) {  // Foutieve query
380             return false;
381         } elseif ($result->columnCount() == 0) { // UPDATE, INSERT query
382             warning('Resultset has no columns, expecting 1 or more columns');
383             return false;
384         }
385         $results = $result->fetchAll();
386         $count = count($results);
387         if ($count == 1) {
388             return $results[0];
389         }
390         if (count($results) > 1) {
391             notice('Unexpected '.count($results).' rows, expecting 1 row');
392         } elseif (!$allow_empty_results) {
393             notice('Row not found');
394         }
395         return false;
396     }

399      * Fetch a single value
400      * The resultset may contain only 1 record with only 1 column.
401      *
402      *
403      * @param string $sql  The SQL query
404      * @param bool $allow_empty_results  true: Suppress the notice when no record is found.
405      * @return string|null|false
406      */
407     function fetchValue($sql, $allow_empty_results = false) {
408         $row = $this->fetchRow($sql, $allow_empty_results);
409         if (!$row) {
410             return false;
411         }
412         if (count($row) > 1) {
413             notice('Unexpected number of columns('.count($row).'), expecting 1 column');
414             return false;
415         }
416         return current($row);

419     /**
420      * Een MySQL dump bestand importeren
421      *
422      * @param string $filepath De bestandsnaam inclisief path van het sql bestand
423      * @param string $error_message return de foutmelding
424      * @param bool $strip_php_comments Bij true worden de regels met "/*" beginnen en eindingen met  "* /" genegeerd
425      * @param false|callback $progress_callback Deze callback word na elke voltooide query aangeroepen met het regelnummer als parameter.
426      * @return bool
427      */
428     function import($filepath, &$error_message, $strip_php_comments = false, $progress_callback = false) {
429         if ($progress_callback && !is_callable($progress_callback)) {
430             notice('Invalid $progress_callback', $progress_callback);
431             $progress_callback = false;
432         }
433         $fp = fopen($filepath, 'r');
434         $filepath = (strpos($filepath, PATH) === 0) ? substr($filepath, strlen(PATH)) : $filepath; // Het PATH van de $filepath afhalen zodat eventuele fouten een kort path tonen
435         if (!$fp) { // Kan het bestand niet geopend worden?
436             $error_message = 'File "'.$filepath.'" not found';
437             return false;
438         }
439         $queries = array();
440         $query = '';
441         $concaternated_query = '';
442         $explode_exceptions = array('&euro', '&amp', '&eacute', '&eacute'); // Het einde van een query wordt verkeerd ge-explode bij deze waardes
443         $line_nr = 0;
444         while (($line = fgets($fp)) !== false) {
445             $line_nr++;
446             $line = utf8_encode(rtrim($line));
447             if ($line == '' || preg_match('/^[ ]*--/', $line)) { // Geen lege of MySQL-commentaar regel?
448                 continue;
449             }
450             if ($strip_php_comments && substr($line, 0, 2) == '/*' && substr(rtrim($line), -2) == '*'.'/') { // Geen /* * / commentaar?
451                 continue;
452             }
453             $pieces = explode(';', $line);
454             if (count($pieces) == 1) {
455                 $concaternated_query .= $line."\n";
456                 continue;
457             }
458             foreach ($pieces as $index => $piece) {
459                 $concaternated_query .= $piece;
460                 // De query controleren of deze foutief is afgekapt
461                 foreach ($explode_exceptions as $tag) {
462                     if (substr($concaternated_query, 1 - strlen($tag)) == $tag) {
463                         $concaternated_query .= $piece.';';
464                         continue 2; // volgende piece
465                     }
466                 }
467                 if ($concaternated_query != '') {
468                     if (!$this->query($concaternated_query)) {
469                         $error_message = 'Invalid SQL statement in "'.$filepath.'" on line '.$line_nr;
470                         fclose($fp);
471                         return false;
472                     }
473                     if ($progress_callback) {
474                         call_user_func($progress_callback, $line_nr);
475                     }
476                     $concaternated_query = '';
477                 }
478             }
479         }
480         fclose($fp);
481         $concaternated_query = trim($concaternated_query);
482         if ($concaternated_query != '') {
483             if (!$this->query($concaternated_query)) {
484                 $error_message = 'Invalid SQL statement in "'.$filepath.'" on line '.$line_nr;
485                 return false;
486             }
487         }
488         return true;
489     }

505     /**
506      * De sql query in het debug_blok overzichtelijk weergeven
507      * De keywords van een sql query dik(<b>) maken en de querytijd een kleur geven (rood voor trage queries, orange voor middelmatige en grijs voor snelle queries)
508      *
509      * @param string $sql
510      * @param int $truncated Number of bytes that where truncated
511      * @param float $duration
512      * @param array $backtrace
513      * @return string
514      */
515     function renderLog($entry, $meta) {
516         $sql = htmlspecialchars($entry, ENT_COMPAT, 'ISO-8859-15');
517         static $regex = null;
518         if ($regex === null) {
519             $startKeywords = array('SELECT', 'UPDATE', 'ANALYSE', 'ALTER TABLE', 'REPLACE INTO', 'INSERT INTO', 'DELETE', 'CREATE TABLE', 'CREATE DATABASE', 'DESCRIBE', 'TRUNCATE TABLE', 'TRUNCATE', 'SHOW', 'SET', 'START TRANSACTION', 'ROLLBACK');
520             $inlineKeywords = array('AND', 'AS', 'ASC', 'BETWEEN', 'BY', 'COLLATE', 'COLUMN', 'CURRENT_DATE', 'DESC', 'DISTINCT', 'FROM', 'GROUP', 'HAVING', 'IF', 'IN', 'INNER', 'IS', 'JOIN', 'KEY', 'LEFT', 'LIKE', 'LIMIT', 'OFFSET', 'NOT', 'NULL', 'ON', 'OR', 'ORDER', 'OUTER', 'RIGHT', 'SELECT', 'SET', 'TO', 'UNION', 'VALUES', 'WHERE');
521             $regex = '/^'.implode('\b|^', $startKeywords).'\b|\b'.implode('\b|\b', $inlineKeywords).'\b/';
522         }
523         $sql = preg_replace($regex, '<span class="sql-keyword">\\0</span>', $sql);
524         $sql = preg_replace('/`[^`]+`/', '<span class="sql-identifier">\\0</span>', $sql);
525         if (isset($meta['truncated'])) {
526             $kib = round($meta['truncated'] / 1024);
527             $sql .= '<span class="logentry-alert">...'.$kib.' KiB truncated</span>';
528         }
529         echo '<td class="logentry-code">', $sql, '</td>';
530         $duration = $meta['duration'];
531         if ($duration > 0.2) {
532             $color = 'logentry-alert';
533         } elseif ($duration > 0.05) {
534             $color = 'logentry-warning';
535         } else {
536             $color = 'logentry-debug';
537         }
538         echo '<td class="logentry-number ', $color, '"><b>', format_parsetime($duration), '</b>&nbsp;sec</td>';
539     }

562      * @param Sql|string $statement

564     function checkWarnings($statement) {
565         if (isset($this->reportWarnings) && $this->reportWarnings === true) {
566             $info = array();
567             if ($statement instanceof Sql) {
568                 $info['SQL'] = (string) $statement;
569             }
570             $start = microtime(true);
571             $this->previousInsertId = parent::lastInsertId();
572             $warnings = parent::query('SHOW WARNINGS');
573             if ($warnings === false) {
574                 $this->reportError('SHOW WARNINGS');
575                 return;
576             }
577             $this->logger->totalDuration += (microtime(true) - $start);
578             if ($warnings->rowCount()) {
579                 foreach ($warnings->fetchAll(\PDO::FETCH_ASSOC) as $warning) {
580                     notice('SQL '.strtolower($warning['Level']).' ['.$warning['Code'].'] '.$warning['Message'], $info);
581                 }
582                 // @todo Clear warnings
583                 // PDO/MySQL doesn't clear the warnings before CREATE/DROP DATABASE queries.
584             }
585         }
