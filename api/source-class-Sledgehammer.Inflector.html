 88     /**
 89      * Returns the plural form of the word in the string
 90      *
 91      * @param string $singular
 92      * @return string
 93      */
 94     static function pluralize($singular) {
 95         // save some time in the case that singular and plural are the same
 96         if (in_array(strtolower($singular), self::$uncountable)) {
 97             return $singular;
 98         }
 99         // check for irregular singular forms
100         foreach (self::$irregular as $pattern => $result) {
101             $pattern = '/'.$pattern.'$/i';
102             if (preg_match($pattern, $singular)) {
103                 return preg_replace($pattern, $result, $singular);
104             }
105         }
106 
107         // check for matches using regular expressions
108         foreach (self::$plural as $pattern => $result) {
109             if (preg_match($pattern, $singular)) {
110                 return preg_replace($pattern, $result, $singular);
111             }
112         }
113         return $singular;
114     }

143     /**
144      * Create a model name from a plural table name.
145      *
146      * customers => Customer
147      * cart_items => CartItem
148      *
149      * @param string $table
150      * @param string $prefix  Database/table prefix
151      * @return string
152      */
153     static function modelize($table, $prefix = '') {
154         if ($prefix != '' && substr($table, 0, strlen($prefix)) == $prefix) {
155             $table = substr($table, strlen($prefix)); // Strip prefix
156         }
157         $words = explode('_', $table);
158         foreach ($words as $i => $word) {
159             $words[$i] = ucfirst($word);
160         }
161         $last = count($words) - 1;
162         $words[$last] = Inflector::singularize($words[$last]);
163         return implode('', $words);
164     }
