  7  * Een generieke snelle manier om een csv bestand in te lezen.

 94     /**
 95      * Schrijf de waarden in de $iterator weg naar $filename die in de constuctor is meegegeven.
 96      * D.m.v "php://output" kan het csv bestand direct naar de browser verstuurd worden.
 97      *
 98      * @param string $filename
 99      * @param iterator $iterator Brongegevens voor de csv. Elementen worden bepaald d.m.v. de $keys in de $this->columns array.
100      * @param array|null $columns
101      * @param string|char $delimiter
102      * @param string|char $enclosure
103      * @return void
104      */
105     static function write($filename, $iterator, $columns = null, $delimiter = ';', $enclosure = '"') {
106         $fp = fopen($filename, 'w');
107         if (!$fp) {
108             throw new \Exception('Failed to open "'.$filename.'" for writing');
109         }
110         // De kolomnamen op de eerste csv regel zetten
111         if ($columns === null) {
112             // Detecteer colomnamen (gebruik de array keys)
113             if (is_array($iterator)) {
114                 reset($iterator);
115                 $row = current($iterator);
116             } else {
117                 $iterator->rewind();
118                 $row = $iterator->current();
119             }
120             $columnKeys = array_keys($row);
121             $columns = $columnKeys;
122         } elseif ($columns !== false) {
123             $columnKeys = array_keys($columns);
124         }
125         fputcsv($fp, $columns, $delimiter, $enclosure);
126 
127         foreach ($iterator as $row) {
128             $values = array();
129             foreach ($columnKeys as $key) {
130                 $values[] = $row[$key];
131             }
132             fputcsv($fp, $values, $delimiter, $enclosure);
133         }
134         fclose($fp);
135     }

137     /**
138      * Het csv bestand (opnieuw) openen en de eerste rij inlezen als kolomnamen.
139      */
140     function rewind() {
141         $this->index = null;
142         if ($this->fp) {
143             fclose($this->fp);
144         }
145         $this->fp = fopen($this->filename, 'r');
146         if (!$this->fp) {
147             throw new \Exception('Couldn\'t open file "'.$this->filename.'"');
148         }
149         // Kolommen controleren
150         $keys = fgetcsv($this->fp, 0, $this->delimiter, $this->enclosure);
151         $column_isset = array();
152         foreach ($keys as $column) {
153             if (isset($column_isset[$column])) {
154                 if ($this->columns === null || array_search($column, $this->columns) !== false) {
155                     throw new \Exception('Column "'.$column.'" is ambiguous');
156                 }
157             } else {
158                 $column_isset[$column] = true;
159             }
160         }
161         if ($this->columns === null) {
162             $this->keys = $keys;
163         } else {
164             foreach ($this->columns as $key => $column) {
165                 $index = array_search($column, $keys);
166                 if ($index !== false) {
167                     $this->keys[$index] = $key;
168                 } else {
169                     throw new \Exception('Column: "'.$column.'" not found. Available columns: "'.implode('", "', $keys).'"');
170                     return false;
171                 }
172             }
173         }
174         $this->index = -1;
175         $this->next();
176     }

178     /**
179      * De rij inlezen en naar de volgende regel gaan.
180      * @link http://php.net/manual/en/iterator.next.php
181      * @return void
182      */
183     function next() {
184         $this->values = array();
185         $row = fgetcsv($this->fp, 0, $this->delimiter, $this->enclosure);
186         if ($row) { // Is het einde (eof) nog niet bereikt?
187             $this->index++;
188             foreach ($this->keys as $index => $key) {
189                 if (isset($row[$index])) { // Is er voor deze kolom een waarde?
190                     $this->values[$key] = $row[$index];
191                 } else {
192                     $filename = (strpos($this->filename, PATH) === 0) ? substr($this->filename, strlen(PATH)) : $this->filename; // Waar mogelijk het PATH er van af halen
193                     notice('Row too short, missing column '.$index.'('.$this->columns[$key].') in '.$filename.' on line '.$this->index + 2, $row); // @todo Calculate line offset compared to the index ()
194                 }
195             }
196         } else {
197             $this->index = null;
198         }
199     }
