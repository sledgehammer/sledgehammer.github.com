  6 /**
  7  * Text, a string class for handeling (multibyte) strings with OOP syntax
  8  * Modelled after the C# String class.
  9  * @link http://msdn.microsoft.com/en-us/library/system.string.aspx
 10  *
 11  * @property-read int $length  The number of characters
 12  *
 13  * @package Core
 14  */
 15 class Text extends Object implements \ArrayAccess {

 17     /**
 18      * The string in UTF-8
 19      * @var string
 20      */
 21     private $text;

 23     /**
 24      * Construct the Text object and convert $text to UTF-8.
 25      *
 26      * @param string $text  The text
 27      * @param string|array $charset  string: The charset of $text; array: Autodetect encoding, example: array('ASCII', 'UTF-8', 'ISO-8859-15'); null: defaults to Framework::$charset
 28      */
 29     function __construct($text, $charset = null) {
 30         if ($text instanceof Text) {
 31             $this->text = $text->text;
 32             if ($charset !== null && $charset !== 'UTF-8') {
 33                 notice('Invalid charset given, an Text object will alway be UTF-8 encoded');
 34             }
 35             return;
 36         }
 37         if ($charset === null) {
 38             $charset = Framework::$charset;
 39         } elseif (is_array($charset)) {
 40             $charset = mb_detect_encoding($text, $charset, true);
 41             if ($charset === false) {
 42                 notice('Unable to detect charset');
 43                 $this->text = mb_convert_encoding($text, 'UTF-8');
 44                 return;
 45             }
 46         }
 47         $this->text = mb_convert_encoding($text, 'UTF-8', $charset);
 48     }

 51      * Allow Text objects to be used as php strings.
 52      *
 53      * @return string
 54      */
 55     function __toString() {
 56         return $this->text;

124     /**
125      * Returns a truncated copy of this text.
126      * Only appends the given suffix when the text was trucated.
127      *
128      * @param int $maxLenght
129      * @param string $suffix [optional] Defaults to  the "..." character
130      * @return Text
131      */
132     function truncate($maxLenght, $suffix = null) {
133         if ($this->length < $maxLenght) {
134             return new Text($this->text, 'UTF-8');
135         }
136         if ($suffix === null) {
137             $suffix = html_entity_decode('&hellip;', ENT_NOQUOTES, 'UTF-8');
138         } else {
139             $suffix = new Text($suffix);
140         }
141         $pos = strrpos($this->substring(0, $maxLenght), ' ');
142         return $this->substring(0, $pos).$suffix;
143     }

145     /**
146      * Returns a substring from this text.
147      * Similar to substr()
148      *
149      * @param int $offset
150      * @param int $length  (optional)
151      * @return Text
152      */
153     function substring($offset, $length = null) {
154         if ($length === null) {
155             if (mb_internal_encoding() == 'UTF-8') {
156                 return new Text(mb_substr($this->text, $offset), 'UTF-8');
157             }
158             if ($offset < 0) {
159                 $length = $offset * -1;
160             } else {
161                 $length = $this->length - $offset;
162             }
163         }
164         return new Text(mb_substr($this->text, $offset, $length, 'UTF-8'), 'UTF-8');
165     }

167     /**

172     function reverse() {
173         $result = '';
174         for ($i = $this->length - 1; $i >= 0; $i--) {
175             $result .= mb_substr($this->text, $i, 1, 'UTF-8');
176         }
177         return new Text($result, 'UTF-8');
178     }

192      * Split a string by the $separator
193      * Similar to explode()
194      *
195      * @param string $separator
196      * @param int|null $limit  (optional)
197      * @return array
198      */
199     function split($separator, $limit = null) {
200         $strings = explode($separator, $this->text, $limit);
201         $texts = array();
202         foreach ($strings as $text) {
203             $texts[] = new Text($text, 'UTF-8');
204         }
