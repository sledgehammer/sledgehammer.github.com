  7  * A dialog popup with where selected choice is posted back the server.
  8  * Compatible with Twitter Bootrap css and $().modal()
  9  *
 10  * @package MVC
 11  */
 12 class Modal extends Object implements View, Import {
 13 
 14     private $title;
 15     private $body;
 16     private $choices;
 17     private $identifier = 'answer';
 18     private $method = 'post';
 19     private $close = false;
 20 
 21     /**
 22      *
 23      * @param string $title
 24      * @param string $body html
 25      * @param array $choices
 26      * @param array $options [optional]
 27      */
 28     function __construct($title, $body, $choices = array(), $options = array()) {
 29         $this->title = $title;
 30         $this->body = $body;
 31         $this->choices = $choices;
 32         foreach ($options as $option => $value) {
 33             $this->$option = $value;
 34         }
 35     }

 37     function initial($default) {
 38         $indexed = is_indexed($this->choices);
 39         if ($indexed) {
 40             $key = array_search($default, $this->choices);
 41             if ($key === false) {
 42                 foreach ($this->choices as $index => $choice) {
 43                     if (is_array($choice) && $choice['label'] === $default) {
 44                         $key = $index;
 45                         break;
 46                     }
 47                 }
 48             }
 49         } else {
 50             $key = $default;
 51         }
 52         if ($key !== false) {
 53             if (is_array($this->choices[$key]) === false) {
 54                 $this->choices[$key] = array(
 55                     'label' => $this->choices[$key]
 56                 );
 57             }
 58             $this->choices[$key]['class'] = 'btn btn-primary';
 59         }
 60     }
 61 
 62     function import(&$error, $request = null) {
 63         if ($request === null) {
 64             $request = $_POST;
 65         }
 66         if (extract_element($request, $this->identifier, $answer) == false) {
 67             $error = false;
 68             return null;
 69         }
 70         $indexed = is_indexed($this->choices);
 71         if ($indexed) {
 72             if (in_array($answer, $this->choices)) {
 73                 return $answer;
 74             }
 75             foreach ($this->choices as $choice) {
 76                 if (is_array($choice) && $choice['label'] === $answer) {
 77                     return $answer;
 78                 }
 79             }
 80         } elseif (isset($this->choices[$answer])) {
 81             return $answer;
 82         }
 83         $error = 'Unexpected anwser "'.$answer.'", expecting "'.implode(', ', array_keys($this->choices)).'"';
 84             return null;
 85     }
 86 
 87     function getHeaders() {
 88         return array(
 89             'title' => $this->title
 90         );
 91     }
 92 
 93     function render() {
 94         echo "<div class=\"modal\">\n";
 95         echo "\t<div class=\"modal-header\">";
 96         echo '<h3>';
 97         if ($this->close) {
 98             echo '<button class="close" data-dismiss="modal">&times;</button>';
 99         }
100         echo HTML::escape($this->title), "</h3></div>\n";
101         echo "\t<div class=\"modal-body\">\n\t\t", $this->body, "\n\t</div>\n";
102         if (count($this->choices) !== 0) {
103             echo "\t<form class=\"modal-footer\" action=\"".URL::getCurrentURL()."\" method=\"".$this->method."\">\n";
104             $indexed = is_indexed($this->choices);
105             foreach (array_reverse($this->choices) as $answer => $choice) {
106                 if (is_array($choice) === false) {
107                     $choice = array('label' => $choice);
108                 }
109                 $choice['type'] = 'submit';
110                 $choice['name'] = $this->identifier;
111                 if ($indexed) {
112                     $choice['value'] = $choice['label'];
113                 } else {
114                     $choice['value'] = $answer;
115                 }
116                 $button = new Button($choice);
117                 echo "\t\t", $button, "\n";
118             }
119             echo "\t</form>\n";
120         }
121         echo '</div>';
122     }
