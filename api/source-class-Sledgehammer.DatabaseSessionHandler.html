  5 namespace Sledgehammer;
  6 /**
  7  * Sessie gegevens 'centraal' opslaan in een database
  8  * Hierdoor kan een Loadbalancer in 'round robin' mode draaien.
  9  * @package Core
 10  */
 11 class DatabaseSessionHandler extends Object {
 12 
 13     /**
 14      * De tabel waar de sessie gegevens in worden opgeslagen. (Zal standaard de session_name() als tabelnaam gebruiken)
 15      * @var string $table
 16      */
 17     private $table;
 18 
 19     /**
 20      * The database link where the session is stored.
 21      * @var string $dbLink
 22      */
 23     private $dbLink = 'default';
 24 
 25     /**
 26      * Contructor
 27      * @param array $options  Hiermee kun je de "table" en "dbLink" overschrijven
 28      */
 29     function __construct($options = array()) {
 30         $availableOptions = array('table', 'dbLink');
 31         foreach ($options as $property => $value) {
 32             if (in_array($property, $availableOptions)) {
 33                 $this->$property = $value;
 34             } else {
 35                 notice('Invalid option "'.$property.'"', 'Use: '.human_implode(' or ', $availableOptions));
 36             }
 37         }
 38         if ($this->table === null) {
 39             $this->table = session_name();
 40         }
 41     }
 42 
 43     /**
 44      * Set the session handler to this instance.
 45      */
 46     function init() {
 47         session_set_save_handler(array($this, 'noop'), array($this, 'noop'), array($this, 'read'), array($this, 'write'), array($this, 'destroy'), array($this, 'cleanup'));
 48         register_shutdown_function('session_write_close');
 49     }
 50 
 51     /**
 52      * Callback for $open & $close.
 53      * @return boolean
 54      */
 55     function noop() {
 56         return true;
 57     }
 58 
 59     /**
 60      * Load session-data based for the given $id.
 61      * @param string $id session_id
 62      * @return string|boolean
 63      */
 64     function read($id) {
 65         $db = getDatabase($this->dbLink);
 66         try {
 67             $gegevens = $db->fetch_value('SELECT session_data FROM '.$db->quoteIdentifier($this->table).' WHERE id = '.$db->quote($id), true);
 68         } catch (\Exception $e) {
 69             report_exception($e);
 70             $gegevens = false;
 71         }
 72         if ($gegevens == false) {
 73             if ($db->errno == 1146) { // Table doesnt exist?
 74                 $db->query('CREATE TABLE '.$db->quoteIdentifier($this->table).' (id varchar(32) NOT NULL, last_used INT(10) UNSIGNED, session_data TEXT, PRIMARY KEY (id));'); // Sessie tabel maken
 75             }
 76             return '';
 77         }
 78         return $gegevens;
 79     }
 80 
 81     /**
 82      * Write session-data for the given $id
 83      *
 84      * @param string $id
 85      * @param string $gegevens
 86      * @return boolean
 87      */
 88     function write($id, $gegevens) {
 89         try {
 90             $db = getDatabase($this->dbLink);
 91             $result = $db->query('REPLACE INTO '.$db->quoteIdentifier($this->table).' VALUES ('.$db->quote($id).', "'.time().'", '.$db->quote($gegevens).')');
 92             if ($result) {
 93                 return true;
 94             }
 95         } catch (\Exception $e) {
 96             report_exception($e);
 97         }
 98         return false;
 99     }
100 
101     /**
102      * Destroy the session-data for the given $id
103      *
104      * @param string $id
105      * @return boolean
106      */
107     function destroy($id) {
108         $db = getDatabase($this->dbLink);
109         if ($db->query('DELETE FROM '.$db->quoteIdentifier($this->table).' WHERE id = '.$db->quote($id))) {
110             return true;
111         }
112         return false;
113     }
114 
115     /**
116      * Garbage collection
117      * @param int $maxlifetime
118      */
119     function cleanup($maxlifetime) {
120         $db = getDatabase($this->dbLink);
121         $verouderd = time() - $maxlifetime;
122         if ($db->query('DELETE FROM '.$db->quoteIdentifier($this->table).' WHERE last_used < '.$db->quote($verouderd))) {
123             return true;
124         }
125         return false;
126     }
127 
128 }
