52         $rObject = new \ReflectionObject($this);
53         $methods = array();
54         foreach ($rObject->getMethods() as $rMethod) {
55             if (in_array($rMethod->name, array('__get', '__set', '__call', '__toString'))) {
56                 continue;
57             }
58             if ($rMethod->isPublic()) {
59                 $scope = 'public';
60             } elseif ($rMethod->isProtected()) {
61                 $scope = 'protected';
62             } elseif ($rMethod->isPrivate()) {
63                 $scope = 'private';
64             }
65             $parameters = array();

67                 $param = '$'.$rParam->name;
68                 if ($rParam->isDefaultValueAvailable()) {

70                 }
71                 $parameters[] = $param;
72             }
73             $methods[$scope][] = syntax_highlight($rMethod->name, 'method').'('.implode(', ', $parameters).')';
74         }
75         $methodsText = '';
76         $glue = '<br />&nbsp;&nbsp;';
77         foreach ($methods as $scope => $mds) {
78             $methodsText .= '<b>'.$scope.' methods</b>'.$glue.implode($glue, $mds).'<br /><br />';
79         }
80         throw new InfoException('Method: "'.$method.'" doesn\'t exist in a "'.get_class($this).'" object.', $methodsText);
81     }
