182     /**
183      * Parse the output from the vardump and render the html version.
184      *
185      * @param int $indentationLevel  The level of indentation (for arrays & objects)
186      */
187     private function parseVardump($indentationLevel = 0) {
188         if ($this->vardump[$this->offset] == '&') { // A reference?
189             echo '&amp;';
190             $this->offset++;
191         }
192         // NULL
193         if ($this->part(4) === 'NULL') {
194             $this->renderType('symbol', 'null');
195             $this->offset += 4;
196             return;
197         }
198         // Recursive
199         if ($this->part(11) === '*RECURSION*') {
200             $this->renderType('keyword', '*RECURSION* ');
201             $this->renderType('comment', '// This variable is already shown');
202             $this->offset += 11;
203             return;
204         }
205         // A type. Example "int(4)"
206         $parenthesesOpenPos = $this->position('(');
207         if ($parenthesesOpenPos === false) {
208             throw new \Exception('Unknown datatype "'.$this->part(25).'"');
209         }
210         $newlinePos = $this->position("\n");
211         if (self::$xdebug && $newlinePos !== false && $newlinePos < $parenthesesOpenPos) { // No "(" on the same line?
212             $type = $this->part($newlinePos);
213             $length = -1;
214             $this->renderType('method', $type);
215             echo '(?) ';
216             $this->renderType('keyword', '?RECURSION?');
217             $this->offset += $newlinePos;
218             return;
219         } else {
220             $type = $this->part($parenthesesOpenPos);
221             $this->offset += $parenthesesOpenPos + 1;
222             $parenthesesClosePos = $this->position(')');
223             $length = $this->part($parenthesesClosePos);
224             $this->offset += $parenthesesClosePos + 1;
225 
226             if (self::$xdebug && substr($type, 0, 5) === 'class') {
227                 preg_match('/^class (.+)#/', $type, $matches);
228                 $class = $matches[1];
229                 $type = 'object';
230                 $this->offset += 3; // " {\n"
231             }
232         }
233 
234         switch ($type) {
235             // boolean (true en false)
236             case 'bool':
237                 $this->renderType('symbol', $length);
238                 return;
239 
240             // numbers (int, float)
241             case 'int':
242             case 'float':
243             case 'double':
244                 $this->renderType('number', $length);
245                 return;
246 
247             // text (string)
248             case 'string':
249                 $this->offset += 2; // strip ' "'
250                 $quote = '&#39;'; // single quote (&apos)
251                 if (self::$xdebug) {
252                     $endQuotePos = $this->position("\"\n");
253                     if ($endQuotePos === false) { // vardump only has 1 line.
254                         $endQuotePos = strlen($this->vardump) - $this->offset - 1;
255                     }
256                     if ($endQuotePos != $length) {
257                         $quote = '&quot;'; // double quote, because xdebug renders newlines as "\n"
258                         $length = $endQuotePos;
259                     }
260                 }
261                 echo $quote;
262                 $this->renderType('string', $this->part($length));
263                 echo $quote;
264                 $this->offset += $length + 1; // '"' = 1
265                 return;
266 
267             // Resources (file, gd, curl)
268             case 'resource':
269                 $newlinePos = $this->position("\n"); // @todo check if this is not to greedy
270                 if ($newlinePos === false) { // A dump(resource)
271                     $resourceType = substr($this->vardump, $this->offset);
272                 } else {
273                     $resourceType = $this->part($newlinePos);
274                 }
275                 $this->offset += strlen($resourceType);
276                 $resourceType = preg_replace('/^.*\(|\)$/', '', $resourceType); // strip " type of (" ")"
277                 $this->renderType('resource', 'resource '.$length.'('.$resourceType.')');
278                 return;
279 
280             case 'array':
281                 $this->renderType('method', 'array');
282                 if (self::$xdebug) {
283                     if ($this->part(3, $this->position("{\n") + (($indentationLevel + 1) * 2) + 2) === '...') { // Start the next line with "..."
284                         echo '( &hellip; )';
285                         $this->offset += $this->position("}") + 1; // "}"
286                         return;
287                     }
288                 }
289                 if ($length == 0) {
290                     $this->offset += $this->position("}") + 1; // "}"
291                     echo '()';
292                     return;
293                 }
294                 echo "(<span data-dump=\"container\">\n";
295                 if ($length !== -1) {
296                     $this->offset += $this->position("{\n") + 2;
297                 }
298                 $this->parseArrayContents($length, $indentationLevel + 1);
299                 $this->renderIndent($indentationLevel);
300                 echo "</span>)";
301                 $this->assertIndentation($indentationLevel);
302                 $this->offset += ($indentationLevel * 2) + 1; // "}"
303                 return;
304 
305             case 'object':
306                 if (self::$xdebug === false) {
307                     $class = $length;
308                     $this->offset += $this->position('(') + 1; // strip "#?? ("
309                     $parenthesesClosePos = $this->position(')');
310                     $length = $this->part($parenthesesClosePos);
311                     $this->offset += $parenthesesClosePos + 4; // ") {\n" = 4
312                 }
313                 $this->renderType('class', $class);
314                 if (self::$xdebug) {
315                     if ($this->part(3, (($indentationLevel + 1) * 2)) === '...') { // Start the next line with "..."
316                         echo ' { &hellip; }';
317                         $this->offset += $this->position("}") + 1; // "}"
318                         return;
319                     }
320                 }
321                 echo " {<span data-dump=\"container\">\n";
322                 $this->parseObjectContents($length, $indentationLevel + 1);
323                 $this->renderIndent($indentationLevel);
324                 echo "</span>}";
325                 $this->assertIndentation($indentationLevel);
326                 $this->offset += ($indentationLevel * 2) + 1; // "}"
327                 break;
328 
329             default:
330                 throw new \Exception('Unknown datatype "'.$type.'"');
331         }
332     }

334     /**
335      * Parse the contents of an array.
336      *
337      * @param int $length  The number of items in this array.
338      * @param int $indentationLevel  The number of spaces the elements are indented.
339      */
340     private function parseArrayContents($length, $indentationLevel) {
341         $indent = $indentationLevel * 2; // var_dump uses 2 spaces per indent.
342         for ($i = 0; $i < $length; $i++) {
343             if (self::$xdebug && $this->vardump[$this->offset] === "\n" && $this->part(18, ($indentationLevel * 2) + 1) === '(more elements)...') {
344                 echo "\n";
345                 $this->renderIndent($indentationLevel);
346                 echo "&hellip; ";
347                 $this->renderType('comment', "// more elements\n");
348                 $this->offset += ($indentationLevel * 2) + 20;
349                 return;
350             }
351             $this->assertIndentation($indentationLevel);
352             $this->offset += $indent + 1; // strip spaces and "[" or "'"
353             // extract key
354             if (self::$xdebug) {
355                 $numericKey = ($this->vardump[$this->offset - 1] === "["); // is "'" for strings
356             } else {
357                 $numericKey = $this->vardump[$this->offset] !== '"';
358             }
359             $arrowPos = $this->position("=>\n");
360             if (self::$xdebug) {
361                 $key = $this->part($arrowPos - 2); // strip "] " or "' "
362             } else {
363                 if ($numericKey) {
364                     $key = $this->part($arrowPos - 1); // strip "]"
365                 } else {
366                     $key = $this->part($arrowPos - 3, 1); // strip '"' + '"]'
367                 }
368             }
369             $this->offset += $arrowPos + 3 + $indent; // "=>\n" = 3
370             // render key
371             $this->renderIndent($indentationLevel);
372             if ($numericKey) {
373                 $this->renderType('number', $key);
374             } else {
375                 echo '&#39;';
376                 $this->renderType('string', $key);
377                 echo '&#39;';
378             }
379             $this->renderType('operator', ' => ');
380             if (self::$xdebug && $this->part(3, $indent + 2) === '...') {
381                 echo "&hellip;\n";
382                 $this->offset += $indent + 7; // "  ...\n\n" = 7
383                 continue;
384             }
385             // Value
386             $this->parseVardump($indentationLevel);
387             $this->offset += 1; // "\n"
388             echo ",\n";
389         }
390     }

392     /**
393      * Parse the contents of an object.
394      *
395      * @param int $length  The number of attribute in this object.
396      * @param int $indentationLevel  The number of spaces the elements are indented.
397      */
398     private function parseObjectContents($length, $indentationLevel) {
399         $indent = $indentationLevel * 2; // var_dump uses 2 spaces per indent.
400         for ($i = 0; $i < $length; $i++) {
401             $this->assertIndentation($indentationLevel);
402             $this->offset += $indent; // strip spaces
403             // Extract attribute
404             $arrowPos = $this->position("=>\n");
405             if (self::$xdebug) {
406                 $attribute = $this->part($arrowPos - 1);
407             } else {
408                 $attribute = $this->part($arrowPos - 2, 1); // Strip "[" + "]"
409             }
410             // Render attribute
411             $this->renderIndent($indentationLevel);
412             $this->parseAttribute($attribute);
413             $this->offset += $arrowPos + 3 + $indent; // "=>\n" = 3
414 
415             $this->renderType('operator', ' -> ');
416 
417             // Value
418             if (self::$xdebug && $this->part(3, $indent + 2) == '...') {
419                 echo "\n";
420                 $this->renderIndent($indentationLevel);
421                 echo "&hellip;\n\n";
422                 $this->offset += ($indentationLevel * 2) + 7; // "\n  ...\n" = 7
423                 return;
424             }
425             $this->parseVardump($indentationLevel);
426             $this->offset += 1; // "\n"
427             echo ",\n";
428         }
429     }

431     /**
432      * Check if the leading spaces in current line matches the indent level.
433      *
434      * @param int $level
435      */
436     private function assertIndentation($level) {
437         if (preg_match('/^[\s]+/', $this->part(($level * 2) + 1), $match)) {
438             if (strlen($match[0]) == ($level * 2)) { // correct length?
439                 return; // valid
440             }
441         }
442         if ($level == 0) {
443             return;
444         }
445         throw new InfoException('Invalid indentation at offset '.$this->offset.', Expecting "'.str_repeat(' ', $level * 2).'", got "'.$this->part(($level * 2) + 1).'"', array('Near' => $this->part(100)));
446     }

448     /**
449      * Resolves the variablename and renders the "dump($x) in filename.php on line X" line.
450      */
451     private function renderTrace() {
452         $style = array(
453             'font: 13px/22px \'Helvetica Neue\', Helvetica, sans-serif',
454             'border-radius: 4px 4px 0 0',
455             'background-color: '.self::$colors['current'],
456             'margin: 15px 5px 0 5px',
457             'padding: 3px 15px',
458             'color: '.self::$colors['foreground'],
459             // reset styling
460             'text-shadow: none',
461         );
462         echo "<div style=\"".implode(';', $style)."\">";
463         if (self::$xdebug) {
464             echo '<span style="float:right;color:'.self::$colors['comment'].'" title="Add \'xdebug.overload_var_dump = Off\' to php.ini for complete output.">Limited by xdebug</span>';
465         }
466         $this->renderType('comment', $this->trace['invocation']);
467         echo '(<span style="margin: 0 3px;">';
468 
469         if (substr($this->trace['file'], -14) !== ' eval()\'d code') {
470             $file = file($this->trace['file']);
471             $line = $file[($this->trace['line'] - 1)];
472             $line = preg_replace('/.*(dump|DebugR::dump|debugr)\(/i', '', $line); // Alles voor de dump aanroep weghalen
473             $argument = preg_replace('/\);.*/', '', $line); // Alles na de dump aanroep weghalen
474             $argument = trim($argument);
475             if (preg_match('/^\$[a-z_]+[a-z_0-9]*$/i', $argument)) { // $var?
476                 $this->renderType('variable', $argument);
477             } elseif (preg_match('/^(?P<function>[a-z_]+[a-z_0-9]*)\((?<arguments>[^\)]*)\)$/i', $argument, $matches)) { // function()?
478                 $this->renderType('method', $matches['function']);
479                 echo '(', $this->escape($matches['arguments']), ')';
480             } elseif (preg_match('/^(?P<object>\$[a-z_]{1}[a-z_0-9]*)\-\>(?P<attribute>[a-z_]{1}[a-z_0-9]*)(?P<element>\[.+\]){0,1}$/i', $argument, $matches)) { // $object->attribute or $object->attribute[12]?
481                 $this->renderType('variable', $matches['object']);
482                 $this->renderType('operator', '->');
483                 $this->renderType('attribute', $matches['attribute']);
484                 if (isset($matches['element'])) {
485                     echo '['.$this->escape(substr($matches['element'], 1, -1)).']';
486                 }
487             } elseif (preg_match('/^(?P<object>\$[a-z_]+[a-z_0-9]*)\-\>(?P<method>[a-z_]+[a-z_0-9]*)\((?<arguments>[^\)]*)\)$/i', $argument, $matches)) { // $object->method()?
488                 $this->renderType('variable', $matches['object']);
489                 $this->renderType('operator', '->');
490                 $this->renderType('method', $matches['method']);
491                 echo '(', $this->escape($matches['arguments']), ')';
492             } else {
493                 echo $this->escape($argument);
494             }
495         }
496         echo '</span>)&nbsp;';
497         $this->renderType('comment', ' in ');
498         echo '/', str_replace('\\', '/', str_replace(PATH, '', $this->trace['file']));
499         $this->renderType('comment', ' on line ');
500         echo $this->trace['line'];
501         echo "</div>\n";
502     }

515     /**

517      * @param string $searchString  The
518      * @param int $offset Relative offset
519      * @return int|false
520      */

522         $pos = strpos($this->vardump, $searchString, $this->offset + $offset);

540     /**
541      * Parse an attribute name from an object.
542      *
543      * @param string $attribute Examples: '"log"', '"error_types:private"' or '"error_types":"ErrorHandler":private'
544      */
545     private function parseAttribute($attribute) {
546         if (self::$xdebug) {
547             if (preg_match('/(public|protected|private) \$(.+)$/', $attribute, $matches)) {
548                 $this->renderType('attribute', $matches[2]);
549                 if ($matches[1] !== 'public') {
550                     echo '<span style="font-size:9px">:';
551                     $this->renderType('comment', $matches[1]);
552                     echo '</span>';
553                 }
554                 return;
555             } else {
556                 throw new InfoException('Invalid attribute', array('attribute' => $attribute));
557             }
558         }
559         $parts = explode(':', $attribute);
560         $partsCount = count($parts);
561         switch ($partsCount) {
562 
563             case 1: // Is de scope niet opgegeven?
564                 $this->renderType('attribute', substr($attribute, 1, -1));
565                 break;
566 
567             case 2:
568                 $this->renderType('attribute', substr($parts[0], 1, -1));
569                 echo '<span style="font-size:9px">:';
570                 $this->renderType('comment', $parts[1]);
571                 echo '</span>';
572                 break;
573 
574             case 3: // Sinds 5.3 staat bij er naast :private ook van welke class deze private is. bv: "max_string_length_backtrace":"ErrorHandler":private
575                 $this->renderType('attribute', substr($parts[0], 1, -1));
576                 echo '<span style="font-size:9px" title="'.$this->escape(substr($parts[1], 1, -1)).'">:';
577                 $this->renderType('comment', $parts[2]);
578                 echo '</span>';
579                 break;
580 
581             default:
582                 throw new InfoException('Unexpected number of parts: '.$partsCount, $parts);
583         }
584     }
