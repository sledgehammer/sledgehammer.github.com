 56     /**
 57      * Vraag de headers op en werk de interne headers array bij.
 58      * @return array
 59      */
 60     function getHeaders() {
 61 
 62         $headers = array(
 63             'http' => array(
 64                 'Content-Type' => $this->contentType,
 65             ),
 66             'charset' => Framework::$charset,
 67             'htmlParameters' => array(),
 68             'bodyParameters' => array(),
 69         );
 70 
 71         if (defined('WEBPATH') && WEBPATH != '/' && file_exists(PATH.'application/public/favicon.ico')) {
 72             $headers['link']['favicon'] = array('rel' => 'shortcut icon', 'href' => WEBROOT.'favicon.ico', 'type' => 'image/x-icon');
 73         }
 74         // $headers['http']['Content-Type'] = 'application/xhtml+xml';
 75         if (Framework::$errorHandler->html) {
 76             $headers['css']['debug'] = WEBROOT.'core/css/debug.css';
 77         }
 78         $this->headers = merge_headers($headers, $this->content);
 79         if (empty($this->headers['title'])) {
 80             notice('getHeaders() should contain a "title" element for a HTMLDocument');
 81         }
 82         return $this->headers;

 85     /**
 86      * Het document genereren
 87      *
 88      * @return void
 89      */
 90     function render() {
 91         if ($this->headers == null) {
 92             notice(get_class($this).'->getHeaders() should be executed before '.get_class($this).'->render()');
 93         }
 94         $variables = array(
 95             'charset' => $this->headers['charset'],
 96             'title' => array_value($this->headers, 'title'),
 97             'head' => array(),
 98             'htmlParameters' => implode_xml_parameters($this->headers['htmlParameters']),
 99             'bodyParameters' => implode_xml_parameters($this->headers['bodyParameters']),
100             'body' => $this->content,
101             'showStatusbar' => $this->showStatusbar,
102         );
103 
104         $validHeaders = array('http', 'title', 'charset', 'css', 'meta', 'link', 'javascript', 'htmlParameters', 'bodyParameters');
105         foreach ($this->headers as $key => $value) {
106             if (!in_array($key, $validHeaders)) {
107                 notice('Invalid header: "'.$key.'", expecting "'.human_implode('" or "', $validHeaders, '", "').'"');
108             }
109         }
110 
111         // tags binnen de <head> instellen
112         $head = array(
113             'meta' => array(),
114             'link' => array(),
115         );
116         if (isset($this->headers['meta'])) {
117             $head['meta'] = $this->headers['meta'];
118         }
119         if (isset($this->headers['link'])) {
120             $head['link'] = $this->headers['link'];
121         }
122         if (isset($this->headers['css'])) {
123             foreach ($this->headers['css'] as $url) {
124                 $head['link'][] = array('href' => $url, 'type' => 'text/css', 'rel' => 'stylesheet');
125             }
126         }
127         $eot = ($this->doctype === 'xhtml') ? ' />' : '>'; // End of Tag instellen
128         foreach ($head as $tag => $tags) {
129             foreach ($tags as $parameters) {
130                 $variables['head'][] = '<'.$tag.implode_xml_parameters($parameters).$eot;
131             }
132         }
133         if (isset($this->headers['javascript'])) {
134             foreach ($this->headers['javascript'] as $identifier => $url) {
135                 if (is_int($identifier)) {
136                     $identifier = $parameters['src'];
137                 }
138                 ob_start();
139                 javascript_once($url, $identifier);
140                 $variables['head'][] = ob_get_clean();
141             }
142         }
143         $template = new Template('doctype/'.$this->doctype.'.php', $variables);
144         $template->render();
145     }
