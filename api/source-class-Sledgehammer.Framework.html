 33     /**
 34      * List required sledgehammer modules and sort on depedency.
 35      * (The module without depedency comes first.)
 36      *
 37      * @param string $modulesPath
 38      * @return array
 39      */
 40     static function getModules($modulesPath = null) {
 41         if ($modulesPath === null) {
 42             $modulesPath = MODULES_DIR;
 43             $applicationPath = APPLICATION_DIR;
 44         } else {
 45             $applicationPath = dirname($modulesPath).DIRECTORY_SEPARATOR.'application'.DIRECTORY_SEPARATOR;
 46         }
 47         static $cache = array();
 48         if (isset($cache[$modulesPath])) {
 49             return $cache[$modulesPath];
 50         }
 51         $required_modules = array();
 52         $module_info = array(
 53             'application' => array(
 54                 'name' => 'Application',
 55                 'path' => $applicationPath,
 56                 'required_modules' => self::detectModules($modulesPath),
 57                 'optional_modules' => array(),
 58                 'application' => true
 59             )
 60         );
 61         // Fetch all required_modules
 62         self::appendModules($modulesPath, $required_modules, $module_info, 'application', 'detectModules()');
 63         if (!file_exists($applicationPath)) {
 64             unset($required_modules[array_search('application', $required_modules)]); // De application is zelf niet required
 65         }
 66         // Sort by dependancy
 67         $sorted_modules = array();
 68         $cyclic_dependency_check = count($required_modules) + 1;
 69         while (count($required_modules) > 0) { // Loop until all required_modules are sorted
 70             if ($cyclic_dependency_check == count($required_modules)) { // No modules are sorted in the previous while loop?
 71                 throw new \Exception('Cyclic depedency detected for modules "'.implode('" and "', $required_modules).'"');
 72             }
 73             $cyclic_dependency_check = count($required_modules);
 74             foreach ($required_modules as $index => $module) {
 75                 $all_dependancy_are_met = true;
 76                 foreach ($module_info[$module]['required_modules'] as $required_module) {
 77                     if (!in_array($required_module, $sorted_modules)) { // is the required depencancy not met
 78                         $all_dependancy_are_met = false;
 79                         break;
 80                     }
 81                 }
 82                 if ($all_dependancy_are_met) {
 83                     $sorted_modules[] = $module;
 84                     unset($required_modules[$index]);
 85                 }
 86             }
 87         }
 88         // Merge the sorted modules with the module_info array
 89         $modules = array();
 90         foreach ($sorted_modules as $module) {
 91             $modules[$module] = $module_info[$module];
 92         }
 93         $cache[$modulesPath] = $modules;
 94         return $modules;
 95     }

103     static function initModule($path) {
104         if (in_array(substr($path, -1), array('\\', '/')) == false) { // Is er geen trailing "/" opgegeven in het path?
105             $path .= DIRECTORY_SEPARATOR; // De trailing slash toevoegen
106         }

108             notice('Module path: "'.$path.'" not found');
109         }
110         if (file_exists($path.'functions.php')) {
111             include_once($path.'functions.php');
112         }
113         if (file_exists($path.'init.php')) {
114             include($path.'init.php');
115         }

118     /**
119      * Retreiving module info and find all dependend (required) modules
120      *
121      * @param string $modulesPath  De map waar de "modules" in staan
122      * @param array $required_modules  Dit is een array met reeds toegevoegde modules. Zodat er elke modules maximaal 1x wordt ingeladen.
123      * @param array $module_info  In dit array worden gegevens uit de ini bestanden gezet, zodat deze maar 1x ingeladen worden.
124      * @param string $module  Dit is de naam van de module die toegevoegd zal worden aan de $required_modules (inclusief modules waar deze van afhandelijk is)
125      * @param string $required_by  De {$module} is een afhankelijkheid van de {$required_by} module.
126      */
127     private static function appendModules($modulesPath, &$required_modules, &$module_info, $module, $required_by) {
128         if (in_array($module, $required_modules)) { // Is this module already included
129             return;
130         }
131         $required_modules[] = $module;
132         if (!isset($module_info[$module])) {
133             if ($module === 'application') {
134                 throw new \Exception('Info for the application "module" must be configured');
135             } else {
136                 $module_path = $modulesPath.$module.DIRECTORY_SEPARATOR;
137                 if (file_exists($module_path) == false) {
138                     warning('Module: "'.$module.'" is missing, but is required by "'.$required_by.'"');
139                 } elseif (file_exists($module_path.'module.ini') == false) {
140                     notice('Missing module.ini for module: "'.$module.'" required by "'.$required_by.'"');
141                     $module_info[$module] = array('name' => $module);
142                 } else {
143                     $module_info[$module] = parse_ini_file($module_path.'module.ini');
144                     if ($module_info[$module] === false) {
145                         warning('"'.$module.'/module.ini" is corrupted');
146                         $module_info[$module] = array('name' => $module);
147                     }
148                 }
149                 $module_info[$module]['path'] = $module_path;
150             }
151             if (isset($module_info[$module]['required_modules'])) {
152                 $module_info[$module]['required_modules'] = explode(',', $module_info[$module]['required_modules']);
153                 foreach ($module_info[$module]['required_modules'] as $index => $required_module) {
154                     $module_info[$module]['required_modules'][$index] = trim($required_module);
155                 }
156             } else {
157                 $module_info[$module]['required_modules'] = ($module == 'core') ? array() : array('core');
158             }
159             if (isset($module_info[$module]['optional_modules'])) {
160                 $module_info[$module]['optional_modules'] = explode(',', $module_info[$module]['optional_modules']);
161                 foreach ($module_info[$module]['optional_modules'] as $index => $optional_module) {
162                     $module_info[$module]['optional_modules'][$index] = trim($optional_module);
163                 }
164             } else {
165                 $module_info[$module]['optional_modules'] = array();
166             }
167         }
168         foreach ($module_info[$module]['required_modules'] as $required_dependancy) {
169             self::appendModules($modulesPath, $required_modules, $module_info, $required_dependancy, $module);
170         }
171         foreach ($module_info[$module]['optional_modules'] as $recommended_dependancy) {
172             if (file_exists($modulesPath.$recommended_dependancy.'/module.ini')) {
173                 self::appendModules($modulesPath, $required_modules, $module_info, $recommended_dependancy, $module);
174             }
175         }
176     }

206     static function initSession() { // [void]
207         if (headers_sent($file, $line)) {
208             notice('Session could not be started. Output started in '.$file.' on line '.$line);
209         } else {
210             if (isset($_SESSION)) { // Is de sessie al gestart?
211                 return;
212             }
213             // Voorkom PHPSESSID in de broncode bij zoekmachines
214             if (empty($_SERVER['HTTP_USER_AGENT'])) { // Is de browser meegegeven?
215                 // Er is geen user_agent/browser opgegeven, waarschijnlijk dan geen browser
216                 ini_set('url_rewriter.tags', ''); // Geen PHPSESSID in de html code stoppen
217             }
218             session_start();
219         }

222     /**
223      * Detecteer alle modules in de modules map.
224      *
225      * @param string $modulesPath
226      * @return array
227      */
228     private static function detectModules($modulesPath) {
229         $modules = array();
230         $Directory = new \DirectoryIterator($modulesPath);
231         foreach ($Directory as $entry) {
232             if ($entry->isDir() && substr($entry->getFilename(), 0, 1) != '.') { // Is het een niet verborgen map
233                 $modules[] = $entry->getFilename();
234             }
235         }
236         return $modules;
237     }
