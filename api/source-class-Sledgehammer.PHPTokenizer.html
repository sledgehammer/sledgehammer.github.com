  6 /**
  7  * A tokenizer that gives context to tokens of php internal tokenizer.
  8  * Serves as a helper for the PHPAnalyer class.
  9  *
 10  * Tokens:
 11  *   T_HTML        Inline html output
 12  *   T_PHP         PHP-code that issn't a type or namespace definition
 13  *   T_NAMESPACE   The name of a namespace, A empty string '' indicates a global scope
 14  *   T_USE         A namespace or a classname including a namespace
 15  *   T_USE_ALIAS   The alias of for the T_USE namespace
 16  *   T_INTERFACE   An interface that is defined
 17  *   T_CLASS       A class that is defined
 18  *   T_EXTENDS     The parent class/interface
 19  *   T_IMPLEMENTS  The interface(s) that are implemented
 20  *   T_FUNCTION    A function/method that is defined
 21  *   T_TYPE_HINT   The type that hinted in the function or catch block.
 22  *   T_PARAMETER   A parameter/variable of the defined function
 23  *   T_PARAMETER_VALUE  The default value of the parameter
 24  *
 25  *   T_OBJECT       An class that is used in the code
 26  *   T_CALL         An global function that is called in the code
 27  *   T_METHOD_CALL  An method that is called in the code
 28  *
 29  * @package Core
 30  */

135     /**
136      * Iterator::next()
137      * @return void
138      */
139     function next() {
140         if ($this->valid === 'LAST') {
141             $this->valid = false;
142             $this->current = null;
143             return;
144         }
145         $this->key++;
146         $count = count($this->tokens);
147         $value = '';
148         $line = $this->lineNumber;
149 
150         for (; $this->tokenIndex < $count; $this->tokenIndex++) {
151             $token = $this->tokens[$this->tokenIndex];
152             if ($this->tokenIndex === ($count - 1)) { // laatste token?
153                 $nextToken = false;
154                 $this->valid = 'LAST';
155             } else {
156                 $nextToken = $this->tokens[$this->tokenIndex + 1];
157             }
158             if (is_array($token)) {
159                 $tokenContents = $token[1];
160                 $this->lineNumber = $token[2];
161             } else {
162                 $tokenContents = $token;
163             }
164             $value .= $tokenContents;
165             $method = 'parse_'.$this->state;
166             if (method_exists($this, $method) == false) {
167                 $this->failure('Invalid state');
168                 break;
169             }
170             $result = $this->$method($token, $nextToken);
171             switch ($result['action']) {
172 
173                 case 'CONTINUE': // $token belongs to the current state
174                     break;
175 
176                 case 'CONTINUE_AS':
177                     $this->state = $result['state'];
178                     break;
179 
180                 case 'LAST_TOKEN': // The current $token is the last token of the current state.
181                     $this->state = $result['state'];
182                     $this->current = array($result['token'], $value, $line);
183                     $this->tokenIndex++;
184                     return;
185 
186                 case 'SWITCH': // The current token belongs to another state
187                     if (empty($result['token'])) {
188                         $this->failure('return-token is required for a SWITCH');
189                     }
190                     $this->state = $result['state'];
191                     if ($value != $tokenContents) {
192                         $this->current = array($result['token'], substr($value, 0, 0 - strlen($tokenContents)), $line);
193                         $this->valid = true;
194                         return;
195                     }
196                     break;
197 
198                 case 'RESCAN': // The current token belongs to another state
199                     $this->state = $result['state'];
200                     if ($value == $tokenContents) {
201                         $value = '';
202                     } else {
203                         $value = substr($value, 0, 0 - strlen($tokenContents));
204                     }
205                     $this->tokenIndex--;
206                     break;
207 
208                 case 'SINGLE_TOKEN':
209                     if (isset($result['state'])) { // state is optional
210                         $this->failure('Can\'t change state for a SINGLE_TOKEN');
211                     }
212                     if ($value == $tokenContents) {
213                         $this->current = array($result['token'], $value, $line);
214                         $this->tokenIndex++;
215                         return;
216                     } else {
217                         $this->current = array($result['tokenBefore'], substr($value, 0, 0 - strlen($tokenContents)), $line);
218                         $this->valid = true;
219                         return;
220                     }
221                     break;
222 
223                 case 'KEYWORD': // Switch state and skip whitespace
224                     $this->state = $result['state'];
225                     $this->expectToken($nextToken, T_WHITESPACE);
226                     $value .= $nextToken[1];
227                     if (isset($result['token'])) {
228                         $this->current = array($result['token'], $value, $line);
229                         $this->tokenIndex += 2;
230                         return;
231                     }
232                     $this->tokenIndex++;
233                     break;
234 
235                 case 'OPERATOR': // Switch state and optionally skip whitespace
236                     $this->state = $result['state'];
237                     if ($nextToken[0] == T_WHITESPACE) {
238                         $value .= $nextToken[1];
239                         if (isset($result['token'])) {
240                             $this->current = array($result['token'], $value, $line);
241                             $this->tokenIndex += 2;
242                             return;
243                         }
244                         $this->tokenIndex++;
245                     } elseif (isset($result['token'])) {
246                         $this->current = array($result['token'], $value, $line);
247                         $this->tokenIndex++;
248                         return;
249                     }
250                     break;
251 
252                 case 'EMPTY_TOKEN':
253                     $this->state = $result['state'];
254                     if ($tokenContents != $value) {
255                         $this->failure('Not yet supported');
256                     }
257                     $this->current = array($result['token'], '', $line);
258                     return;
259 
260                 default:
261                     $this->failure('Unknown action: '.$result['action']);
262                     break;
263             }
264         }
265         if ($value === '') {
266             return;
267         }
268         switch ($this->state) {
269 
270             case 'HTML':
271                 break;
272 
273             case 'PHP':
274                 $this->current = array('T_PHP', $value, $line);
275                 break;
276 
277             default:
278                 $this->failure('Unexpected file ending (state: "'.$this->state.'")');
279                 break;
280         }
281     }

283     /**
284      * Collect tokens until a "<?php" or "<?=" token.
285      *
286      * @param array|string $token
287      * @param array|string $nextToken
288      * @return array
289      */
290     private function parse_HTML($token, $nextToken) {
291         if ($nextToken === false) {
292             return array(
293                 'action' => 'LAST_TOKEN',
294                 'token' => 'T_HTML',
295                 'state' => 'EOF'
296             );
297         }
298         if (in_array($nextToken[0], array(T_OPEN_TAG, T_OPEN_TAG_WITH_ECHO))) {
299             return array(
300                 'action' => 'LAST_TOKEN',
301                 'token' => 'T_HTML',
302                 'state' => 'PHP',
303             );
304         }
305         if (in_array($token[0], array(T_OPEN_TAG, T_OPEN_TAG_WITH_ECHO))) { // Dit is geen HTML token
306             return array(
307                 'action' => 'SWITCH',
308                 'token' => 'T_HTML',
309                 'state' => 'PHP',
310             );
311         }
312         return array(
313             'action' => 'CONTINUE'
314         );
315     }

317     /**
318      * Detect tokens that indicate context change.
319      *
320      * @param array|string $token
321      * @param array|string $nextToken
322      * @return array
323      */
324     private function parse_PHP($token, $nextToken) {
325         if ($token[0] == T_CLOSE_TAG) { // end of php section?
326             return array(
327                 'action' => 'LAST_TOKEN',
328                 'token' => 'T_PHP',
329                 'state' => 'HTML',
330             );
331         }
332         if ($token == '{') {
333             return array(
334                 'action' => 'SINGLE_TOKEN',
335                 'token' => 'T_OPEN_BRACKET',
336                 'tokenBefore' => 'T_PHP',
337             );
338         }
339         if ($token == '}') {
340             return array(
341                 'action' => 'SINGLE_TOKEN',
342                 'token' => 'T_CLOSE_BRACKET',
343                 'tokenBefore' => 'T_PHP',
344             );
345         }
346 //      if ($nextToken === false) { // end of file?
347 //          return array('LAST TOKEN?');
348 //      }
349 
350         switch ($token[0]) {
351             case T_NAMESPACE: return array('action' => 'OPERATOR', 'token' => 'T_PHP', 'state' => 'NAMESPACE');
352             case T_USE: return array('action' => 'KEYWORD', 'token' => 'T_PHP', 'state' => 'USE');
353             case T_INTERFACE: return array('action' => 'KEYWORD', 'token' => 'T_PHP', 'state' => 'INTERFACE');
354             case T_CLASS: return array('action' => 'KEYWORD', 'token' => 'T_PHP', 'state' => 'CLASS');
355             case T_EXTENDS: return array('action' => 'KEYWORD', 'token' => 'T_PHP', 'state' => 'EXTENDS');
356             case T_IMPLEMENTS: return array('action' => 'KEYWORD', 'token' => 'T_PHP', 'state' => 'IMPLEMENTS');
357             case T_FUNCTION: return array('action' => 'CONTINUE_AS', 'state' => 'FUNCTION');
358             case T_NEW: return array('action' => 'KEYWORD', 'state' => 'TYPE');
359             case T_INSTANCEOF: return array('action' => 'KEYWORD', 'state' => 'TYPE');
360             case T_CATCH: return array('action' => 'CONTINUE_AS', 'state' => 'PARAMETERS');
361             case T_CURLY_OPEN: return array('action' => 'CONTINUE_AS', 'state' => 'COMPLEX_VARIABLE');
362             case T_DOLLAR_OPEN_CURLY_BRACES: return array('action' => 'CONTINUE_AS', 'state' => 'COMPLEX_VARIABLE');
363             case T_STRING:
364                 if ($nextToken == '(') {
365                     $previousToken = $this->tokens[$this->tokenIndex - 1];
366                     $type = ($previousToken[0] == T_OBJECT_OPERATOR) ? 'T_METHOD_CALL' : 'T_CALL';
367                     return array('action' => 'SINGLE_TOKEN', 'token' => $type, 'tokenBefore' => 'T_PHP');
368                 }
369                 break;
370         }
371         return array('action' => 'CONTINUE');
372     }

399      * @param array|string $token

401      * @return array

403     private function parse_USE($token, $nextToken) {

421         $this->expectTokens($token, array(T_STRING, T_NS_SEPARATOR));

567     /**
568      * Collect parameters and default values of a function.
569      *
570      * @param array|string $token
571      * @param array $nextToken
572      * @return array
573      */
574     private function parse_PARAMETERS($token, $nextToken) {
575         if ($token == ')') {
576             return array('action' => 'CONTINUE_AS', 'state' => 'PHP');
577         }
578         if ($token == '=') { // A default value?
579             return array('action' => 'OPERATOR', 'state' => 'PARAMETER_VALUE', 'token' => 'T_PHP');
580         }
581         if (in_array($nextToken[0], array(T_STRING, T_NS_SEPARATOR, T_ARRAY))) { // Type hint?
582             return array('action' => 'LAST_TOKEN', 'token' => 'T_PHP', 'state' => 'PARAMETER_TYPE_HINT');
583         }
584         if ($token[0] == T_VARIABLE) {
585             return array('action' => 'SINGLE_TOKEN', 'token' => 'T_PARAMETER', 'tokenBefore' => 'T_PHP');
586         }
587         $this->expectTokens($token, array(T_WHITESPACE, T_COMMENT, ',', '(', '&'));
588         return array('action' => 'CONTINUE');
589     }

606     /**
607      * Collect the default value of a function parameter.
608      *
609      * @param array|string $token
610      * @param array $nextToken
611      * @return array
612      */
613     private function parse_PARAMETER_VALUE($token, $nextToken) {
614         $classConstantTokens = array(T_DOUBLE_COLON, T_NS_SEPARATOR);
615         if (in_array($token[0], $classConstantTokens) || in_array($nextToken[0], $classConstantTokens)) { // A constant value from inside an class?
616             return array('action' => 'CONTINUE');
617         }
618         $valueTokens = array(T_STRING, T_LNUMBER, T_CONSTANT_ENCAPSED_STRING);
619         if (in_array($token[0], $valueTokens)) {
620             return array('action' => 'LAST_TOKEN', 'token' => 'T_PARAMETER_VALUE', 'state' => 'PARAMETERS');
621         }
622         if ($token[0] == T_ARRAY) { // default parameter is a array literal?
623             $this->arrayDepth = 0;
