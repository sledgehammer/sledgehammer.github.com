  6: /**
  7:  * Een tokenizer voor htmlcode.
  8:  * Met name geschikt voor html met fouten.
  9:  * Met de uitvoer kun je de exacte (foute) html weer reconstrueren.
 10:  *
 11:  * Vaak is een oplossing mogelijk met DOMDocument of SimpleXML.
 12:  * Gebruik in die gevallen niet deze tokenizer.
 13:  *
 14:  * Gebruik deze tokenizer als je een tag/attribuut wilt vervangen en de overige (mogelijk foutieve) html intact wilt laten.
 15:  * Daarnaast is het triviaal om een syntax highlighter te maken met deze tokenizer.
 16:  *
 17:  * Tokens
 18:  * -------------------
 19:  *
 20:  * <T_TAG T_ATTRIBUTE=T_VALUE>T_TEXT</T_CLOSE_TAG>
 21:  * ^     ^ T_WHITESPACE      ^
 22:  * ^ T_OPEN                  ^ T_CLOSE
 23:  *
 24:  *
 25:  * T_TAG        Het type element 'div', 'a', 'br', enz. Kan een <a> of <br /> zijn.
 26:  * T_WHITESPACE Een spatie of newline binnen een tag.
 27:  * T_ATTRIBUTE  De naam van het attribuut bv 'href'
 28:  * T_VALUE      De waarde van een attribuut
 29:  * T_TEXT      De tekst voor of na een T_TAG of T_CLOSE_TAG
 30:  * T_CLOSE_TAG  Het type sluit tag bevat 'a' van een '</a>'
 31:  * T_LT         Een '<' die niet voor opmaak gebruikt wordt
 32:  * T_DELIMITER  De " of ' die om de T_VALUE staan
 33:  * T_COMMENT    De inhoud van een comment blok
 34:  * T_SCRIPT     De inhoud van de een <script> blok
 35:  *
 36:  * T_CDATA
 37:  * T_DTD_ENTITY
 38:  * T_INVALID    Een character dat niet klopt. Deze zal de parsen moeten/kunnen negeren.
 39:  *
 40:  * @todo T_DTD_ATTRIBUTES opslitsen in meerdere T_DTD_ATTRIBUTE tokens
 41:  *
 42:  * Niet alle tokens hebben een type. Deze tokens bestaan dan uit de data-string i.p.v. een array(token-type, data-string)
 43:  *
 44:  * @link http://en.wikipedia.org/wiki/HTML_element#Syntax
 45:  * @package Core
 46:  */

127:         if ($this->iteratorKey === 0) {

145:      * @return array Token
146:      */
147:     function parseToken() {
148:         try {
149:             $startPos = $this->position;
150:             switch ($this->state) {
151: 
152:                 case 'CONTENT':
153:                     $this->parseContent();
154:                     break;
155: 
156:                 case 'TAG':
157:                     $this->parseTag();
158:                     break;
159: 
160:                 case 'TAG_BODY':
161:                     $this->parseTagBody();
162:                     break;
163: 
164:                 case 'VALUE':
165:                     $this->parseValue();
166:                     break;
167: 
168:                 case 'COMMENT':
169:                     $this->parseHtmlComment();
170:                     break;
171: 
172:                 case 'DTD_ELEMENTS':
173:                     $this->parseDTDElements();
174:                     break;
175: 
176:                 case 'SCRIPT':
177:                     $this->parseScript();
178:                     break;
179: 
180:                 default:
181:                     throw new \Exception('Unknown state: "'.$this->state.'"');
182:             }
183:         } catch (\Exception $exception) {
184:             if ($exception->getMessage() == 'HTML_TOKENIZER_EOF') {
185:                 $this->warning('Unexpected end of stream, state: '.$this->state);
186:             } else {
187:                 throw $exception;
188:             }
189:         }
190:         if ($startPos === $this->position) {
191:             throw new \Exception('No new tokens. State: '.$this->state.', position: '.$startPos);
192:         }
193:         return array_shift($this->tokenQueue);
194:     }

196:     /**
197:      * Verzameld alle T_TEXT tokens
198:      *
199:      */
200:     function parseContent() {
201:         // @todo T_GT token '>' detectie .
202:         $ltPos = $this->strpos('<');
203:         if ($ltPos === false) { // Geen tags meer gevonden?
204:             $this->buildLastToken('T_TEXT');
205:             return;
206:         }
207:         $this->state = 'TAG';
208:         if ($ltPos !== $this->position) { // Begint de tag niet direct?
209:             $this->buildToken($ltPos - 1, 'T_TEXT');
210:             return;
211:         }
212:         $this->parseTag();
213:     }

215:     function parseTag() {
216:         if ($this->getChar() !== '<') {
217:             throw new \Exception('parseTag() saninty check failed');
218:         }
219:         $nextChar = $this->substr(1, 1);
220:         if (preg_match('/[>1-9'.$this->wsPattern.']/', $nextChar)) { // Is het geen begin van een tag?
221:             $this->warning('A "T_TEXT" block contains a "<" character');
222:             $this->buildToken($this->position, 'T_LT');
223:             $this->state = 'CONTENT';
224:             return;
225:         }
226:         if ($this->substr(0, 4) == '<!--') {
227:             $this->state = 'COMMENT';
228:             $this->buildToken($this->position + 3);
229:             return;
230:         }
231:         if ($nextChar == '?') {
232:             $this->parseParserTag();
233:             return;
234:         }
235:         if ($nextChar == '!') {
236:             $this->parseDTDTag();
237:             return;
238:         }
239:         if ($nextChar == '/') {
240:             $this->tagType = 'T_CLOSE_TAG';
241:             $this->buildToken($this->position + 1, 'T_OPEN');
242:         } else {
243:             $this->tagType = 'T_TAG';
244:             $this->buildToken($this->position, 'T_OPEN');
245:         }
246:         $pos = $this->firstOccurrance(array_merge(array('>'), $this->wsArray), $match);
247:         if ($pos === false) {
248:             $this->warning('Unclosed '.$this->tagType);
249:             $this->buildLastToken($this->tagType);
250:             return;
251:         }
252:         if ($pos == $this->position) { // Wordt er tag vermeld aan het begin van de tag?
253:             $this->warning('No tag defined');
254:         } else {
255:             $this->currentTag = ($this->tagType == 'T_TAG') ? $this->substr(0, $pos - $this->position) : null;
256:             $this->buildToken($pos - 1, $this->tagType); // Tag is bekend
257:         }
258:         $this->state = 'TAG_BODY'; // We gaan over tot het parsen van de attributen
259:     }

261:     /**
262:      *
263:      */
264:     function parseTagBody() {
265:         $this->collectWhitespace();
266:         // Detect ending
267:         $pos = $this->firstOccurrance(array('>', '/>'), $match);
268:         if ($pos === $this->position) { // Einde tagBody
269:             $this->buildToken($this->position + strlen($match) - 1, 'T_CLOSE');
270:             $this->state = 'CONTENT';
271:             if ($this->currentTag == 'script') {
272:                 if ($match == '>') {
273:                     $this->state = 'SCRIPT';
274:                 } else {
275:                     $this->warning('Self-closing "<script />" tags aren\'t supported by all browsers');
276:                 }
277:             }
278:             return;
279:         }
280: 
281:         // Attribute naam
282:         $type = ($this->tagType == 'T_TAG') ? 'T_ATTRIBUTE' : 'T_INVALID';
283:         $char = $this->getChar();
284:         if (in_array($char, array('/', '!', '?'))) {
285:             $this->warning('Invalid character: "'.$char.'"');
286:             $this->buildToken($this->position, 'T_INVALID');
287:             return;
288:         }
289:         $pos = $this->firstOccurrance(array_merge($this->wsArray, array('=', '>')), $match);
290:         if ($pos === false) {
291:             $this->warning('Unclosed '.$this->tagType);
292:             $this->buildLastToken($type);
293:             return;
294:         }
295:         if ($this->tagType == 'T_CLOSE_TAG') {
296:             $this->warning('Attributes are not allowed in close tags');
297:         }
298:         $this->buildToken($pos - 1, $type);
299:         $this->collectWhitespace('T_INVALID');
300:         if ($this->getChar() !== '=') { // Heeft dit attribuut geen waarde?
301:             return;
302:         }
303:         $this->buildToken($this->position); // assignment operator
304:         $this->collectWhitespace();
305:         $this->state = 'VALUE';
306:     }

308:     function parseValue() {
309:         $invalid = ($this->tagType === 'T_CLOSE_TAG'); // T_CLOSE_TAG mogen geen attributen bevatten.
310:         $t_value = ($invalid ? 'T_INVALID' : 'T_VALUE');
311:         $t_delimiter = ($invalid ? 'T_INVALID' : 'T_DELIMITER');
312: 
313:         $delimiter = $this->getChar();
314:         if (in_array($delimiter, array('"', "'"))) { // Heeft de waarde delimiters?
315:             $this->buildToken($this->position, $t_delimiter);
316:             $pos = $this->strpos($delimiter);
317:             if ($pos === false) {
318:                 $this->warning('Undelimited VALUE');
319:                 $this->buildLastToken($t_value);
320:                 return;
321:             }
322:             if ($pos !== $this->position) { // *Geen* Lege(alt="") waarde?
323:                 $this->buildToken($pos - 1, $t_value);
324:             }
325:             $this->buildToken($this->position, $t_delimiter);
326:             $this->state = 'TAG_BODY';
327:             return;
328:         }
329:         $pos = $this->firstOccurrance(array_merge($this->wsArray, array('>')));
330:         if ($pos === false) {
331:             $this->warning('Undelimited VALUE');
332:             $this->buildLastToken($t_value);
333:             return;
334:         }
335:         $this->buildToken($pos - 1, $t_value);
336:         $this->state = 'TAG_BODY';
337:     }

339:     function parseHtmlComment() {
340:         $pos = $this->strpos('-->');
341:         if ($pos === false) {
342:             $this->warning('Unterminated T_COMMENT. Use "-->" ');
343:             $this->buildLastToken('T_COMMENT');
344:             return;
345:         }
346:         if ($this->position === $pos) {
347:             $this->warning('Empty T_COMMENT block');
348:         } else {
349:             $this->buildToken($pos - 1, 'T_COMMENT');
350:         }
351:         $this->buildToken($this->position + 2);
352:         $this->state = 'CONTENT';
353:     }

386:             }

389:             } else {

398:         if ($pos === false) {

411:     function parseDTDAttributes($state) {
412:         $pos = $this->firstOccurrance(array('[', '>'), $match);
413:         if ($pos === false) {
414:             $this->warning('Inline DTD tag not terminated');
415:             $this->buildLastToken('T_DTD_ATTRIBUTES');
416:             return;
417:         }
418:         if ($pos === $this->position) {
419:             if ($state == 'OPEN') {
420:                 // $this->warning(' No DTD attributes'); // Niet verplicht?
421:             }
422:         } else {
423:             $this->buildToken($pos - 1, 'T_DTD_ATTRIBUTES');
424:         }
425:         $this->buildToken($this->position);
426:         if ($match == '[') { // DTD met elementen
427:             $this->state = 'DTD_ELEMENTS';
428:             $this->dtdLevel++;
429:         } else { // Einde DTD tag
430:             if ($this->dtdLevel == 0) {
431:                 $this->state = 'CONTENT';
432:             }
433:         }
434:     }

436:     function parseDTDElements() {
437:         $this->collectWhitespace();
438:         $pos = $this->firstOccurrance(array('<!', ']', '>'), $match);
439:         if ($pos === false) {
440:             $this->warning('Invalid DTD elements');
441:             $this->buildLastToken('T_DTD_ATTRIBUTES');
442:             return;
443:         }
444:         if ($match == '>') {
445:             $this->warning('Invalid DTD elements');
446:             $this->dtdLevel = 0;
447:             if ($this->position != $pos) {
448:                 $this->buildToken($pos - 1, 'T_INVALID');
449:             }
450:             $this->buildToken($this->position);
451:             $this->state = 'CONTENT'; // @todo verify of het dan weer om content gaat
452:             return;
453:         }
454:         if ($match == ']') { // Einde collectie?
455:             $this->dtdLevel--;
456:             $this->buildToken($this->position);
457:             $this->parseDTDAttributes('CLOSE');
458:             return;
459:         }
460:         if ($this->position != $pos) {
461:             $this->buildToken($pos - 1, 'T_INVALID');
462:         }
463:         $this->parseDTDTag();
464:     }

466:     function parseScript() {
467:         $pos = $this->stripos('</script');
468:         if ($pos === false) {
469:             $this->warning('Unterminated <script> tag');
470:             $this->buildLastToken('T_SCRIPT');
471:             return;
472:         }
473:         $this->state = 'TAG';
474:         if ($this->position == $pos) { // No script body?
475:             $this->parseTag();
476:             return;
477:         }
478:         $this->buildToken($pos - 1, 'T_SCRIPT');
479:     }

481:     function collectWhitespace($tokenType = 'T_WHITESPACE') {
482:         $whitespace = false;
483:         for ($i = $this->position; $i < $this->htmlLength; $i++) {
484:             if (in_array($this->html{$i}, $this->wsArray)) {
485:                 $whitespace = true;
486:             } else {
487:                 break;
488:             }
489:         }
490:         if ($whitespace) {
491:             $this->buildToken($i - 1, $tokenType);
492:             return;
493:         }
494:     }

496:     /**
497:      * Een token instellen o.b.v. van de eind positie van de token.
498:      * Zal de $this->position bijwerken
499:      */
500:     function buildToken($endPosition, $tokenType = null) {
501:         $length = $endPosition + 1 - $this->position;
502:         if ($length <= 0) {
503:             if ($this->position >= $this->htmlLength) {
504:                 throw new \Exception('EOF reached, parsing '.$tokenType.' token failed');
505:             }
506:             throw new \Exception('Invalid lenght for token: '.$tokenType.', start: '.$this->position.' state: '.$this->state);
507:         }
508:         $value = substr($this->html, $this->position, $length);
509:         $this->position = $endPosition + 1;
510:         if ($tokenType === null) {
511:             $this->tokenQueue[] = $value;
512:         } else {
513:             $this->tokenQueue[] = array($tokenType, $value);
514:         }
515:     }

517:     function buildLastToken($type) {
518:         if ($this->htmlLength !== $this->position) {
519:             $this->buildToken($this->htmlLength - 1, $type);
520:         }
521:     }

564:     /**
565:      * @return int de positie van de needle die het eerste voorkomt in de string
566:      */
567:     function firstOccurrance($needles, &$match = null) {
568:         $pos = false;
569:         foreach ($needles as $needle) {
570:             $needlePos = $this->strpos($needle);
571:             if ($needlePos === false) {
572:                 continue;
573:             }
574:             if ($pos === false) {
575:                 $pos = $needlePos;
576:                 $match = $needle;
577:             } elseif ($pos > $needlePos) {
578:                 $pos = $needlePos;
579:                 $match = $needle;
580:             }
