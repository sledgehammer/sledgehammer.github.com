 43     function where($conditions) {
 44         if ($this->isConverted) {
 45             return parent::where($conditions);
 46         }
 47         if ($this->data instanceof Collection && is_array($conditions)) {
 48             $convertedConditions = array();
 49             foreach ($conditions as $field => $value) {
 50                 if (isset($this->mapping[$field])) {
 51                     $convertedConditions[$this->mapping[$field]] = $value;
 52                     unset($conditions[$field]);
 53                 }
 54             }
 55             if (count($convertedConditions) !== 0) { // There are conditions the low-level collection can handle?
 56                 $collection = new RepositoryCollection($this->data->where($convertedConditions), $this->model, $this->repository, $this->mapping);
 57                 if (count($conditions) == 0) {
 58                     return $collection;
 59                 }
 60                 return $collection->where($conditions); // Apply the remaining conditions
 61             } elseif (count($conditions) === 0) { // An empty array was given as $conditions?
 62                 return new RepositoryCollection(clone $this->data, $this->model, $this->repository, $this->mapping);
 63             }
 64         }
 65         return parent::where($conditions);
 66     }
