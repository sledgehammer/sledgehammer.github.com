  5 namespace Sledgehammer;
  6 /**
  7  * Observable, an Event/Listener
  8  *
  9  * Example subclass:
 10  * class Button extends Observable {
 11  *
 12  *   protected $events = array(
 13  *     'click' => array()
 14  *   );
 15  *   // optional method that gets executed when the 'click' event is fired.
 16  *   protected function onClick($sender) {}
 17  * }
 18  *
 19  * $button = new Button();
 20  * $button->on('click', function ($sender) {
 21  *   echo 'clicked';
 22  * });
 23  *
 24  * // Or property style event-binding: (Only 1 listeners per event)
 25  *
 26  * $button->onClick = function () {
 27  *   echo 'clicked';
 28  * };
 29  *
 30  * // reset all listeners.
 31  * $this->onClick = null;
 32  *
 33  * @package Core
 34  */
 35 abstract class Observable extends Object {
 36 
 37     /**
 38      * Storage array for the properties with KVO (Key Value Observer) listeners
 39      * @var array
 40      */
 41     private $__kvo = array();
 42 
 43     /**
 44      * The events/listeners. array($event1 => array($listener1, ...), ...)
 45      * @abstract @var array
 46      */
 47     protected $events = array();
 48 
 49     /**
 50      * Trigger an event.
 51      *
 52      * @param string $event
 53      * @param stdClass $sender
 54      * @param mixed $args (optional)
 55      */
 56     function trigger($event, $sender, $args = null) {
 57         if (isset($this->events[$event])) {
 58             $method = 'on'.ucfirst($event);
 59             $arguments = func_get_args();
 60             array_shift($arguments);
 61             if (method_exists($this, $method)) {
 62                 call_user_func_array(array($this, $method), $arguments);
 63             }
 64             foreach ($this->events[$event] as $callback) {
 65                 call_user_func_array($callback, $arguments);
 66             }
 67         } else {
 68             notice('Event: "'.$event.'" not registered', 'Available events: '.quoted_human_implode(', ', array_keys($this->events)));
 69         }
 70     }
 71 
 72     /**
 73      * Add a callback for an event
 74      *
 75      * @param string $event
 76      * @param Closure|string|array $callback
 77      * @return string identifier
 78      */
 79     function on($event, $callback) {
 80         if (is_callable($callback) == false) {
 81             throw new InfoException('Parameter $callback issn\'t callable', $callback);
 82         }
 83         if ($this->hasEvent($event) === false) {
 84             $availableEvents = array_keys($this->events);
 85             $availableEvents[] = 'change';
 86             foreach (array_keys(get_public_vars($this)) as $property) {
 87                 $availableEvents[] = 'change:'.$property;
 88             }
 89             throw new InfoException('Event: "'.$event.'" not registered', 'Available events: '.quoted_human_implode(', ', array_keys($this->events)));
 90         }
 91         $this->events[$event][] = $callback;
 92         end($this->events[$event]);
 93         $identifier = key($this->events[$event]);
 94         if ($event === 'change') {
 95             $properties = array_merge(array_keys(get_public_vars($this)), array_keys($this->__kvo));
 96             $self = $this;
 97             foreach ($properties as $property) {
 98                 $this->on('change:'.$property, function ($sender, $new, $old) use ($self, $property) {
 99                         $self->trigger('change', $sender, $property, $new, $old);
100                     });
101             }
102         }
103         if (preg_match('/^change:([a-z0-9]+)$/i', $event, $matches)) {
104             $property = $matches[1];
105             if (array_key_exists($matches[1], $this->__kvo) === false) {
106                 $this->__kvo[$property] = $this->$property;
107                 unset($this->$property);
108             }
109         }
110         return $identifier;
111     }
112 
113     /**
114      * Check if the observable has registered the given event.
115      *
116      * @param string $event
117      * @return bool
118      */
119     function hasEvent($event) {
120         $found = array_key_exists($event, $this->events);
121         if ($found) {
122             return true;
123         }
124         if ($event === 'change') {
125             return ((count(get_public_vars($this)) + count($this->__kvo)) !== 0); // A class without public properties doesn't have a change event.
126         }
127         if (preg_match('/^change:([a-z0-9]+)$/i', $event, $matches)) {
128             if (array_key_exists($matches[1], $this->__kvo)) {
129                 return true;
130             }
131             $reflection = new \ReflectionObject($this);
132             $properties = $reflection->getProperties(\ReflectionProperty::IS_PUBLIC);
133             foreach ($properties as $property) {
134                 if ($property->name === $matches[1]) { // a public property exists?
135                     return true;
136                 }
137             }
138         }
139         return false;
140     }
141 
142     /**
143      * Remove a callback from an event
144      *
145      * @param string $event
146      * @param string $identifier
147      */
148     function off($event, $identifier) {
149         if ($this->hasEvent($event) === false) {
150             warning('Event: "'.$event.'" not registered', 'Available events: '.quoted_human_implode(', ', array_keys($this->events)));
151             return false;
152         }
153         if (empty($this->events[$event][$identifier])) {
154             warning('Identifier: "'.$identifier.'" not found in listeners for event: "'.$event.'"', 'Available identifiers: '.quoted_human_implode(', ', array_keys($this->events[$event])));
155             return false;
156         }
157         unset($this->events[$event][$identifier]);
158         return true;
159     }
160 
161     /**
162      * Getting a non-existing or KVO property
163      *
164      * @param string $property
165      * @return mixed
166      */
167     function __get($property) {
168         if (array_key_exists($property, $this->__kvo)) {
169             return $this->__kvo[$property];
170         }
171         return parent::__get($property);
172     }
173 
174     /**
175      * Setting a non-existing or KVO property
176      *
177      * @param string $property
178      * @param mixed $value
179      */
180     function __set($property, $value) {
181         if (preg_match('/^on[A-Z]/', $property)) { // An event? like "onClick"
182             $event = lcfirst(substr($property, 2));
183             if ($this->hasEvent($event)) {
184                 $this->events[$event] = array(); // Reset listeners.
185                 if ($value !== null) {
186                     $this->on($event, $value);
187                 }
188                 return;
189             }
190         }
191         if (array_key_exists($property, $this->__kvo)) { // A property with a change listener?
192             $this->__kvo[$property];
193             $this->trigger('change:'.$property, $this, $value, $this->__kvo[$property]);
194             $this->__kvo[$property] = $value;
195             return;
196         }
197         parent::__set($property, $value);
198     }

200 }
