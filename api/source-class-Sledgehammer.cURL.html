200      * Wait for all requests to complete
201      *
202      * @throws \Exception

204     static function synchronize() {
205         if (self::$pool === null) {
206             return 0;
207         }
208         do {
209             // Wait for (incomming) data
210             if (curl_multi_select(self::$pool, 0.2) === -1) {
211                 throw new \Exception('Failed to detect changes in the cURL multi handle');
212             }
213             $wait = false;
214             foreach (cURL::$requests as $curl) {
215                 if ($curl->isComplete() == false) {
216                     $wait = true;
217                 }
218             }
219         } while ($wait);
220         if (self::$keepalive === false && self::$tranferCount === 0) {
221             curl_multi_close(self::$pool);
222             self::$pool = null;
223         }
224     }

287     /**
288      * Check if this request is complete
289      *
290      * @throws \Exception
291      * @return boolean
292      */
293     function isComplete() {
294         if ($this->state !== 'RUNNING') {
295             return true;
296         }
297         // Add messages from the curl_multi handle to the $messages array.
298         do {
299             $error = curl_multi_exec(self::$pool, $active);
300         } while ($error === CURLM_CALL_MULTI_PERFORM);
301         if ($error !== CURLM_OK) {
302             throw new \Exception('['.self::multiErrorName($error).'Failed to execute cURL multi handle');
303         }
304         $queued = 0;
305         do {
306             $message = curl_multi_info_read(self::$pool, $queued);
307             if ($message !== false) {
308                 // Scan the (global) curl pool for curl handle specified in the handle
309                 foreach (cURL::$requests as $index => $curl) {
310                     if ($curl->handle === $message['handle']) {
311                         unset(cURL::$requests[$index]); // Cleanup global curl pool
312                         $curl->state = 'COMPLETED';
313                         $error = $message['result'];
314                         if ($error !== CURLE_OK) {
315                             $curl->state = 'ERROR';
316                             throw new InfoException('['.self::errorName($error).'] '.curl_error($curl->handle), $curl->options);
317                         }
318                         $tranferCount = self::$tranferCount;
319                         $curl->trigger('load', $curl);
320                         if ($curl === $this) {
321                             return true;
322                         }
323                         break;
324                     }
325                 }
326             }
327         } while ($queued > 0);
328         return false;
329     }

362     /**
363      * Start the request
364      * 1. Set the CURLOPT_* options
365      * 2. Add the request to the pool
366      * 3. Starts the request.
367      *
368      * @param array $options CURLOPT_* options
369      * @throws \Exception
370      */
371     private function start($options) {
372         $this->state = 'ERROR';
373         cURL::$requests[] = $this; // Watch changes
374         // Setting options
375         foreach ($options as $option => $value) {
376             if (curl_setopt($this->handle, $option, $value) === false) {
377                 throw new \Exception('Setting option:'.self::optionName($option).' failed');
378             }
379             if (ENVIRONMENT === 'development') {
380                 $option = self::optionName($option);
381             }
382             $this->options[$option] = $value;
383         }
384 
385         // multi curl init
386         if (self::$pool === null) {
387             self::$pool = curl_multi_init();
388             if (self::$pool === false) {
389                 throw new \Exception('Failed to create cURL multi handle');
390             }
391             if (self::$keepalive) {
392                 register_shutdown_function(function () {
393                     cURL::$keepalive = false; // Close the multi handle when all requests are completed.
394                     cURL::synchronize();
395                 });
396             }
397         }
398         // Add request
399         $error = curl_multi_add_handle(self::$pool, $this->handle);
400         while ($error === CURLM_CALL_MULTI_PERFORM) {
401             $error = curl_multi_exec(self::$pool, $active);
402         }
403         if ($error !== CURLM_OK) {
404             throw new \Exception('['.self::multiErrorName($error).'] Failed to add cURL handle');
405         }
406         self::$tranferCount++;
407         // Start request
408         do {
409             $error = curl_multi_exec(self::$pool, $active);
410         } while ($error === CURLM_CALL_MULTI_PERFORM);
411         if ($error !== CURLM_OK) {
412             throw new \Exception('['.self::multiErrorName($error).'] Failed to execute cURL multi handle');
413         }
414         $this->state = 'RUNNING';
415     }

418      * Remove the request from the pool
419      *
420      * @throws \Exception
421      */
422     private function stop() {
423         if ($this->state !== 'ABORTED') {
424             $error = curl_multi_remove_handle(self::$pool, $this->handle);
425             self::$tranferCount--;
426             if ($error !== CURLM_OK) {
427                 throw new \Exception('['.self::multiErrorName($error).'] Failed to remove cURL handle');
428             }
429             if (self::$tranferCount === 0 && self::$keepalive === false) {
430                 curl_multi_close(self::$pool);
431                 self::$pool = null;
432             }
433         }
434         $this->state = 'ABORTED';
435     }
